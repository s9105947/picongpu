80  using SuperCellSize = typename T_Mapping::SuperCellSize;

82  // number of cells in a superCell
83  constexpr uint32_t numCells = pmacc::math::CT::volume< SuperCellSize >::type::value;
84  constexpr uint32_t numWorkers = T_numWorkers;
85  PMACC_CONSTEXPR_CAPTURE int dim = T_Mapping::Dim;

87  uint32_t const workerIdx = threadIdx.x;
88 
89  DataSpace< dim > const blockCell(
90  mapper.getSuperCellIndex( DataSpace< dim >( blockIdx ) ) *
91  SuperCellSize::toRT()
92  );
93 
94  // origin in area from local GPU
95  DataSpace< dim > nullSourceCell(
96  mapper.getSuperCellIndex( DataSpace< dim > () ) *
97  SuperCellSize::toRT()
98  );
99 
100  ForEachIdx<
101  IdxConfig<
102  numCells,
103  numWorkers
104  >
105  >{ workerIdx }(
106  [&](
107  uint32_t const linearIdx,
108  uint32_t const
109  )
110  {
111  // cell index within the superCell
112  DataSpace< dim > const cellIdx = DataSpaceOperations< dim >::template map< SuperCellSize >( linearIdx );
113 
114  DataSpace< T_Mapping::Dim > const sourceCell( blockCell + cellIdx );
115  DataSpace< dim > targetCell( sourceCell - nullSourceCell );
116 
117  /* defines if the virtual worker needs to copy the value of
118  * the cell to to the exchange box
119  */
120  bool copyValue = true;
121 
122  for( uint32_t d = 0; d < dim; ++d )
123  {
124  if( direction[ d ] == -1 )
125  {
126  if( cellIdx[ d ] < SuperCellSize::toRT()[ d ] - exchangeSize[ d ] )
127  copyValue = false;
128  targetCell[ d ] -= SuperCellSize::toRT()[ d ] - exchangeSize[ d ];
129  }
130  else if( direction[d] == 1 && cellIdx[ d ] >= exchangeSize[d] )
131  copyValue = false;
132  }
133 
134  if( copyValue )
135  exchangeBox( targetCell ) = srcBox( sourceCell );
136  }
137  );
138  }
