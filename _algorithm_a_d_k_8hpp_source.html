55  template<bool T_linPol>
56  struct AlgorithmADK
57  {
70  template<typename EType, typename BType, typename ParticleType >
71  HDINLINE uint32_t
72  operator()( const BType bField, const EType eField, ParticleType& parentIon, float_X randNr )
73  {
74 
75  float_X const protonNumber = GetAtomicNumbers<ParticleType>::type::numberOfProtons;
76  float_X const chargeState = attribute::getChargeState(parentIon);
77 
78  /* verify that ion is not completely ionized */
79  if( chargeState < protonNumber )
80  {
81  uint32_t const cs = math::float2int_rd(chargeState);
82  float_X const iEnergy = typename GetIonizationEnergies<ParticleType>::type{ }[cs];
83 
84  constexpr float_X pi = pmacc::algorithms::math::Pi< float_X >::value;
85  /* electric field in atomic units - only absolute value */
86  float_X const eInAU = math::abs( eField ) / ATOMIC_UNIT_EFIELD;
87 
88  /* the charge that attracts the electron that is to be ionized:
89  * equals `protonNumber - #allInnerElectrons`
90  */
91  float_X const effectiveCharge = chargeState + float_X( 1.0 );
92  /* effective principal quantum number (unitless) */
93  float_X const nEff = effectiveCharge / math::sqrt( float_X( 2.0 ) * iEnergy );
94  /* nameless variable for convenience dFromADK*/
95  float_X const dBase = float_X( 4.0 ) * util::cube( effectiveCharge ) /
96  ( eInAU * util::quad( nEff ) ) ;
97  float_X const dFromADK = math::pow( dBase, nEff );
98 
99  /* ionization rate (for CIRCULAR polarization)*/
100  float_X rateADK = eInAU * util::square( dFromADK ) /
101  ( float_X( 8.0 ) * pi * effectiveCharge ) *
102  math::exp( float_X( -2.0 ) * util::cube( effectiveCharge ) /
103  ( float_X( 3.0 ) * util::cube( nEff ) * eInAU )
104  );
105 
106  /* in case of linear polarization the rate is modified by an additional factor */
107  if( T_linPol )
108  {
109  /* factor from averaging over one laser cycle with LINEAR polarization */
110  float_X const polarizationFactor = math::sqrt(
111  float_X( 3.0 ) * util::cube( nEff ) * eInAU /
112  ( pi * util::cube( effectiveCharge ) )
113  );
114 
115  rateADK *= polarizationFactor;
116  }
117 
118  /* simulation time step in atomic units */
119  float_X const timeStepAU = float_X( DELTA_T / ATOMIC_UNIT_TIME );
120  /* ionization probability
121  *
122  * probability = rate * time step
123  * --> for infinitesimal time steps
124  *
125  * the whole ensemble should then follow
126  * P = 1 - exp(-rate * time step) if the laser wavelength is
127  * sampled well enough
128  */
129  float_X const probADK = rateADK * timeStepAU;
130 
131  /* ionization condition */
132  if( randNr < probADK )
133  {
134  /* return number of macro electrons to produce */
135  return 1u;
136  }
137  }
138  /* no ionization */
139  return 0u;
140  }
141  };
