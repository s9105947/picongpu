41 struct One_minus_beta_times_n
42 {
47 
48  // Taylor just includes a method, When includes just enum
49 
50  HDINLINE picongpu::float_32 operator()(const vector_64& n, const Particle & particle) const
51  {
52  // 1/gamma^2:
53 
54  const picongpu::float_64 gamma_inv_square(particle.get_gamma_inv_square<When::now > ());
55 
56  //picongpu::float_64 value; // storage for 1-\beta \times \vec n
57 
58  // if energy is high enough to cause numerical errors ( equals if 1/gamma^2 is close enough to zero)
59  // chose a Taylor approximation to to better calculate 1-\beta \times \vec n (which is close to 1-1)
60  // is energy is low, then the approximation will cause a larger error, therefor calculate
61  // 1-\beta \times \vec n directly
62  // with 0.18 the relative error will be below 0.001% for a Taylor series of 1-sqrt(1-x) of 5th order
63  if (gamma_inv_square < picongpu::GAMMA_INV_SQUARE_RAD_THRESH)
64  {
65  const picongpu::float_64 cos_theta(particle.get_cos_theta<When::now > (n)); // cosine between looking vector and momentum of particle
66  const picongpu::float_64 taylor_approx(cos_theta * Taylor()(gamma_inv_square) + (1.0 - cos_theta));
67  return (taylor_approx);
68  }
69  else
70  {
71  const vector_64 beta(particle.get_beta<When::now > ()); // calculate v/c=beta
72  return (1.0 - beta * n);
73  }
74 
75  }
76 };
