43  template< size_t T_direction>
44  template< class T_Particle >
45  DINLINE float_X
46  MomentumComponent<T_direction>::operator()( T_Particle& particle ) const
47  {
48  // read existing attributes
49  const float3_X mom = particle[momentum_];
50 
51  // calculate new attribute: |p| and p.[x|y|z]
52  const float_X momAbs = math::abs(mom);
53  const float_X momCom = mom[T_direction];
54 
55  // total momentum == 0 then perpendicular measure shall be zero, too
56  // values: [-1.:1.]
57  const float_X momComOverTotal = (momAbs > float_X(0.)) ?
58  momCom / momAbs :
59  float_X(0.);
60 
61  // return attribute
62  return momComOverTotal;
63  }
