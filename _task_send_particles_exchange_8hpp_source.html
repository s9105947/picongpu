58  bool executeIntern()
59  {
60  switch (state)
61  {
62  case Init:
63  break;
64  case WaitForBash:
65 
66  if (nullptr == Environment<>::get().Manager().getITaskIfNotFinished(tmpEvent.getTaskId()) &&
67  nullptr == Environment<>::get().Manager().getITaskIfNotFinished(lastSendEvent.getTaskId()))
68  {
69  state = InitSend;
70  //bash is finished
71  __startTransaction();
72  lastSize = parBase.getParticlesBuffer().getSendExchangeStack(exchange).getDeviceParticlesCurrentSize();
73  lastSendEvent = parBase.getParticlesBuffer().asyncSendParticles(__getTransactionEvent(), exchange);
74  initDependency = lastSendEvent;
75  __endTransaction();
76  state = WaitForSend;
77  }
78 
79  break;
80  case InitSend:
81  break;
82  case WaitForSend:
83  if (nullptr == Environment<>::get().Manager().getITaskIfNotFinished(tmpEvent.getTaskId()))
84  {
85  PMACC_ASSERT(lastSize <= maxSize);
86  //check for next bash round
87  if (lastSize == maxSize)
88  {
89  ++retryCounter;
90  init(); //call init and run a full send cycle
91 
92  }
93  else
94  state = WaitForSendEnd;
95  }
96  break;
97  case WaitForSendEnd:
98  if (nullptr == Environment<>::get().Manager().getITaskIfNotFinished(lastSendEvent.getTaskId()))
99  {
100  state = Finished;
101  return true;
102  }
103  break;
104  case Finished:
105  return true;
106  default:
107  return false;
108  }
109 
110  return false;
111  }
