92  HINLINE void operator()(ThreadParams* params, const uint32_t restartChunkSize)
93  {
94  std::string const speciesName = FrameType::getName();
95  log<picLog::INPUT_OUTPUT > ("ADIOS: (begin) load species: %1%") % speciesName;
96  DataConnector &dc = Environment<>::get().DataConnector();
97  GridController<simDim> &gc = Environment<simDim>::get().GridController();
98 
99  std::string particlePath = params->adiosBasePath + std::string(ADIOS_PATH_PARTICLES) +
100  speciesName + std::string("/");
101  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
102 
103  /* load particle without copying particle data to host */
104  auto speciesTmp = dc.get< ThisSpecies >( FrameType::getName(), true );
105 
106  /* count total number of particles on the device */
107  uint64_t totalNumParticles = 0;
108 
109  /* load particles info table entry for ONE process
110  (note: this is NOT necessarily THIS process!)
111  particlesInfo is (part-count, scalar pos, x, y, z) */
112  uint64_t particlesInfo[5];
113 
114  uint64_t start = 5 * gc.getGlobalRank();
115  uint64_t count = 5; // ADIOSCountParticles: uint64_t
116  ADIOS_SELECTION* piSel = adios_selection_boundingbox( 1, &start, &count );
117 
118  ADIOS_CMD(adios_schedule_read( params->fp,
119  piSel,
120  (particlePath + std::string("particles_info")).c_str(),
121  0,
122  1,
123  (void*)particlesInfo ));
124 
125  /* start a blocking read of all scheduled variables */
126  ADIOS_CMD(adios_perform_reads( params->fp, 1 ));
127  adios_selection_delete(piSel);
128 
129  /* Run a prefix sum over the numParticles[0] element in particlesInfo
130  * to retreive the offset of particles before gc.getGlobalRank() */
131  uint64_t particleOffset = 0;
132 
133  uint64_t fullParticlesInfo[gc.getGlobalSize()];
134 
135  MPI_CHECK(MPI_Allgather( particlesInfo, 1, MPI_UINT64_T,
136  fullParticlesInfo, 1, MPI_UINT64_T,
137  gc.getCommunicator().getMPIComm() ));
138 
139  for (size_t i = 0; i < gc.getGlobalSize(); ++i)
140  {
141  /* this comparison is potentially harmful, since the order of ranks
142  is not necessarily the same in subsequent MPI jobs.
143  But due to the wrong sorting by rank in `ADIOSCountParticles.hpp`
144  while calculating the `myParticleOffset` we have to immitate that. */
145  if( i < gc.getGlobalRank() )
146  particleOffset += fullParticlesInfo[i];
147  if( i == gc.getGlobalRank() )
148  totalNumParticles = fullParticlesInfo[i];
149  }
150 
151  log<picLog::INPUT_OUTPUT > ("ADIOS: Loading %1% particles from offset %2%") %
152  (long long unsigned) totalNumParticles % (long long unsigned) particleOffset;
153 
154  AdiosFrameType hostFrame;
155  log<picLog::INPUT_OUTPUT > ("ADIOS: malloc mapped memory: %1%") % speciesName;
156  /*malloc mapped memory*/
157  ForEach<typename AdiosFrameType::ValueTypeSeq, MallocMemory<bmpl::_1> > mallocMem;
158  mallocMem(forward(hostFrame), totalNumParticles);
159 
160  log<picLog::INPUT_OUTPUT > ("ADIOS: get mapped memory device pointer: %1%") % speciesName;
161  /*load device pointer of mapped memory*/
162  AdiosFrameType deviceFrame;
163  ForEach<typename AdiosFrameType::ValueTypeSeq, GetDevicePtr<bmpl::_1> > getDevicePtr;
164  getDevicePtr(forward(deviceFrame), forward(hostFrame));
165 
166  ForEach<typename AdiosFrameType::ValueTypeSeq, LoadParticleAttributesFromADIOS<bmpl::_1> > loadAttributes;
167  loadAttributes(forward(params), forward(hostFrame), particlePath, particleOffset, totalNumParticles);
168 
169  if (totalNumParticles != 0)
170  {
171  pmacc::particles::operations::splitIntoListOfFrames(
172  *speciesTmp,
173  deviceFrame,
174  totalNumParticles,
175  restartChunkSize,
176  localDomain.offset,
177  totalCellIdx_,
178  *(params->cellDescription),
179  picLog::INPUT_OUTPUT()
180  );
181 
182  /*free host memory*/
183  ForEach<typename AdiosFrameType::ValueTypeSeq, FreeMemory<bmpl::_1> > freeMem;
184  freeMem(forward(hostFrame));
185  }
186  log<picLog::INPUT_OUTPUT > ("ADIOS: ( end ) load species: %1%") % speciesName;
187  }
188 };
