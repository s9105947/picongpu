107  // verbose output of loaded frequencies if verbose level PHYSICS is set:
108  log<PIConGPUVerboseRadiation::PHYSICS >("freq: %1% \t %2%") % i % frequencyDB[i];
109  frequencyDB[i] *= UNIT_TIME;
110  }
111
