54  bool executeIntern()
55  {
56  switch (state)
57  {
58  case WaitForReceived:
59  break;
60  case RunCopy:
61  state = WaitForFinish;
62  __startTransaction();
63  exchange->getHostBuffer().setCurrentSize(newBufferSize);
64  if (exchange->hasDeviceDoubleBuffer())
65  {
66 
67  Environment<>::get().Factory().createTaskCopyHostToDevice(exchange->getHostBuffer(),
68  exchange->getDeviceDoubleBuffer());
69  Environment<>::get().Factory().createTaskCopyDeviceToDevice(exchange->getDeviceDoubleBuffer(),
70  exchange->getDeviceBuffer(),
71  this);
72  }
73  else
74  {
75 
76  Environment<>::get().Factory().createTaskCopyHostToDevice(exchange->getHostBuffer(),
77  exchange->getDeviceBuffer(),
78  this);
79  }
80  __endTransaction();
81  break;
82  case WaitForFinish:
83  break;
84  case Finish:
85  return true;
86  default:
87  return false;
88  }
89 
90  return false;
91  }

98  void event(id_t, EventType type, IEventData* data)

108  state = RunCopy;
