42 HINLINE float3_X laserLongitudinal(uint32_t currentStep, float_X& phase)
43 {
44  float_X envelope = float_X(AMPLITUDE);
45  float3_X elong(float3_X::create(0.0));
46 
47  // a symmetric pulse will be initialized at position z=0 for
48  // a time of RAMP_INIT * PULSE_LENGTH + LASER_NOFOCUS_CONSTANT = INIT_TIME.
49  // we shift the complete pulse for the half of this time to start with
50  // the front of the laser pulse.
51  const float_64 mue = 0.5 * INIT_TIME;
52 
53  /* initialize the laser not in the first cell is equal to a negative shift
54  * in time
55  */
56  constexpr float_X laserTimeShift = laser::initPlaneY * CELL_HEIGHT / SPEED_OF_LIGHT;
57  const float_64 runTime = (DELTA_T * currentStep - laserTimeShift) - mue;
58  const float_64 f = SPEED_OF_LIGHT / WAVE_LENGTH;
59 
60  const float_64 w = 2.0 * PI * f;
61 
62  const float_64 endUpramp = -0.5 * LASER_NOFOCUS_CONSTANT;
63  const float_64 startDownramp = 0.5 * LASER_NOFOCUS_CONSTANT;
64 
65  const float_64 tau = PULSE_LENGTH * sqrt(2.0);
66 
67  float_64 correctionFactor = 0.0;
68 
69  if (runTime > startDownramp)
70  {
71  // downramp = end
72  const float_64 exponent =
73  ((runTime - startDownramp)
74  / PULSE_LENGTH / sqrt(2.0));
75  envelope *= math::exp(-0.5 * exponent * exponent);
76  correctionFactor = (runTime - startDownramp)/(tau*tau*w);
77  }
78  else if(runTime < endUpramp)
79  {
80  // upramp = start
81  const float_64 exponent = ((runTime - endUpramp) / PULSE_LENGTH / sqrt(2.0));
82  envelope *= math::exp(-0.5 * exponent * exponent);
83  correctionFactor = (runTime - endUpramp)/(tau*tau*w);
84  }
85 
86  phase += float_X(w * runTime) + LASER_PHASE;
87 
88  if( Polarisation == LINEAR_X )
89  {
90  elong.x() = float_X(envelope * (math::sin(phase) + correctionFactor * math::cos(phase)));
91  }
92  else if( Polarisation == LINEAR_Z )
93  {
94  elong.z() = float_X(envelope * (math::sin(phase) + correctionFactor * math::cos(phase)));
95  }
96  else if( Polarisation == CIRCULAR )
97  {
98  elong.x() = float_X(envelope / sqrt(2.0) * (math::sin(phase) + correctionFactor * math::cos(phase)));
99  elong.z() = float_X(envelope / sqrt(2.0) * (math::cos(phase) + correctionFactor * math::sin(phase)));
100  }
101 
102  return elong;
103 }
