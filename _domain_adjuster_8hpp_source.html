129  void updateLocalDomainOffset( size_t const dim )
130  {
131  pmacc::GridController< simDim > & gc = pmacc::Environment< simDim >::get( ).GridController( );
132 
133  int mpiPos( gc.getPosition( )[ dim ] );
134  int numMpiRanks = gc.getGlobalSize();
135 
136  // gather mpi position in the direction we are checking
137  std::vector< int > mpiPositions( numMpiRanks );
138  MPI_CHECK( MPI_Allgather(
139  &mpiPos,
140  1,
141  MPI_INT,
142  mpiPositions.data(),
143  1,
144  MPI_INT,
145  gc.getCommunicator().getMPIComm()
146  ));
147 
148  // gather local sizes in the direction we are checking
149  std::vector< uint64_t > allLocalSizes( numMpiRanks );
150  uint64_t lSize = static_cast< uint64_t >( m_localDomainSize[ dim ] );
151  MPI_CHECK( MPI_Allgather(
152  &lSize,
153  1,
154  MPI_UINT64_T,
155  allLocalSizes.data(),
156  1,
157  MPI_UINT64_T,
158  gc.getCommunicator().getMPIComm()
159  ));
160 
161  uint64_t offset = 0u;
162  for( size_t i = 0u; i < mpiPositions.size(); ++i )
163  {
164  if( mpiPositions[ i ] < mpiPos )
165  offset += allLocalSizes[ i ];
166  }
167 
168  /* since we are not doing independent reduces per slice we need
169  * to adjust the offset result by dividing with the number of
170  * MPI ranks in all other dimensions.
171  */
172  offset /= static_cast< uint64_t >( m_numDevices.productOfComponents() / m_numDevices[ dim ] );
173  m_localDomainOffset[ dim ] = static_cast< int >( offset );
174 
175  }

181  void multipleOfSuperCell( size_t const dim )
182  {
183  int const sCellSize = SuperCellSize::toRT()[ dim ];
184  // round up to full supercells
185  int const validLocalSize =
186  ( ( m_localDomainSize[ dim ] + sCellSize - 1 ) / sCellSize ) *
187  sCellSize;

233  void greaterEqualThanAbsorber( size_t const dim )
234  {
235  int validLocalSize = m_localDomainSize[ dim ];
236 
237  bool const isAbsorberEnabled = !m_isPeriodic[ dim ];
238  bool const isBoundaryDevice = ( m_mpiPosition[ dim ] == 0 || m_mpiPosition[ dim ] == m_numDevices[ dim ] - 1 );
239  if( isAbsorberEnabled && isBoundaryDevice )
240  {
241  size_t boundary = m_mpiPosition[ dim ] == 0u ? 0u : 1u;
242  int maxAbsorberCells = ABSORBER_CELLS[ dim ][ boundary ];
243 
244  if( m_movingWindowEnabled && dim == 1u )
245  {
246  /* since the device changes their position during the simulation
247  * the negative and positive absorber cells must fit into the domain
248  */
249  maxAbsorberCells = static_cast< int >(
250  std::max(
251  ABSORBER_CELLS[ dim ][ 0 ],
252  ABSORBER_CELLS[ dim ][ 1 ]
253  )
254  );
255  }
256 
257  if( m_localDomainSize[ dim ] < maxAbsorberCells )
258  {
259  int const sCellSize = SuperCellSize::toRT()[ dim ];
260  // round up to full supercells
261  validLocalSize =
262  ( ( maxAbsorberCells + sCellSize - 1 ) / sCellSize ) *
263  sCellSize;
264  }
265 
266  if( validLocalSize != m_localDomainSize[ dim ] )
267  {
268  showMessage(
269  dim,
270  "Local grid size must be greater or equal than the largest absorber.",
271  m_localDomainSize[ dim ],
272  validLocalSize
273  );
274 
275  m_localDomainSize[ dim ] = validLocalSize;
276  }
277  }

288  void deriveLocalDomainSize( size_t const dim )
289  {
290  if( m_movingWindowEnabled && dim == 1u )
291  {
292 
293  pmacc::mpi::MPIReduce mpiReduce;
294

302  );

306  pmacc::nvidia::functors::Min(),
307  &globalMin,

309  1,

311  );

313  // local size must be equal for all devices in y direction
314  if( m_isMaster && globalMax != globalMin )
315  {

319  m_localDomainSize[ dim ],
320  globalMax
321  );
322  }

334  void deriveGlobalDomainSize( size_t const dim )
335  {
336  uint64_t validGlobalGridSize = 0u;
337 
338  deriveLocalDomainSize( dim );
339 
340  if( m_movingWindowEnabled && dim == 1u )
341  {
342  // the local sizes in slide direction must be equal sized
343  validGlobalGridSize = static_cast< uint64_t >( m_localDomainSize[ dim ] * m_numDevices[ dim ] );
344  }
345  else
346  {
347  uint64_t localDomainSize = static_cast< uint64_t >( m_localDomainSize[ dim ] );
348  pmacc::mpi::MPIReduce mpiReduce;
349  mpiReduce(
350  pmacc::nvidia::functors::Add(),
351  &validGlobalGridSize,
352  &localDomainSize,
353  1,
354  pmacc::mpi::reduceMethods::AllReduce()
355  );
356  /* since we are not doing independent reduces per slice we need
357  * to adjust the reduce result by dividing the sizes of all other dimensions
358  * we are not check within the method call
359  */
360  validGlobalGridSize /= static_cast< uint64_t >(
361  m_numDevices.productOfComponents() / m_numDevices[ dim ]
362  );
363 
364  }
365 
366  if( m_isMaster && validGlobalGridSize != static_cast< uint64_t >( m_globalDomainSize[ dim ] ) )
367  {
368  showMessage(
369  dim,
370  "Invalid global grid size.",
371  m_globalDomainSize[ dim ],
372  validGlobalGridSize
373  );
374  }
375 
376  m_globalDomainSize[ dim ] = static_cast< int >( validGlobalGridSize );
377  }
