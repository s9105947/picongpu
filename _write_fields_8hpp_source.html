133 template< typename Solver, typename Species >
134 class WriteFields<FieldTmpOperation<Solver, Species> >
135 {
136 public:
137  /*
138  * This is only a wrapper function to allow disable nvcc warnings.
139  * Warning: calling a __host__ function from __host__ __device__
140  * function.
141  * Use of PMACC_NO_NVCC_HDWARNING is not possible if we call a virtual
142  * method inside of the method were we disable the warnings.
143  * Therefore we create this method and call a new method were we can
144  * call virtual functions.
145  */
146  PMACC_NO_NVCC_HDWARNING
147  HDINLINE void operator()(ThreadParams* tparam)
148  {
149  this->operator_impl(tparam);
150  }
151 
152 private:
153  typedef typename FieldTmp::ValueType ValueType;
154 
157  static std::string getName()
158  {
159  return FieldTmpOperation<Solver, Species>::getName();
160  }
161 
163  static std::vector<float_64> getUnit()
164  {
165  typedef typename FieldTmp::UnitValueType UnitType;
166  UnitType unit = FieldTmp::getUnit<Solver>();
167  const uint32_t components = GetNComponents<ValueType>::value;
168  return CreateUnit::createUnit(unit, components);
169  }
170 
171  HINLINE void operator_impl(ThreadParams* params)
172  {
173  DataConnector &dc = Environment<>::get().DataConnector();
174 
175  /*## update field ##*/
176 
177  /*load FieldTmp without copy data to host*/
178  PMACC_CASSERT_MSG(
179  _please_allocate_at_least_one_FieldTmp_in_memory_param,
180  fieldTmpNumSlots > 0
181  );
182  auto fieldTmp = dc.get< FieldTmp >( FieldTmp::getUniqueId( 0 ), true );
183  /*load particle without copy particle data to host*/
184  auto speciesTmp = dc.get< Species >( Species::FrameType::getName(), true );
185 
186  fieldTmp->getGridBuffer().getDeviceBuffer().setValue(ValueType::create(0.0));
187  /*run algorithm*/
188  fieldTmp->template computeValue< CORE + BORDER, Solver >(*speciesTmp, params->currentStep);
189 
190  EventTask fieldTmpEvent = fieldTmp->asyncCommunication(__getTransactionEvent());
191  __setTransactionEvent(fieldTmpEvent);
192  /* copy data to host that we can write same to disk*/
193  fieldTmp->getGridBuffer().deviceToHost();
194  dc.releaseData( Species::FrameType::getName() );
195  /*## finish update field ##*/
196 
197  /*wrap in a one-component vector for writeField API*/
198  const traits::FieldPosition<fields::CellType, FieldTmp>
199  fieldPos;
200 
201  std::vector<std::vector<float_X> > inCellPosition;
202  std::vector<float_X> inCellPositonComponent;
203  for( uint32_t d = 0; d < simDim; ++d )
204  inCellPositonComponent.push_back( fieldPos()[0][d] );
205  inCellPosition.push_back( inCellPositonComponent );
206 
209  const float_X timeOffset = 0.0;
210 
211  params->gridLayout = fieldTmp->getGridLayout();
212  const bool isDomainBound = traits::IsFieldDomainBound< FieldTmp >::value;
213  /*write data to HDF5 file*/
214  Field::writeField(
215  params,
216  getName(),
217  getUnit(),
218  FieldTmp::getUnitDimension<Solver>(),
219  inCellPosition,
220  timeOffset,
221  fieldTmp->getHostDataBox(),
222  ValueType(),
223  isDomainBound
224  );
225 
226  dc.releaseData( FieldTmp::getUniqueId( 0 ) );
227 
228  }

230 };
