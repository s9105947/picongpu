68 template<uint64_t lvl_, class membership_>
69 struct LogLvl
70 {
71  typedef membership_ Parent;
72  static constexpr uint64_t lvl = lvl_;
73 
74  /* This operation is only allowed for LogLvl with the same Parent type.
75  * Create a LogLvl that contains two levels. At least one lvl has to be true
76  */
77  template<class OtherLogLvl >
78  LogLvl < (OtherLogLvl::lvl | lvl), membership_> operator+(const OtherLogLvl&)
79  {
80  return LogLvl < (OtherLogLvl::lvl | lvl), membership_ > ();
81  }
82 
83 };

88 template<class LogLevel>
89 class VerboseLog
90 {
91 private:
92  typedef typename LogLevel::Parent LogParent;
93  static constexpr uint64_t logLvl = LogLevel::lvl;
94 public:
95 
96  VerboseLog(const char* msg) : fmt(msg)
97  {
98  }
99 
100  ~VerboseLog()
101  {
102  typedef LogLvl<(logLvl & LogParent::log_level), LogParent> LogClass;
103  /* check if a bit in the mask is set
104  * If you get an linker error in the next two lines you have not used
105  * DEFINE_LOGLVL makro to define a named logLvl
106  */
107  if (logLvl & LogParent::log_level) /*compiletime check*/
108  {
109  std::cout << LogParent::getName() << " " << getLogName(LogClass()) <<
110  "(" << (logLvl & LogParent::log_level) << ")" << " | " << fmt << std::endl;
111  }
112  }
113 
114  template <typename T>
115  VerboseLog& operator %(T value)
116  {
117  if (logLvl & LogParent::log_level) /*compiletime check*/
118  fmt % value;
119  return *this;
120  }
121 
122 protected:
123  boost::format fmt;
124 };
