95  constexpr bool isSupportEven = ( supp % 2 == 0 );
96  relayPoint[d] = RelayPoint< isSupportEven >()(

101  );
102  }
103 
104  Line< floatD_X > line;
105  const float_X chargeDensity = charge / CELL_VOLUME;

107  /* Esirkepov implementation for the current deposition */

114  /* calculate positions for the second virtual particle */
115  for (uint32_t d = 0; d < simDim; ++d)

117  line.m_pos0[d] = calc_InCellPos(
118  posStart[d],

128 
129  deposit(

135  );
136

141  for (uint32_t d = 0; d < simDim; ++d)
142  {
143  /* switched start and end point */
144  line.m_pos1[d] = calc_InCellPos(
145  posEnd[d],
