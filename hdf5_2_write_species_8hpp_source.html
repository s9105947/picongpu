141 
142

144  typedef bmpl::vector<multiMask,localCellIdx> TypesToDelete;
145  typedef typename RemoveFromSeq<ParticleAttributeList, TypesToDelete>::type ParticleCleanedAttributeList;
146 
147  /* add totalCellIdx for hdf5 particle*/
148  typedef typename MakeSeq<
149  ParticleCleanedAttributeList,

152

162  template<typename Space>
163  HINLINE void operator()(ThreadParams* params,
164  const Space domainOffset)
165  {
166  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) write species: %1%") % T_SpeciesFilter::getName();
167  DataConnector &dc = Environment<>::get().DataConnector();
168  /* load particle without copy particle data to host */
169  auto speciesTmp = dc.get< ThisSpecies >( ThisSpecies::FrameType::getName(), true );
170 
171  /* count number of particles for this species on the device */
172  uint64_t numParticles = 0;
173 
174  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) count particles: %1%") % T_SpeciesFilter::getName();
175 
176  // enforce that the filter interface is fulfilled
177  particles::filter::IUnary< typename T_SpeciesFilter::Filter > particleFilter{ params->currentStep };
178  /* at this point we cast to uint64_t, before we assume that per device
179  * less then 1e9 (int range) particles will be counted
180  */
181  numParticles = uint64_t( pmacc::CountParticles::countOnDevice< CORE + BORDER >(
182  *speciesTmp,
183  *(params->cellDescription),
184  params->localWindowToDomainOffset,
185  params->window.localDimensions.size,
186  particleFilter
187  ));
188 
189 
190  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) count particles: %1% = %2%") % T_SpeciesFilter::getName() % numParticles;
191  Hdf5FrameType hostFrame;
192  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) malloc mapped memory: %1%") % T_SpeciesFilter::getName();
193  /*malloc mapped memory*/
194  ForEach<typename Hdf5FrameType::ValueTypeSeq, MallocMemory<bmpl::_1> > mallocMem;
195  mallocMem(forward(hostFrame), numParticles);
196  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) malloc mapped memory: %1%") % T_SpeciesFilter::getName();
197 
198  if (numParticles != 0)
199  {
200 
201  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) get mapped memory device pointer: %1%") % T_SpeciesFilter::getName();
202  /*load device pointer of mapped memory*/
203  Hdf5FrameType deviceFrame;
204  ForEach<typename Hdf5FrameType::ValueTypeSeq, GetDevicePtr<bmpl::_1> > getDevicePtr;
205  getDevicePtr(forward(deviceFrame), forward(hostFrame));
206  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) get mapped memory device pointer: %1%") % T_SpeciesFilter::getName();
207 
208  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) copy particle to host: %1%") % T_SpeciesFilter::getName();
209  typedef bmpl::vector< typename GetPositionFilter<simDim>::type > usedFilters;
210  typedef typename FilterFactory<usedFilters>::FilterType MyParticleFilter;
211  MyParticleFilter filter;
212  /* activate filter pipeline if moving window is activated */
213  filter.setStatus(MovingWindow::getInstance().isEnabled());
214  filter.setWindowPosition(params->localWindowToDomainOffset,
215  params->window.localDimensions.size);
216 
217  /* int: assume < 2e9 particles per device */
218  GridBuffer<int, DIM1> counterBuffer(DataSpace<DIM1>(1));
219  AreaMapping < CORE + BORDER, MappingDesc > mapper(*(params->cellDescription));
220 
221  constexpr uint32_t numWorkers = pmacc::traits::GetNumWorkers<
222  pmacc::math::CT::volume< SuperCellSize >::type::value
223  >::value;
224 
225  /* this sanity check costs a little bit of time but hdf5 writing is slower */
226  PMACC_KERNEL( CopySpecies< numWorkers >{} )(
227  mapper.getGridDim(),
228  numWorkers
229  )(
230  counterBuffer.getDeviceBuffer().getPointer(),
231  deviceFrame, speciesTmp->getDeviceParticlesBox(),
232  filter,
233  domainOffset,
234  totalCellIdx_,
235  mapper,
236  particleFilter
237  );
238  counterBuffer.deviceToHost();
239  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) copy particle to host: %1%") % T_SpeciesFilter::getName();
240  __getTransactionEvent().waitForFinished();
241  log<picLog::INPUT_OUTPUT > ("HDF5: all events are finished: %1%") % T_SpeciesFilter::getName();
242 
243  PMACC_ASSERT((uint64_t) counterBuffer.getHostBuffer().getDataBox()[0] == numParticles);
244  }
245 
246  /* We rather do an allgather at this point then letting libSplash
247  * do an allgather during write to find out the global number of
248  * particles.
249  */
250  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) collect particle sizes for %1%") % T_SpeciesFilter::getName();
251 
252  ColTypeUInt64 ctUInt64;
253  ColTypeDouble ctDouble;
254  GridController<simDim>& gc = Environment<simDim>::get().GridController();
255 
256  const uint64_t numRanks( gc.getGlobalSize() );
257  const uint64_t myRank( gc.getGlobalRank() );
258 
259  /* For collective write calls we need the information:
260  * - how many particles will be written globally
261  * - what is my particle offset within this global data set
262  *
263  * interleaved in array:
264  * numParticles for mpi rank, mpi rank
265  *
266  * the mpi rank is an arbitrary quantity and might change after a
267  * restart, but we only use it to order our patches and offsets
268  */
269  std::vector<uint64_t> particleCounts( 2 * numRanks, 0u );
270  uint64_t myParticlePatch[ 2 ];
271  myParticlePatch[ 0 ] = numParticles;
272  myParticlePatch[ 1 ] = myRank;
273 
274  /* we do the scan over MPI ranks since it does not matter how the
275  * global rank or scalar position (which are not idential) are
276  * ordered as long as the particle attributes are also written in
277  * the same order (which is by global rank) */
278  uint64_t numParticlesOffset = 0;
279  uint64_t numParticlesGlobal = 0;
280 
281  // avoid deadlock between not finished pmacc tasks and mpi blocking collectives
282  __getTransactionEvent().waitForFinished();
283  MPI_CHECK(MPI_Allgather(
284  myParticlePatch, 2, MPI_UINT64_T,
285  &(*particleCounts.begin()), 2, MPI_UINT64_T,
286  gc.getCommunicator().getMPIComm()
287  ));
288 
289  for( uint64_t r = 0; r < numRanks; ++r )
290  {
291  numParticlesGlobal += particleCounts.at(2 * r);
292  if( particleCounts.at(2 * r + 1) < myParticlePatch[ 1 ] )
293  numParticlesOffset += particleCounts.at(2 * r);
294  }
295  log<picLog::INPUT_OUTPUT > ("HDF5: (end) collect particle sizes for %1%") % T_SpeciesFilter::getName();
296 
297  /* dump non-constant particle records to hdf5 file */
298  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) write particle records for %1%") % T_SpeciesFilter::getName();
299 
300  const std::string speciesPath( std::string("particles/") + T_SpeciesFilter::getName() );
301 
302  ForEach<typename Hdf5FrameType::ValueTypeSeq, hdf5::ParticleAttribute<bmpl::_1> > writeToHdf5;
303  writeToHdf5(
304  params,
305  forward(hostFrame),
306  speciesPath,
307  numParticles,
308  numParticlesOffset,
309  numParticlesGlobal
310  );
311 
312  /* write constant particle records to hdf5 file
313  * ions with variable charge due to a boundElectrons attribute do not write charge
314  */
315  using hasBoundElectrons = typename pmacc::traits::HasIdentifier<
316  FrameType,
317  boundElectrons
318  >::type;
319  detail::GetChargeOrZero< FrameType > const getChargeOrZero;
320  if( ! hasBoundElectrons::value && getChargeOrZero.hasChargeRatio )
321  {
322  const float_64 charge( getChargeOrZero() );
323  std::vector<float_64> chargeUnitDimension( NUnitDimension, 0.0 );
324  chargeUnitDimension.at(SIBaseUnits::time) = 1.0;
325  chargeUnitDimension.at(SIBaseUnits::electricCurrent) = 1.0;
326 
327  writeConstantRecord(
328  params,
329  speciesPath + std::string("/charge"),
330  numParticlesGlobal,
331  charge,
332  UNIT_CHARGE,
333  chargeUnitDimension
334  );
335  }
336 
337  detail::GetMassOrZero< FrameType > const getMassOrZero;
338  if( getMassOrZero.hasMassRatio )
339  {
340  const float_64 mass( getMassOrZero() );
341  std::vector<float_64> massUnitDimension( NUnitDimension, 0.0 );
342  massUnitDimension.at(SIBaseUnits::mass) = 1.0;
343 
344  writeConstantRecord(
345  params,
346  speciesPath + std::string("/mass"),
347  numParticlesGlobal,
348  mass,
349  UNIT_MASS,
350  massUnitDimension
351  );
352  }
353 
354  /* openPMD ED-PIC: write additional attributes */
355  const float_64 particleShape( GetShape<ThisSpecies>::type::support - 1 );
356  params->dataCollector->writeAttribute( params->currentStep,
357  ctDouble,
358  speciesPath.c_str(),
359  "particleShape",
360  &particleShape );
361 
362  traits::GetSpeciesFlagName<ThisSpecies, current<> > currentDepositionName;
363  const std::string currentDeposition( currentDepositionName() );
364  ColTypeString ctCurrentDeposition( currentDeposition.length() );
365  params->dataCollector->writeAttribute( params->currentStep,
366  ctCurrentDeposition,
367  speciesPath.c_str(),
368  "currentDeposition",
369  currentDeposition.c_str() );
370 
371  traits::GetSpeciesFlagName<ThisSpecies, particlePusher<> > particlePushName;
372  const std::string particlePush( particlePushName() );
373  ColTypeString ctParticlePush( particlePush.length() );
374  params->dataCollector->writeAttribute( params->currentStep,
375  ctParticlePush,
376  speciesPath.c_str(),
377  "particlePush",
378  particlePush.c_str() );
379 
380  traits::GetSpeciesFlagName<ThisSpecies, interpolation<> > particleInterpolationName;
381  const std::string particleInterpolation( particleInterpolationName() );
382  ColTypeString ctParticleInterpolation( particleInterpolation.length() );
383  params->dataCollector->writeAttribute( params->currentStep,
384  ctParticleInterpolation,
385  speciesPath.c_str(),
386  "particleInterpolation",
387  particleInterpolation.c_str() );
388 
389  const std::string particleSmoothing("none");
390  ColTypeString ctParticleSmoothing(particleSmoothing.length());
391  params->dataCollector->writeAttribute( params->currentStep,
392  ctParticleSmoothing,
393  speciesPath.c_str(),
394  "particleSmoothing",
395  particleSmoothing.c_str() );
396 
397  log<picLog::INPUT_OUTPUT > ("HDF5: (end) write particle records for %1%") % T_SpeciesFilter::getName();
398 
399  /* write species particle patch meta information */
400  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) writing particlePatches for %1%") % T_SpeciesFilter::getName();
401 
402  std::string particlePatchesPath( speciesPath + std::string("/particlePatches") );
403 
404  /* offset and size of our particle patches
405  * - numPatches: we write as many patches as MPI ranks
406  * - myPatchOffset: we write in the order of the MPI ranks
407  * - myPatchEntries: every MPI rank writes exactly one patch
408  */
409  const Dimensions numPatches( numRanks, 1, 1 );
410  const Dimensions myPatchOffset( myRank, 0, 0 );
411  const Dimensions myPatchEntries( 1, 1, 1 );
412 
413  /* numParticles: number of particles in this patch */
414  params->dataCollector->write(
415  params->currentStep,
416  numPatches,
417  myPatchOffset,
418  ctUInt64, 1,
419  myPatchEntries,
420  (particlePatchesPath + std::string("/numParticles")).c_str(),
421  &numParticles);
422 
423  /* numParticlesOffset: number of particles before this patch */
424  params->dataCollector->write(
425  params->currentStep,
426  numPatches,
427  myPatchOffset,
428  ctUInt64, 1,
429  myPatchEntries,
430  (particlePatchesPath + std::string("/numParticlesOffset")).c_str(),
431  &numParticlesOffset);
432 
433  /* offset: absolute position where this particle patch begins including
434  * global domain offsets (slides), etc.
435  * extent: size of this particle patch, upper bound is excluded
436  */
437  const pmacc::Selection<simDim>& globalDomain = Environment<simDim>::get().SubGrid().getGlobalDomain();
438  const std::string name_lookup[] = {"x", "y", "z"};
439  for (uint32_t d = 0; d < simDim; ++d)
440  {
441  const uint64_t patchOffset =
442  globalDomain.offset[d] +
443  params->window.globalDimensions.offset[d] +
444  params->window.localDimensions.offset[d];
445  const uint64_t patchExtent =
446  params->window.localDimensions.size[d];
447 
448  params->dataCollector->write(
449  params->currentStep,
450  numPatches,
451  myPatchOffset,
452  ctUInt64, 1,
453  myPatchEntries,
454  (particlePatchesPath + std::string("/offset/") +
455  name_lookup[d]).c_str(),
456  &patchOffset);
457  params->dataCollector->write(
458  params->currentStep,
459  numPatches,
460  myPatchOffset,
461  ctUInt64, 1,
462  myPatchEntries,
463  (particlePatchesPath + std::string("/extent/") +
464  name_lookup[d]).c_str(),
465  &patchExtent);
466 
467  /* offsets and extent of the patch are positions (lengths)
468  * and need to be scaled like the cell idx of a particle
469  */
470  OpenPMDUnit<totalCellIdx> openPMDUnitCellIdx;
471  std::vector<float_64> unitCellIdx = openPMDUnitCellIdx();
472 
473  params->dataCollector->writeAttribute(
474  params->currentStep,
475  ctDouble,
476  (particlePatchesPath + std::string("/offset/") +
477  name_lookup[d]).c_str(),
478  "unitSI",
479  &(unitCellIdx.at(d)));
480  params->dataCollector->writeAttribute(
481  params->currentStep,
482  ctDouble,
483  (particlePatchesPath + std::string("/extent/") +
484  name_lookup[d]).c_str(),
485  "unitSI",
486  &(unitCellIdx.at(d)));
487  }
488 
489  OpenPMDUnitDimension<totalCellIdx> openPMDUnitDimension;
490  std::vector<float_64> unitDimensionCellIdx = openPMDUnitDimension();
491 
492  params->dataCollector->writeAttribute(
493  params->currentStep,
494  ctDouble,
495  (particlePatchesPath + std::string("/offset")).c_str(),
496  "unitDimension",
497  1u, Dimensions(7,0,0),
498  &(*unitDimensionCellIdx.begin()));
499  params->dataCollector->writeAttribute(
500  params->currentStep,
501  ctDouble,
502  (particlePatchesPath + std::string("/extent")).c_str(),
503  "unitDimension",
504  1u, Dimensions(7,0,0),
505  &(*unitDimensionCellIdx.begin()));
506 
507 
508  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) writing particlePatches for %1%") % T_SpeciesFilter::getName();
509 
510  /*free host memory*/
511  ForEach<typename Hdf5FrameType::ValueTypeSeq, FreeMemory<bmpl::_1> > freeMem;
512  freeMem(forward(hostFrame));
513  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) writing species: %1%") % T_SpeciesFilter::getName();
514  }
