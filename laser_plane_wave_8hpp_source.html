63  HINLINE float3_X laserLongitudinal( uint32_t currentStep, float_X& phase )
64  {
65  /* initialize the laser not in the first cell is equal to a negative shift
66  * in time
67  */
68  constexpr float_X laserTimeShift = laser::initPlaneY * CELL_HEIGHT / SPEED_OF_LIGHT;
69  const float_64 runTime = DELTA_T * currentStep - laserTimeShift;
70  const float_64 f = SPEED_OF_LIGHT / WAVE_LENGTH;
71 
72  float_64 envelope = float_64(AMPLITUDE );
73  float3_X elong(float3_X::create(0.0));
74 
75  const float_64 mue = 0.5 * RAMP_INIT * PULSE_LENGTH;
76 
77  const float_64 w = 2.0 * PI * f;
78  const float_64 tau = PULSE_LENGTH * sqrt( 2.0 );
79 
80  const float_64 endUpramp = mue;
81  const float_64 startDownramp = mue + LASER_NOFOCUS_CONSTANT;
82 
83  float_64 integrationCorrectionFactor = 0.0;
84 
85  if( runTime > startDownramp )
86  {
87  // downramp = end
88  const float_64 exponent = (runTime - startDownramp) / tau;
89  envelope *= exp( -0.5 * exponent * exponent );
90  integrationCorrectionFactor = ( runTime - startDownramp )/ (w*tau*tau);
91  }
92  else if ( runTime < endUpramp )
93  {
94  // upramp = start
95  const float_64 exponent = (runTime - endUpramp) / tau;
96  envelope *= exp( -0.5 * exponent * exponent );
97  integrationCorrectionFactor = ( runTime - endUpramp )/ (w*tau*tau);
98  }
99 
100  const float_64 timeOszi = runTime - endUpramp;
101  const float_64 t_and_phase = w * timeOszi + LASER_PHASE;
102  // to understand both components [sin(...) + t/tau^2 * cos(...)] see description above
103  if( Polarisation == LINEAR_X )
104  {
105  elong.x() = float_X( envelope * (math::sin(t_and_phase)
106  + math::cos(t_and_phase) * integrationCorrectionFactor));
107  }
108  else if( Polarisation == LINEAR_Z)
109  {
110  elong.z() = float_X( envelope * (math::sin(t_and_phase)
111  + math::cos(t_and_phase) * integrationCorrectionFactor));
112  }
113  else if( Polarisation == CIRCULAR )
114  {
115  elong.x() = float_X( envelope / sqrt(2.0) * (math::sin(t_and_phase)
116  + math::cos(t_and_phase) * integrationCorrectionFactor));
117  elong.z() = float_X( envelope / sqrt(2.0) * (math::cos(t_and_phase)
118  - math::sin(t_and_phase) * integrationCorrectionFactor));
119  }
120 
121 
122  phase = float_X(0.0);
123 
124  return elong;
125  }
