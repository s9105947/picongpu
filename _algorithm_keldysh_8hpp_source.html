62  template<typename EType, typename BType, typename ParticleType >
63  HDINLINE uint32_t
64  operator()( const BType bField, const EType eField, ParticleType& parentIon, float_X randNr )
65  {
66 
67  const float_X protonNumber = GetAtomicNumbers<ParticleType>::type::numberOfProtons;
68  float_X chargeState = attribute::getChargeState(parentIon);
69 
70  /* verify that ion is not completely ionized */
71  if ( chargeState < protonNumber )
72  {
73  uint32_t const cs = math::float2int_rd(chargeState);
74  const float_X iEnergy = typename GetIonizationEnergies<ParticleType>::type{ }[cs];
75 
76  constexpr float_X pi = pmacc::algorithms::math::Pi< float_X >::value;
77  /* electric field in atomic units - only absolute value */
78  float_X eInAU = math::abs(eField) / ATOMIC_UNIT_EFIELD;
79 
80  /* factor two avoid calculation math::pow(2,5./4.); */
81  const float_X twoToFiveQuarters = 2.3784142300054;
82 
83  /* characteristic exponential function argument */
84  const float_X charExpArg = math::sqrt(util::cube(float_X(2.)*iEnergy))/eInAU;
85 
86  /* ionization rate */
87  float_X rateKeldysh = math::sqrt(float_X(6.)*pi) / twoToFiveQuarters \
88  * iEnergy * math::sqrt(float_X(1.)/charExpArg) \
89  * math::exp(-float_X(2./3.) * charExpArg);
90 
91  /* simulation time step in atomic units */
92  const float_X timeStepAU = float_X(DELTA_T / ATOMIC_UNIT_TIME);
93  /* ionization probability
94  *
95  * probability = rate * time step
96  * --> for infinitesimal time steps
97  *
98  * the whole ensemble should then follow
99  * P = 1 - exp(-rate * time step) if the laser wavelength is
100  * sampled well enough
101  */
102  float_X const probKeldysh = rateKeldysh * timeStepAU;
103 
104  /* ionization condition */
105  if( randNr < probKeldysh )
106  {
107  /* return number of macro electrons to produce */
108  return 1u;
109  }
110  }
111  /* no ionization */
112  return 0u;
113  }
114  };
