249  void restart(uint32_t timeStep, const std::string restartDirectory)
250  {
251  // only load backup if radiation is calculated:
252  if(notifyPeriod.empty())
253  return;
254 
255  if(dependenciesFulfilled && isMaster)
256  {
257  // this will lead to wrong lastRad output right after the checkpoint if the restart point is
258  // not a dump point. The correct lastRad data can be reconstructed from hdf5 data
259  // since text based lastRad output will be obsolete soon, this is not a problem
260  readHDF5file(timeSumArray, restartDirectory + "/" + speciesName + std::string("_radRestart_"), timeStep);
261  log<radLog::SIMULATION_STATE > ("Radiation (%1%): restart finished") % speciesName;
262  }

266  void checkpoint(uint32_t timeStep, const std::string restartDirectory)

283  }

300  void pluginLoad()
301  {
302  if(dependenciesFulfilled)
303  {
304  // allocate memory for all amplitudes for temporal data collection
305  tmp_result = new Amplitude[elements_amplitude()];
306 
307  if(!notifyPeriod.empty())
308  {
309  /*only rank 0 create a file*/
310  isMaster = reduce.hasResult(mpi::reduceMethods::Reduce());
311 
312  radiation = new GridBuffer<Amplitude, DIM1 > (DataSpace<DIM1 > (elements_amplitude())); //create one int on GPU and host
313 
314  freqInit.Init(pathOmegaList);
315  freqFkt = freqInit.getFunctor();
316 
317 
318  Environment<>::get().PluginConnector().setNotificationPeriod(this, notifyPeriod);
319  pmacc::Filesystem<simDim>& fs = Environment<simDim>::get().Filesystem();
320 
321  if (isMaster)
322  {
323  timeSumArray = new Amplitude[elements_amplitude()];
324 
325  /* save detector position / observation direction */
326  detectorPositions = new vector_64[parameters::N_observer];
327  for(uint32_t detectorIndex=0; detectorIndex < parameters::N_observer; ++detectorIndex)
328  {
329  detectorPositions[detectorIndex] = radiation_observer::observation_direction(detectorIndex);
330  }
331 
332  /* save detector frequencies */
333  detectorFrequencies = new float_64[radiation_frequencies::N_omega];
334  for(uint32_t detectorIndex=0; detectorIndex < radiation_frequencies::N_omega; ++detectorIndex)
335  {
336  detectorFrequencies[detectorIndex] = freqFkt(detectorIndex);
337  }
338 
339  }
340 
341  if (isMaster && totalRad)
342  {
343  fs.createDirectory("radiationHDF5");
344  fs.setDirectoryPermissions("radiationHDF5");
345  }
346 
347 
348  if (isMaster && radPerGPU)
349  {
350  fs.createDirectory(folderRadPerGPU);
351  fs.setDirectoryPermissions(folderRadPerGPU);
352  }
353 
354  if (isMaster && totalRad)
355  {
356  //create folder for total output
357  fs.createDirectory(folderTotalRad);
358  fs.setDirectoryPermissions(folderTotalRad);
359  for (unsigned int i = 0; i < elements_amplitude(); ++i)
360  timeSumArray[i] = Amplitude::zero();
361  }
362  if (isMaster && lastRad)
363  {
364  //create folder for total output
365  fs.createDirectory(folderLastRad);
366  fs.setDirectoryPermissions(folderLastRad);
367  }
368 
369  }
370  }

374  void pluginUnload()
375  {
376  if(dependenciesFulfilled && !notifyPeriod.empty())
377  {
378 
379  // Some funny things that make it possible for the kernel to calculate
380  // the absolute position of the particles
381  const SubGrid<simDim>& subGrid = Environment<simDim>::get().SubGrid();
382  DataSpace<simDim> localSize(subGrid.getLocalDomain().size);
383  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(currentStep);
384  DataSpace<simDim> globalOffset(subGrid.getLocalDomain().offset);
385  globalOffset.y() += (localSize.y() * numSlides);
386 
387  // only print data at end of simulation if no dump period was set
388  if (dumpPeriod == 0)
389  {
390  collectDataGPUToMaster();
391  writeAllFiles(globalOffset);
392  }
393 
394  if (isMaster)
395  {
396  __deleteArray(timeSumArray);
397  delete[] detectorPositions;
398  delete[] detectorFrequencies;
399  }
400 
401  __delete(radiation);
402  CUDA_CHECK(cudaGetLastError());
403  }
404 
405  __deleteArray(tmp_result);
406  }

428  std::stringstream GPUpos_str;
429 
430  last_time_step_str << lastStep;
431  current_time_step_str << currentStep;

434  GPUpos_str << "_" <<currentGPUpos[dimIndex];
435 
436  writeFile(radiation->getHostBuffer().getBasePointer(), folderRadPerGPU + "/" + speciesName
437  + "_radPerGPU_pos" + GPUpos_str.str()
438  + "_time_" + last_time_step_str.str()
439  + "-" + current_time_step_str.str() + ".dat");
440  }
441  lastGPUpos = currentGPUpos;
442  }

666  void writeHDF5file(Amplitude* values, std::string name)
667  {
668  splash::SerialDataCollector hdf5DataFile(1);
669  splash::DataCollector::FileCreationAttr fAttr;
670 
671  splash::DataCollector::initFileCreationAttr(fAttr);
672  fAttr.enableCompression = compressionOn;
673 
674  std::ostringstream filename;
675  filename << name << currentStep;
676 
677  hdf5DataFile.open(filename.str().c_str(), fAttr);
678 
679  typename PICToSplash<float_64>::type radSplashType;
680 
681 
682  splash::Dimensions bufferSize(Amplitude::numComponents,
683  radiation_frequencies::N_omega,
684  parameters::N_observer);
685 
686  splash::Dimensions componentSize(1,
687  radiation_frequencies::N_omega,
688  parameters::N_observer);
689 
690  splash::Dimensions stride(Amplitude::numComponents,1,1);
691 
692  /* get the radiation amplitude unit */
693  Amplitude UnityAmplitude(1., 0., 0., 0., 0., 0.);
694  const picongpu::float_64 factor = UnityAmplitude.calc_radiation() * UNIT_ENERGY * UNIT_TIME ;
695 
696  typedef PICToSplash<float_X>::type SplashFloatXType;
697  SplashFloatXType splashFloatXType;
698 
699  for(uint32_t ampIndex=0; ampIndex < Amplitude::numComponents; ++ampIndex)
700  {
701  splash::Dimensions offset(ampIndex,0,0);
702  splash::Selection dataSelection(bufferSize,
703  componentSize,
704  offset,
705  stride);
706 
707  /* save data for each x/y/z * Re/Im amplitude */
708  hdf5DataFile.write(currentStep,
709  radSplashType,
710  3,
711  dataSelection,
712  (meshesPathName + dataLabels(ampIndex)).c_str(),
713  values);
714 
715  /* save SI unit as attribute together with data set */
716  hdf5DataFile.writeAttribute(currentStep,
717  radSplashType,
718  (meshesPathName + dataLabels(ampIndex)).c_str(),
719  "unitSI",
720  &factor);
721 
722  /* position */
723  std::vector<float_X> positionMesh(simDim, 0.0); /* there is no offset - zero */
724  hdf5DataFile.writeAttribute(currentStep,
725  splashFloatXType,
726  (meshesPathName + dataLabels(ampIndex)).c_str(),
727  "position",
728  1u,
729  splash::Dimensions(simDim,0,0),
730  &(*positionMesh.begin()));
731  }
732 
733  /* save SI unit as attribute in the Amplitude group (for convenience) */
734  hdf5DataFile.writeAttribute(currentStep,
735  radSplashType,
736  (meshesPathName + std::string("Amplitude")).c_str(),
737  "unitSI",
738  &factor);
739 
740  /* save detector position / observation direction */
741  splash::Dimensions bufferSizeDetector(3,
742  1,
743  parameters::N_observer);
744 
745  splash::Dimensions componentSizeDetector(1,
746  1,
747  parameters::N_observer);
748 
749  splash::Dimensions strideDetector(3,1,1);
750 
751  for(uint32_t detectorDim=0; detectorDim < 3; ++detectorDim)
752  {
753  splash::Dimensions offset(detectorDim,0,0);
754  splash::Selection dataSelection(bufferSizeDetector,
755  componentSizeDetector,
756  offset,
757  strideDetector);
758 
759  hdf5DataFile.write(currentStep,
760  radSplashType,
761  3,
762  dataSelection,
763  (meshesPathName + dataLabelsDetectorDirection(detectorDim)).c_str(),
764  detectorPositions);
765 
766  /* save SI unit as attribute together with data set */
767  const picongpu::float_64 factorDirection = 1.0 ;
768  hdf5DataFile.writeAttribute(currentStep,
769  radSplashType,
770  (meshesPathName + dataLabelsDetectorDirection(detectorDim)).c_str(),
771  "unitSI",
772  &factorDirection);
773 
774  /* position */
775  std::vector<float_X> positionMesh(simDim, 0.0); /* there is no offset - zero */
776  hdf5DataFile.writeAttribute(currentStep,
777  splashFloatXType,
778  (meshesPathName + dataLabelsDetectorDirection(detectorDim)).c_str(),
779  "position",
780  1u,
781  splash::Dimensions(simDim,0,0),
782  &(*positionMesh.begin()));
783 
784  }
785 
786 
787 
788  /* save detector frequencies */
789  splash::Dimensions bufferSizeOmega(1,
790  radiation_frequencies::N_omega,
791  1);
792 
793  splash::Dimensions strideOmega(1,1,1);
794 
795  splash::Dimensions offset(0,0,0);
796  splash::Selection dataSelection(bufferSizeOmega,
797  bufferSizeOmega,
798  offset,
799  strideOmega);
800 
801  hdf5DataFile.write(currentStep,
802  radSplashType,
803  3,
804  dataSelection,
805  (meshesPathName + dataLabelsDetectorFrequency(0)).c_str(),
806  detectorFrequencies);

808  /* save SI unit as attribute together with data set */

811  radSplashType,
812  (meshesPathName + dataLabelsDetectorFrequency(0)).c_str(),
813  "unitSI",
814  &factorOmega);

818  hdf5DataFile.writeAttribute(currentStep,
819  splashFloatXType,
820  (meshesPathName + dataLabelsDetectorFrequency(0)).c_str(),
821  "position",
822  1u,
823  splash::Dimensions(simDim,0,0),
824  &(*positionMesh.begin()));

831  hdf5DataFile.writeGlobalAttribute( ctOpenPMDversion,
