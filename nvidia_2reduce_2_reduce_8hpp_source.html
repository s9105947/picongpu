307  HINLINE Reduce(const uint32_t byte, const uint32_t sharedMemByte = 4 * 1024) :
308  byte(byte), sharedMemByte(sharedMemByte), reduceBuffer(nullptr)
309  {
310 
311  reduceBuffer = new GridBuffer<char, DIM1 > (DataSpace<DIM1 > (byte));
312  }

314  /* Reduce elements in global gpu memory
315  *
316  * @param func binary functor for reduce which takes two arguments, first argument is the source and get the new reduced value.
317  * Functor must specialize the function getMPI_Op.
318  * @param src a class or a pointer where the reduce algorithm can access the value by operator [] (one dimensional access)
319  * @param n number of elements to reduce
320  *
321  * @return reduced value
322  */
323  template<class Functor, typename Src>
324  HINLINE typename traits::GetValueType<Src>::ValueType operator()(Functor func, Src src, uint32_t n)
325  {
326  /* - the result of a functor can be a reference or a const value
327  * - it is not allowed to create const or reference memory
328  * thus we remove `references` and `const` qualifiers */
329  typedef typename boost::remove_const<
330  typename boost::remove_reference<
331  typename traits::GetValueType<Src>::ValueType
332  >::type
333  >::type Type;
334 
335  uint32_t blockcount = optimalThreadsPerBlock(n, sizeof (Type));
336 
337  uint32_t n_buffer = byte / sizeof (Type);
338 
339  uint32_t threads = n_buffer * blockcount * 2; /* x2 is used thus we can use all byte in Buffer, after we calculate threads/2 */
340 
341 
342 
343  if (threads > n) threads = n;
344  Type* dest = (Type*) reduceBuffer->getDeviceBuffer().getBasePointer();
345 
346  uint32_t blocks = threads / 2 / blockcount;
347  if (blocks == 0) blocks = 1;
348  callReduceKernel< Type >(blocks, blockcount, blockcount * sizeof (Type),
349  src, n, dest, func, pmacc::nvidia::functors::Assign());
350  n = blocks;
351  blockcount = optimalThreadsPerBlock(n, sizeof (Type));
352  blocks = n / 2 / blockcount;
353  if (blocks == 0 && n > 1) blocks = 1;
354 
355 
356  while (blocks != 0)
357  {
358  if (blocks > 1)
359  {
360  uint32_t blockOffset = ceil((double) blocks / blockcount);
361  uint32_t useBlocks = blocks - blockOffset;
362  uint32_t problemSize = n - (blockOffset * blockcount);
363  Type* srcPtr = dest + (blockOffset * blockcount);
364 
365  callReduceKernel< Type >(useBlocks, blockcount, blockcount * sizeof (Type),
366  srcPtr, problemSize, dest, func, func);
367  blocks = blockOffset*blockcount;
368  }
369  else
370  {
371 
372  callReduceKernel< Type >(blocks, blockcount, blockcount * sizeof (Type),
373  dest, n, dest, func, pmacc::nvidia::functors::Assign());
374  }
375 
376  n = blocks;
377  blockcount = optimalThreadsPerBlock(n, sizeof (Type));
378  blocks = n / 2 / blockcount;
379  if (blocks == 0 && n > 1) blocks = 1;
380  }
381 
382  reduceBuffer->deviceToHost();
383  __getTransactionEvent().waitForFinished();
384  return *((Type*) (reduceBuffer->getHostBuffer().getBasePointer()));
385 
386  }
