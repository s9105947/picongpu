233  DINLINE float_X emission_prob(
234  const float_X delta,
235  const float_X chi,
236  const float_X gamma) const
237  {
238  // catch these special values because otherwise a NaN is returned whereas it should be a zero.
239  if(chi == float_X(0.0) || delta == float_X(0.0) || (float_X(1.0) - delta) == float_X(0.0))
240  return float_X(0.0);
241 
242  const float_X mass = frame::getMass<FrameType>();
243  const float_X charge = frame::getCharge<FrameType>();
244 
245  const float_X sqrtOf3 = 1.7320508075688772;
246  const float_X factor = DELTA_T * charge*charge * mass * SPEED_OF_LIGHT / (float_X(4.0) * PI * EPS0 * HBAR*HBAR) *
247  sqrtOf3 / (float_X(2.0) * PI) * chi / gamma;
248 
249  if(enableQEDTerm)
250  {
251  // quantum
252  const float_X z = float_X(2.0/3.0) * delta / ((float_X(1.0) - delta) * chi);
253 
254  return factor * (float_X(1.0) - delta) / delta *
255  (this->curF_1[z] + float_X(1.5) * delta * chi * z * this->curF_2[z]);
256  }
257  else
258  {
259  // classical
260  const float_X z = float_X(2.0/3.0) * delta / chi;
261 
262  return factor / delta * this->curF_1[z];
263  }
264  }

274  DINLINE float_X emission_prob_scaled(
275  const float_X deltaScaled,
276  const float_X chi,
277  const float_X gamma) const
278  {
279  const float_X delta = deltaScaled*deltaScaled*deltaScaled;
280  return float_X(3.0) * deltaScaled*deltaScaled * emission_prob(delta, chi, gamma);
281  }

291  template< typename T_Acc >
292  DINLINE unsigned int numNewParticles(const T_Acc& acc, FrameType& sourceFrame, int localIdx)
293  {
294  using namespace pmacc::algorithms;
295 
296  auto particle = sourceFrame[localIdx];
297 
298  /* particle position, used for field-to-particle interpolation */
299  const floatD_X pos = particle[position_];
300  const int particleCellIdx = particle[localCellIdx_];
301  /* multi-dim coordinate of the local cell inside the super cell */
302  DataSpace<TVec::dim> localCell(DataSpaceOperations<TVec::dim>::template map<TVec > (particleCellIdx));
303  /* interpolation of E-field on the particle position */
304  const picongpu::traits::FieldPosition<typename fields::Solver::NummericalCellType, FieldE> fieldPosE;
305  ValueType_E fieldE = Field2ParticleInterpolation()
306  (cachedE.shift(localCell).toCursor(), pos, fieldPosE());
307  /* interpolation of B-field on the particle position */
308  const picongpu::traits::FieldPosition<typename fields::Solver::NummericalCellType, FieldB> fieldPosB;
309  ValueType_B fieldB = Field2ParticleInterpolation()
310  (cachedB.shift(localCell).toCursor(), pos, fieldPosB());
311 
312  /* All computation below is in the single "real" particle picture.
313  * The macroparticle weighting factor is reintroduced at the end of this code block. */
314  const float3_X mom = particle[momentum_] / particle[weighting_];
315  const float_X mom2 = math::dot(mom, mom);
316  const float3_X mom_norm = mom * math::rsqrt(mom2);
317  const float_X mass = frame::getMass<FrameType>();
318 
319  const float_X gamma = Gamma<>()(mom, mass);
320  const float3_X vel = mom / (gamma * mass); // low accuracy?
321 
322  const float3_X lorentzForceOverCharge = fieldE + math::cross(vel, fieldB);
323  const float_X lorentzForceOverCharge2 = math::dot(lorentzForceOverCharge, lorentzForceOverCharge);
324  const float_X fieldE_long = math::dot(mom_norm, fieldE);
325 
326  // effective magnetic strength (in cgs)
327  const float_X H_eff = math::sqrt(lorentzForceOverCharge2 - fieldE_long*fieldE_long);
328 
329  const float_X charge = math::abs(frame::getCharge<FrameType>());
330 
331  const float_X c = SPEED_OF_LIGHT;
332  // Schwinger limit, unit: V/m (in cgs)
333  const float_X E_S = mass*mass * c*c*c / (charge * HBAR);
334  // quantum-nonlinearity parameter
335  const float_X chi = gamma * H_eff / E_S;
336 
337  const float_X deltaScaled = this->randomGen(acc);
338 
339  const float_X x = emission_prob_scaled(deltaScaled, chi, gamma);
340 
341  // raise a warning if the emission probability is too high.
342  if(picLog::log_level & picLog::CRITICAL::lvl)
343  {
344  if(x > float_X(SINGLE_EMISSION_PROB_LIMIT))
345  {
346  const float_X delta = deltaScaled*deltaScaled*deltaScaled;
347  printf("[SynchrotronPhotons] warning: emission probability is too high: p = %g, at delta = %g, chi = %g, gamma = %g\n",
348  x, delta, chi, gamma);
349  }
350  }
351 
352  if(this->randomGen(acc) < x)
353  {
354  const float_X delta = deltaScaled*deltaScaled*deltaScaled;
355  const float_X photonMom_abs = delta * mass*c * gamma;
356  if(photonMom_abs > SOFT_PHOTONS_CUTOFF_MOM)
357  {
358  this->photon_mom = mom_norm * photonMom_abs * particle[weighting_];
359  return 1;
360  }
361  }
362 
363  return 0;
364  }
