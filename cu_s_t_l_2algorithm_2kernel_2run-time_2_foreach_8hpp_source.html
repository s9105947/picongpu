99 template<int dim>
100 math::Size_t<DIM3> getBestCudaBlockDim(const math::Size_t<dim> gridDimension)
101 {
102  math::Size_t<DIM3> result = math::Size_t<DIM3>::create(1);
103 
104  /* The greatest common divisor of each component of the volume size
105  * and a certain power of two value yield the best suitable block size */
106  const math::Size_t<DIM3> maxThreads =
107  MaxCudaBlockDim<dim>::type::toRT(); /* max threads per axis */
108  for(int i = 0; i < dim; i++)
109  {
110  result[i] = boost::integer::gcd(gridDimension[i], maxThreads[i]);
111  }
112 
113  return result;
114 }

123 #define FOREACH_OPERATOR(Z, N, _) \
124  /* typename C0, ..., typename CN */ \
125  template<typename Zone, BOOST_PP_ENUM_PARAMS(N, typename C), typename Functor> \
126  /* C0 c0, ..., CN cN */ \
127  void operator()(const Zone& p_zone, BOOST_PP_ENUM_BINARY_PARAMS(N, C, c), const Functor& functor) \
128  { \
129  /* C0 c0_shifted = c0(p_zone.offset); ...; CN cN_shifted = cN(p_zone.offset); */ \
130  BOOST_PP_REPEAT(N, SHIFT_CURSOR_ZONE, _) \
131  \
132  if(this->_blockDim == math::Size_t<DIM3>::create(0)) \
133  this->_blockDim = getBestCudaBlockDim(p_zone.size); \
134  \
135  PMACC_VERIFY(this->_blockDim.productOfComponents() <= cudaSpecs::maxNumThreadsPerBlock); \
136  PMACC_VERIFY(this->_blockDim.x() <= cudaSpecs::MaxNumThreadsPerBlockDim::x::value); \
137  PMACC_VERIFY(this->_blockDim.y() <= cudaSpecs::MaxNumThreadsPerBlockDim::y::value); \
138  PMACC_VERIFY(this->_blockDim.z() <= cudaSpecs::MaxNumThreadsPerBlockDim::z::value); \
139  \
140  typename math::Size_t<3>::BaseType blockSize( \
141  this->_blockDim.x(), \
142  this->_blockDim.y(), \
143  this->_blockDim.z() \
144  ); \
145  uint32_t numWorkers = traits::GetNumWorkers< cudaSpecs::MaxNumThreadsPerBlockDim::x::value >::value; \
146  if( numWorkers > blockSize.productOfComponents() ) \
147  numWorkers = blockSize.productOfComponents(); \
148  kernel::detail::SphericMapper<Zone::dim> mapper; \
149  using namespace pmacc; \
150  PMACC_KERNEL(kernel::detail::RT::KernelForeachLockstep{})(mapper.cudaGridDim(p_zone.size, this->_blockDim), numWorkers) \
151  /* c0_shifted, ..., cN_shifted */ \
152  (mapper, blockSize, functor, BOOST_PP_ENUM(N, SHIFTED_CURSOR, _)); \
153  }

167  *
168  * blockDim has to fit into the computing volume.

175 
176  /* operator()(zone, cursor0, cursor1, ..., cursorN-1, functor or lambdaFun)
177  *
178  * \param zone Accepts currently only a zone::SphericZone object (e.g. containerObj.zone())
179  * \param cursorN cursor for the N-th data source (e.g. containerObj.origin())
180  * \param functor or lambdaFun either a functor with N arguments or a N-ary lambda function (e.g. _1 = _2)
181  *
182  * The functor or lambdaFun is called for each cell within the zone.
183  * It is called like functor(*cursor0(cellId), ..., *cursorN(cellId))
184  *
185  */
186  BOOST_PP_REPEAT_FROM_TO(1, BOOST_PP_INC(FOREACH_KERNEL_MAX_PARAMS), FOREACH_OPERATOR, _)
