38  HINLINE float3_X laserLongitudinal( uint32_t currentStep, float_X& phase )
39  {
40  /* initialize the laser not in the first cell is equal to a negative shift
41  * in time
42  */
43  constexpr float_X laserTimeShift = laser::initPlaneY * CELL_HEIGHT / SPEED_OF_LIGHT;
44  const float_64 runTime = DELTA_T * currentStep - laserTimeShift;
45  const float_64 f = SPEED_OF_LIGHT / WAVE_LENGTH;
46 
47  /* calculate focus position relative to the laser initialization plane */
48  const float_X focusPos = FOCUS_POS - laser::initPlaneY * CELL_HEIGHT;
49 
50  float3_X elong(float3_X::create(0.0));
51 
52  // This check is done here on HOST, since std::numeric_limits<float_X>::epsilon() does not compile on laserTransversal(), which is on DEVICE.
53  float_X etrans_norm = float_X(0.0);
54 
55  PMACC_CASSERT_MSG(MODENUMBER_must_be_smaller_than_number_of_entries_in_LAGUERREMODES_vector, MODENUMBER<LAGUERREMODES_t::dim);
56  for ( uint32_t m = 0 ; m <= MODENUMBER ; ++m )
57  etrans_norm += LAGUERREMODES[m];
58  PMACC_VERIFY_MSG( algorithms::math::abs(etrans_norm) > std::numeric_limits<float_X>::epsilon(), "Sum of LAGUERREMODES can not be 0." );
59 
60 
61  // a symmetric pulse will be initialized at position z=0 for
62  // a time of PULSE_INIT * PULSE_LENGTH = INIT_TIME.
63  // we shift the complete pulse for the half of this time to start with
64  // the front of the laser pulse.
65  const float_64 mue = 0.5 * INIT_TIME;
66 
67  //rayleigh length (in y-direction)
68  const float_64 y_R = PI * W0 * W0 / WAVE_LENGTH;
69  //gaussian beam waist in the nearfield: w_y(y=0) == W0
70  const float_64 w_y = W0 * sqrt( 1.0 + ( focusPos / y_R )*( focusPos / y_R ) );
71 
72  float_64 envelope = float_64( AMPLITUDE );
73  if( simDim == DIM2 )
74  envelope *= math::sqrt( float_64( W0 ) / w_y );
75  else if( simDim == DIM3 )
76  envelope *= float_64( W0 ) / w_y;
77  /* no 1D representation/implementation */
78 
79  if( Polarisation == LINEAR_X )
80  {
81  elong.x() = float_X( envelope );
82  }
83  else if( Polarisation == LINEAR_Z )
84  {
85  elong.z() = float_X( envelope );
86  }
87  else if( Polarisation == CIRCULAR )
88  {
89  elong.x() = float_X( envelope / sqrt(2.0) );
90  elong.z() = float_X( envelope / sqrt(2.0) );
91  }
92 
93  phase = float_X(2.0) * float_X(PI ) * float_X(f ) * ( runTime - float_X(mue ) - focusPos / SPEED_OF_LIGHT) + LASER_PHASE;
94 
95  return elong;
96  }

104  HDINLINE float_X simpleLaguerre( const uint32_t n, const float_X x )
105  {
106  //Result for special case n == 0
107  if (n == 0) return float_X(1.0);
108  uint32_t currentN = 1;
109  float_X laguerreNMinus1 = float_X(1.0);
110  float_X laguerreN = float_X(1.0) - x;
111  float_X laguerreNPlus1 = float_X(0.0);
112  while (currentN < n)
113  {
114  //Core statement of the algorithm
115  laguerreNPlus1 = ( ( float_X(2.0) * float_X(currentN) + float_X(1.0) - x) * laguerreN - float_X(currentN) * laguerreNMinus1 ) / float_X(currentN + 1);
116  //Advance by one order
117  laguerreNMinus1 = laguerreN;
118  laguerreN = laguerreNPlus1;
119  currentN++;
120  }
121  return laguerreN;
122  }

132  HDINLINE float3_X laserTransversal( float3_X elong, float_X phase, const float_X posX, const float_X posZ )
133  {
134  /* calculate focus position relative to the laser initialization plane */
135  const float_X focusPos = FOCUS_POS - laser::initPlaneY * CELL_HEIGHT;
136  const float_X r2 = posX * posX + posZ * posZ;
137 
138  //rayleigh length (in y-direction)
139  const float_X y_R = float_X( PI ) * W0 * W0 / WAVE_LENGTH;
140 
141  // the radius of curvature of the beam's wavefronts
142  const float_X R_y = -focusPos * ( float_X(1.0) + ( y_R / focusPos )*( y_R / focusPos ) );
143 
144  // initialize temporary variables
145  float_X etrans = float_X(0.0);
146  float_X etrans_norm = float_X(0.0);
147  PMACC_CASSERT_MSG(MODENUMBER_must_be_smaller_than_number_of_entries_in_LAGUERREMODES_vector, MODENUMBER<LAGUERREMODES_t::dim);
148  for ( uint32_t m = 0 ; m <= MODENUMBER ; ++m )
149  etrans_norm += LAGUERREMODES[m];
150 
151  //beam waist in the near field: w_y(y=0) == W0
152  const float_X w_y = W0 * algorithms::math::sqrt( float_X(1.0) + ( focusPos / y_R )*( focusPos / y_R ) );
154  const float_X xi_y = algorithms::math::atan( -focusPos / y_R );
155 
156  if( Polarisation == LINEAR_X || Polarisation == LINEAR_Z )
157  {
158  for ( uint32_t m = 0 ; m <= MODENUMBER ; ++m )
159  {
160  etrans += LAGUERREMODES[m] * simpleLaguerre( m, float_X(2.0) * r2 / w_y / w_y )
161  * math::exp( -r2 / w_y / w_y ) * math::cos( float_X(2.0) * float_X( PI ) / WAVE_LENGTH * focusPos - float_X(2.0) * float_X( PI ) / WAVE_LENGTH * r2 / float_X(2.0) / R_y + ( 2*m + 1 ) * xi_y + phase )
162  * math::exp( -( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
163  *( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
164  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( float_X(2.0) * PULSE_LENGTH ) / ( float_X(2.0) * PULSE_LENGTH ) );
165  }
166  elong *= etrans / etrans_norm;
167  }
168  else if( Polarisation == CIRCULAR )
169  {
170  for ( uint32_t m = 0 ; m <= MODENUMBER ; ++m )
171  {
172  etrans += LAGUERREMODES[m] * simpleLaguerre( m, float_X(2.0) * r2 / w_y / w_y )
173  * math::exp( -r2 / w_y / w_y ) * math::cos( float_X(2.0) * float_X( PI ) / WAVE_LENGTH * focusPos - float_X(2.0) * float_X( PI ) / WAVE_LENGTH * r2 / float_X(2.0) / R_y + ( 2*m + 1 ) * xi_y + phase )
174  * math::exp( -( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
175  *( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
176  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( float_X(2.0) * PULSE_LENGTH ) / ( float_X(2.0) * PULSE_LENGTH ) );
177  }
178  elong.x() *= etrans / etrans_norm;
179  phase += float_X( PI / 2.0 );
180  etrans = float_X(0.0);
181  for ( uint32_t m = 0 ; m <= MODENUMBER ; ++m )
182  {
183  etrans += LAGUERREMODES[m] * simpleLaguerre( m, float_X(2.0) * r2 / w_y / w_y )
184  * math::exp( -r2 / w_y / w_y ) * math::cos( float_X(2.0) * float_X( PI ) / WAVE_LENGTH * focusPos - float_X(2.0) * float_X( PI ) / WAVE_LENGTH * r2 / float_X(2.0) / R_y + ( 2*m + 1 ) * xi_y + phase )
185  * math::exp( -( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
186  *( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
187  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( float_X(2.0) * PULSE_LENGTH ) / ( float_X(2.0) * PULSE_LENGTH ) );
188  }
189  elong.z() *= etrans / etrans_norm;
190  // reminder: if you want to use phase below, substract pi/2
191  // phase -= float_X( PI / 2.0 );
192  }
193 
194  return elong;
195 
196  }
