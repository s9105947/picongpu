102 int64_t defineAdiosVar(int64_t group_id,
103  const char * name,
104  const char * path,
105  enum ADIOS_DATATYPES type,
106  pmacc::math::UInt64<DIM> dimensions,
107  pmacc::math::UInt64<DIM> globalDimensions,
108  pmacc::math::UInt64<DIM> offset,
109  bool compression,
110  std::string compressionMethod)
111 {
112  int64_t var_id = 0;
113 
114  var_id = adios_define_var(
115  group_id, name, path, type,
116  dimensions.revert().toString(",", "").c_str(),
117  globalDimensions.revert().toString(",", "").c_str(),
118  offset.revert().toString(",", "").c_str()
119  );
120 
121  if(compression)
122  {
123  /* enable adios transform layer for variable */
124  adios_set_transform(var_id, compressionMethod.c_str());
125  }
126 
127  log<picLog::INPUT_OUTPUT > ("ADIOS: Defined varID=%1% for '%2%' at %3% for %4%/%5% elements") %
128  var_id % std::string(name) % offset.toString() % dimensions.toString() % globalDimensions.toString();
129  return var_id;
130 }

515  static void defineFieldVar(ThreadParams* params,
516  uint32_t nComponents, ADIOS_DATATYPES adiosType, const std::string name,
517  std::vector<float_64> unit, std::vector<float_64> unitDimension,
518  std::vector<std::vector<float_X> > inCellPosition, float_X timeOffset)
519  {
520  PICToAdios<float_64> adiosDoubleType;
521  PICToAdios<float_X> adiosFloatXType;
522 
523  const std::string name_lookup_tpl[] = {"x", "y", "z", "w"};
524 
525  /* parameter checking */
526  PMACC_ASSERT( unit.size() == nComponents );
527  PMACC_ASSERT( inCellPosition.size() == nComponents );
528  for( uint32_t n = 0; n < nComponents; ++n )
529  PMACC_ASSERT( inCellPosition.at(n).size() == simDim );
530  PMACC_ASSERT(unitDimension.size() == 7); // seven openPMD base units
531 
532  const std::string recordName( params->adiosBasePath +
533  std::string(ADIOS_PATH_FIELDS) + name );
534 
535  for( uint32_t c = 0; c < nComponents; c++ )
536  {
537  std::stringstream datasetName;
538  datasetName << recordName;
539  if (nComponents > 1)
540  datasetName << "/" << name_lookup_tpl[c];
541 
542  /* define adios var for field, e.g. field_FieldE_y */
543  const char* path = nullptr;
544  int64_t adiosFieldVarId = defineAdiosVar<simDim>(
545  params->adiosGroupHandle,
546  datasetName.str().c_str(),
547  path,
548  adiosType,
549  params->fieldsSizeDims,
550  params->fieldsGlobalSizeDims,
551  params->fieldsOffsetDims,
552  true,
553  params->adiosCompression);
554 
555  params->adiosFieldVarIds.push_back(adiosFieldVarId);
556 
557  /* already add the unitSI and further attribute so `adios_group_size`
558  * calculates the reservation for the buffer correctly */
559  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
560  "position", datasetName.str().c_str(),
561  adiosFloatXType.type, simDim, &(*inCellPosition.at(c).begin()) ));
562 
563  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
564  "unitSI", datasetName.str().c_str(),
565  adiosDoubleType.type, 1, &unit.at(c) ));
566  }
567 
568  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
569  "unitDimension", recordName.c_str(),
570  adiosDoubleType.type, 7, &(*unitDimension.begin()) ));
571 
572  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
573  "timeOffset", recordName.c_str(),
574  adiosFloatXType.type, 1, &timeOffset ));
575 
576  const std::string geometry( "cartesian" );
577  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
578  "geometry", recordName.c_str(),
579  adios_string, 1, (void*)geometry.c_str() ));
580 
581  const std::string dataOrder( "C" );
582  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
583  "dataOrder", recordName.c_str(),
584  adios_string, 1, (void*)dataOrder.c_str() ));
585 
586  if( simDim == DIM2 )
587  {
588  const char* axisLabels[] = {"y", "x"}; // 2D: F[y][x]
589  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
590  "axisLabels", recordName.c_str(),
591  adios_string_array, simDim, axisLabels ));
592  }
593  if( simDim == DIM3 )
594  {
595  const char* axisLabels[] = {"z", "y", "x"}; // 3D: F[z][y][x]
596  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
597  "axisLabels", recordName.c_str(),
598  adios_string_array, simDim, axisLabels ));
599  }
600 
601  // cellSize is {x, y, z} but fields are F[z][y][x]
602  std::vector<float_X> gridSpacing(simDim, 0.0);
603  for( uint32_t d = 0; d < simDim; ++d )
604  gridSpacing.at(simDim-1-d) = cellSize[d];
605 
606  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
607  "gridSpacing", recordName.c_str(),
608  adiosFloatXType.type, simDim, &(*gridSpacing.begin()) ));
609 
610  /* globalSlideOffset due to gpu slides between origin at time step 0
611  * and origin at current time step
612  * ATTENTION: splash offset are globalSlideOffset + picongpu offsets
613  */
614  DataSpace<simDim> globalSlideOffset;
615  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
616  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(params->currentStep);
617  globalSlideOffset.y() += numSlides * localDomain.size.y();
618 
619  // globalDimensions is {x, y, z} but fields are F[z][y][x]
620  std::vector<float_64> gridGlobalOffset(simDim, 0.0);
621  for( uint32_t d = 0; d < simDim; ++d )
622  gridGlobalOffset.at(simDim-1-d) =
623  float_64(cellSize[d]) *
624  float_64(params->window.globalDimensions.offset[d] +
625  globalSlideOffset[d]);
626 
627  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
628  "gridGlobalOffset", recordName.c_str(),
629  adiosDoubleType.type, simDim, &(*gridGlobalOffset.begin()) ));
630 
631  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
632  "gridUnitSI", recordName.c_str(),
633  adiosDoubleType.type, 1, (void*)&UNIT_LENGTH ));
634 
635  const std::string fieldSmoothing( "none" );
636  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
637  "fieldSmoothing", recordName.c_str(),
638  adios_string, 1, (void*)fieldSmoothing.c_str() ));
639  }

912  const std::string& restartDirectory,
913  const std::string& constRestartFilename,
914  const uint32_t restartChunkSize
915  )
916  {
917  // restart is only allowed if the plugin is controlled by the class Checkpoint
918  assert(!m_help->selfRegister);
919 
920  // allow to modify the restart file name
921  std::string restartFilename{ constRestartFilename };
922 
923  std::stringstream adiosPathBase;
924  adiosPathBase << ADIOS_PATH_ROOT << restartStep << "/";
925  mThreadParams.adiosBasePath = adiosPathBase.str();
926  //mThreadParams.isCheckpoint = isCheckpoint;
927  mThreadParams.currentStep = restartStep;
928  mThreadParams.cellDescription = m_cellDescription;
929 
935  ADIOS_CMD(adios_read_init_method(ADIOS_READ_METHOD_BP,
936  mThreadParams.adiosComm,
937  "verbose=3;abort_on_error;"));
938 
939  /* if restartFilename is relative, prepend with restartDirectory */
940  if (!boost::filesystem::path(restartFilename).has_root_path())
941  {
942  restartFilename = restartDirectory + std::string("/") + restartFilename;
943  }
944 
945  std::stringstream strFname;
946  strFname << restartFilename << "_" << mThreadParams.currentStep << ".bp";
947 
948  // adios_read_open( fname, method, comm, lock_mode, timeout_sec )
949  log<picLog::INPUT_OUTPUT > ("ADIOS: open file: %1%") % strFname.str();
950 
951  // when reading in BG_AGGREGATE mode, adios can not distinguish between
952  // "file does not exist" and "stream is not (yet) available, so we
953  // test it our selves
954  if (!boost::filesystem::exists(strFname.str()))
955  throw std::runtime_error("ADIOS: File does not exist.");
956 
957  /* <0 sec: wait forever
958  * >=0 sec: return immediately if stream is not available */
959  float_32 timeout = 0.0f;
960  mThreadParams.fp = adios_read_open(strFname.str().c_str(),
961  ADIOS_READ_METHOD_BP, mThreadParams.adiosComm,
962  ADIOS_LOCKMODE_CURRENT, timeout);
963 
964  /* stream reading is tricky, see ADIOS manual section 8.11.1 */
965  while (adios_errno == err_file_not_found)
966  {
968 #if !defined(_WIN32)
969  /* give the file system 1s of peace and quiet */
970  usleep(1e6);
971 #endif
972  mThreadParams.fp = adios_read_open(strFname.str().c_str(),
973  ADIOS_READ_METHOD_BP, mThreadParams.adiosComm,
974  ADIOS_LOCKMODE_CURRENT, timeout);
975  }
976  if (adios_errno == err_end_of_stream )
977  /* could not read full stream */
978  throw std::runtime_error("ADIOS: Stream terminated too early: " +
979  std::string(adios_errmsg()) );
980  if (mThreadParams.fp == nullptr)
981  throw std::runtime_error("ADIOS: Error opening stream: " +
982  std::string(adios_errmsg()) );
983 
984  /* ADIOS types */
985  AdiosUInt32Type adiosUInt32Type;
986 
987  /* load number of slides to initialize MovingWindow */
988  log<picLog::INPUT_OUTPUT > ("ADIOS: (begin) read attr (%1% available)") %
989  mThreadParams.fp->nattrs;

991  int slideSize;
992  enum ADIOS_DATATYPES slidesType;
993  ADIOS_CMD(adios_get_attr( mThreadParams.fp,
994  (mThreadParams.adiosBasePath + std::string("sim_slides")).c_str(),
995  &slidesType,
996  &slideSize,
997  &slidesPtr ));
998 
999  uint32_t slides = *( (uint32_t*)slidesPtr );
1000  log<picLog::INPUT_OUTPUT > ("ADIOS: value of sim_slides = %1%") %
1001  slides;
1002 
1003  PMACC_ASSERT(slidesType == adiosUInt32Type.type);
1004  PMACC_ASSERT(slideSize == sizeof(uint32_t)); // uint32_t in bytes
1005 
1006  void* lastStepPtr = nullptr;
1007  int lastStepSize;
1008  enum ADIOS_DATATYPES lastStepType;
1009  ADIOS_CMD(adios_get_attr( mThreadParams.fp,
1010  (mThreadParams.adiosBasePath + std::string("iteration")).c_str(),
1011  &lastStepType,
1012  &lastStepSize,
1013  &lastStepPtr ));
1014  uint32_t lastStep = *( (uint32_t*)lastStepPtr );
1015  log<picLog::INPUT_OUTPUT > ("ADIOS: value of iteration = %1%") %
1016  lastStep;
1017 
1018  PMACC_ASSERT(lastStepType == adiosUInt32Type.type);
1019  PMACC_ASSERT(lastStep == restartStep);
1020 
1021  /* apply slides to set gpus to last/written configuration */
1022  log<picLog::INPUT_OUTPUT > ("ADIOS: Setting slide count for moving window to %1%") % slides;
1023  MovingWindow::getInstance().setSlideCounter(slides, restartStep);
1024 
1025  /* re-distribute the local offsets in y-direction
1026  * this will work for restarts with moving window still enabled
1027  * and restarts that disable the moving window
1028  * \warning enabling the moving window from a checkpoint that
1029  * had no moving window will not work
1030  */
1031  GridController<simDim> &gc = Environment<simDim>::get().GridController();
1032  gc.setStateAfterSlides(slides);
1033 
1034  /* set window for restart, complete global domain */
1035  mThreadParams.window = MovingWindow::getInstance().getDomainAsWindow(restartStep);
1036  mThreadParams.localWindowToDomainOffset = DataSpace<simDim>::create(0);
1037 
1038  /* load all fields */
1039  ForEach<FileCheckpointFields, LoadFields<bmpl::_1> > forEachLoadFields;
1040  forEachLoadFields(&mThreadParams);
1041 
1042  /* load all particles */
1043  ForEach<FileCheckpointParticles, LoadSpecies<bmpl::_1> > forEachLoadSpecies;
1044  forEachLoadSpecies(&mThreadParams, restartChunkSize);
1045 
1046  IdProvider<simDim>::State idProvState;
1047  ReadNDScalars<uint64_t, uint64_t>()(mThreadParams,
1048  "picongpu/idProvider/startId", &idProvState.startId,
1049  "maxNumProc", &idProvState.maxNumProc);
1050  ReadNDScalars<uint64_t>()(mThreadParams,
1051  "picongpu/idProvider/nextId", &idProvState.nextId);
1052  log<picLog::INPUT_OUTPUT > ("Setting next free id on current rank: %1%") % idProvState.nextId;
1053  IdProvider<simDim>::setState(idProvState);
1054 
1055  /* free memory allocated in ADIOS calls */
1056  free(slidesPtr);
1057  free(lastStepPtr);
1058 
1059  // avoid deadlock between not finished pmacc tasks and mpi calls in adios
1060  __getTransactionEvent().waitForFinished();
1061 
1062  /* clean shut down: close file and finalize */
1063  adios_release_step( mThreadParams.fp );
1064  ADIOS_CMD(adios_read_close( mThreadParams.fp ));
1065  ADIOS_CMD(adios_read_finalize_method(ADIOS_READ_METHOD_BP));
1066  }

1149  const uint32_t nComponents, const std::string name,
1150  void *ptr)
1151  {

1153  name % nComponents % ptr;
1154 
1155  const bool fieldTypeCorrect( boost::is_same<ComponentType, float_X>::value );

1177  {

1179  {

1181  (z + guardZ) * plane_full_size +
1182  (y + field_guard[1]) * field_full[0] * nComponents;

1185  z * plane_no_guard_size +
