43  void getCurrentSlideInfo(uint32_t currentStep, bool *doSlide, float_64 *offsetFirstGPU)
44  {
45  if (doSlide)
46  *doSlide = false;
47 
48  if (offsetFirstGPU)
49  *offsetFirstGPU = 0.0;
50 
51  if (slidingWindowEnabled)
52  {
53  /* Sliding stayed enabled but if we reach the end step where we should stop sliding
54  * the moving window is freezed.
55  * All offsets will stay constant until the end of the simulation.
56  */
57  if (currentStep >= endSlidingOnStep)
58  currentStep = endSlidingOnStep;
59 
60  const SubGrid<simDim>& subGrid = Environment<simDim>::get().SubGrid();
61 
62  /* speed of the moving window */
63  const float_64 windowMovingSpeed = float_64(SPEED_OF_LIGHT);
64 
65  /* defines in which direction the window moves
66  *
67  * 0 == x, 1 == y , 2 == z direction
68  *
69  * note: currently only y direction is supported
70  */
71  const uint32_t moveDirection = 1;
72 
73  /* the moving window is smaller than the global domain by exactly one
74  * GPU (local domain size)
75  * \todo calculation of the globalWindowSizeInMoveDirection is constant should be
76  * only done once in it's own central object/api
77  */
78  const uint32_t globalWindowSizeInMoveDirection =
79  subGrid.getGlobalDomain().size[moveDirection] - subGrid.getLocalDomain().size[moveDirection];
80 
81  const uint32_t gpuNumberOfCellsInMoveDirection = subGrid.getLocalDomain().size[moveDirection];
82 
83  /* unit PIConGPU length */
84  const float_64 cellSizeInMoveDirection = float_64(cellSize[moveDirection]);
85 
86  const float_64 deltaWayPerStep = (windowMovingSpeed * float_64(DELTA_T));
87 
88  /* How many cells the virtual particle with speed of light is pushed forward
89  * at the begin of the simulation.
90  * The number of cells is round up thus we avoid window moves and slides
91  * depends on half cells.
92  */
93  const uint32_t virtualParticleInitialStartCell = math::ceil(
94  float_64(globalWindowSizeInMoveDirection) * (float_64(1.0) - movePoint)
95  );
96 
97  /* Is the time step when the virtual particle **passed** the GPU next to the last
98  * in the current to the next step
99  */
100  const uint32_t firstSlideStep = math::ceil(
101  float_64(subGrid.getGlobalDomain().size[moveDirection] - virtualParticleInitialStartCell) *
102  cellSizeInMoveDirection / deltaWayPerStep
103  ) - 1;
104 
105  /* way which the virtual particle must move before the window begins
106  * to move the first time [in pic length] */
107  const float_64 wayToFirstMove =
108  float_64(globalWindowSizeInMoveDirection - virtualParticleInitialStartCell) *
109  cellSizeInMoveDirection;
110  /* Is the time step when the virtual particle **passed** the moving window
111  * in the current to the next step
112  * Signed type of firstMoveStep to allow for edge case movePoint = 0.0
113  * for a moving window right from the start of the simulation.
114  */
115  const int32_t firstMoveStep = math::ceil(
116  wayToFirstMove / deltaWayPerStep
117  ) - 1;
118 
119  if (firstMoveStep <= int32_t(currentStep) )
120  {
121  /* calculate the current position of the virtual particle */
122  const float_64 virtualParticleWayPassed =
123  deltaWayPerStep * float_64(currentStep);
124  const uint32_t virtualParticleWayPassedInCells = uint32_t(
125  math::floor(virtualParticleWayPassed / cellSizeInMoveDirection)
126  );
127  const uint32_t virtualParticlePositionInCells =
128  virtualParticleWayPassedInCells + virtualParticleInitialStartCell;
129 
130  /* calculate the position of the virtual particle after the current step is calculated */
131  const float_64 nextVirtualParticleWayPassed =
132  deltaWayPerStep * float_64(currentStep + 1);
133  const uint32_t nextVirtualParticleWayPassedInCells =
134  uint32_t(math::floor(nextVirtualParticleWayPassed / cellSizeInMoveDirection));
135  /* This position is used to detect the point in time where the virtual particle
136  * moves over a GPU border.
137  */
138  const uint32_t nextVirtualParticlePositionInCells =
139  nextVirtualParticleWayPassedInCells + virtualParticleInitialStartCell;
140 
141  /* within the to be simulated time step (currentStep -> currentStep+1)
142  * the virtual particle will have reached at least the position
143  * of the cell behind the end of the initial global domain
144  * (also true for all later time steps)
145  */
146  const bool endOfInitialGlobalDomain = firstSlideStep <= currentStep;
147 
148  /* virtual particle will pass a GPU border during the current
149  * (to be simulated) time step
150  */
151  const bool virtualParticlePassesGPUBorder =
152  (nextVirtualParticlePositionInCells % gpuNumberOfCellsInMoveDirection) <
153  (virtualParticlePositionInCells % gpuNumberOfCellsInMoveDirection);
154 
155  if (endOfInitialGlobalDomain && virtualParticlePassesGPUBorder)
156  {
157  incrementSlideCounter(currentStep);
158  if (doSlide)
159  *doSlide = true;
160  }
161 
162  /* valid range for the offset is [0;GPU number of cells in move direction) */
163  if (offsetFirstGPU)
164  {
165  /* since the moving window in PIConGPU always starts on the
166  * first plane (3D) / row (2D) of GPUs in move direction, this
167  * calculation is equal to the globalWindow.offset in move direction
168  *
169  * note: also works with windowMovingSpeed > c
170  */
171  *offsetFirstGPU = nextVirtualParticlePositionInCells % gpuNumberOfCellsInMoveDirection;
172  }
173  }
174  }

186  void incrementSlideCounter(const uint32_t currentStep)
187  {
188  // do not slide twice in one simulation step
189  if (isSlidingWindowActive( currentStep ) && lastSlideStep < currentStep)
190  {
191  slideCounter++;
192  lastSlideStep = currentStep;
193  }
194  }

221  void setEndSlideOnStep(int32_t step)
222  {
223  // maybe we have a underflow in the cast, this is fine because it results in a very large number
224  const uint32_t maxSlideStep = static_cast<uint32_t>(step);
225  if ( maxSlideStep < lastSlideStep)
226  throw std::runtime_error("It is not allowed to stop the moving window in the past.");
227 
228  endSlidingOnStep = maxSlideStep;
229 
230  static bool firstCall = true;
231  /* Disable or enable sliding window only in the first call.
232  * Later changes of step will not influence if the sliding window is activated.
233  */
234  if (firstCall && endSlidingOnStep != 0u)
235  slidingWindowEnabled = true;
236 
237  firstCall = false;
238  }

334  Window getWindow(uint32_t currentStep)
335  {
336  const SubGrid<simDim>& subGrid = Environment<simDim>::get().SubGrid();
337 
338  /* Without moving window, the selected window spans the whole global domain.
339  * \see https://github.com/ComputationalRadiationPhysics/picongpu/wiki/PIConGPU-domain-definitions
340  *
341  * The window's global offset is therefore zero inside the global domain.
342  * The window's global and local size are equal to the SubGrid quantities.
343  * The local window offset is the offset within the global window which
344  * is equal to the local domain offset of the GPU.
345  */
346  Window window;
347  window.localDimensions = subGrid.getLocalDomain();
348  window.globalDimensions = Selection<simDim>(subGrid.getGlobalDomain().size);
349 
350  /* moving window can only slide in y direction */
351  if (slidingWindowEnabled)
352  {
353  /* the moving window is smaller than the global domain by exactly one
354  * GPU (local domain size) in moving (y) direction
355  */
356  window.globalDimensions.size.y() -= subGrid.getLocalDomain().size.y();
357 
358  float_64 offsetFirstGPU = 0.0;
359  getCurrentSlideInfo(currentStep, nullptr, &offsetFirstGPU);
360 
361  /* while moving, the windows global offset within the global domain is between 0
362  * and smaller than the local domain's size in y.
363  */
364  window.globalDimensions.offset.y() = offsetFirstGPU;
365 
366  /* set top/bottom if there are no communication partners
367  * for this GPU in the respective direction */
368  const Mask comm_mask = Environment<simDim>::get().GridController().getCommunicationMask();
369  const bool isTopGpu = !comm_mask.isSet(TOP);
370  const bool isBottomGpu = !comm_mask.isSet(BOTTOM);
371 
372  if (isTopGpu)
373  {
374  /* the windows local offset within the global window is reduced
375  * by the global window offset within the global domain
376  */
377  window.localDimensions.size.y() -= offsetFirstGPU;
378  }
379  else
380  {
381  window.localDimensions.offset.y() = subGrid.getLocalDomain().offset.y() - offsetFirstGPU;
382  if (isBottomGpu)
383  {
384  window.localDimensions.size.y() = offsetFirstGPU;
385  }
386  }
387  }
388 
389  return window;
390  }
