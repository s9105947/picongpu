80  // restart simulation by loading from persistent data
81  // the simulation will start after restartStep
82  log<picLog::SIMULATION_STATE > ("Restarting simulation from timestep %1% in directory '%2%'") %
83  restartStep % restartDirectory;
84 
85  Environment<>::get().PluginConnector().restartPlugins(restartStep, restartDirectory);
86  __getTransactionEvent().waitForFinished();
87 
88  CUDA_CHECK(cudaDeviceSynchronize());
89  CUDA_CHECK(cudaGetLastError());
90 
91  GridController<simDim> &gc = Environment<simDim>::get().GridController();
92 
93  // avoid deadlock between not finished pmacc tasks and MPI_Barrier
94  __getTransactionEvent().waitForFinished();
95  /* can be spared for better scalings, but guarantees the user
96  * that the restart was successful */
97  MPI_CHECK(MPI_Barrier(gc.getCommunicator().getMPIComm()));
98 
99  log<picLog::SIMULATION_STATE > ("Loading from persistent data finished");
100  }

107  template<typename T_Species = bmpl::_1>
108  struct LogOmegaP
109  {
110  void operator()()
111  {
112  /* The omega_p calculation is based on species' densityRatio
113  * relative to the BASE_DENSITY. Thus, it is only accurate
114  * for species with macroparticles sampled by density,
115  * but not necessarily for derived ones.
116  */
117  using FrameType = typename T_Species::FrameType;
118  const float_32 charge = frame::getCharge<FrameType>();
119  const float_32 mass = frame::getMass<FrameType>();
120  const auto densityRatio = traits::GetDensityRatio< T_Species >::type::getValue( );
121  const auto density = BASE_DENSITY * densityRatio;
122  log<picLog::PHYSICS >("species %2%: omega_p * dt <= 0.1 ? %1%") %
123  (sqrt(density * charge / mass * charge / EPS0) * DELTA_T) %
124  FrameType::getName();
125  }
126  };
