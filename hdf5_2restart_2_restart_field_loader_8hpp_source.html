48 class RestartFieldLoader
49 {
50 public:
51  template<class Data>
52  static void loadField(Data& field, const uint32_t numComponents, std::string objectName, ThreadParams *params)
53  {
54  log<picLog::INPUT_OUTPUT > ("Begin loading field '%1%'") % objectName;
55  const DataSpace<simDim> field_guard = field.getGridLayout().getGuard();
56 
57  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(params->currentStep);
58  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
59 
60  field.getHostBuffer().setValue(float3_X::create(0.0));
61 
62  const std::string name_lookup[] = {"x", "y", "z"};
63 
64  /* globalSlideOffset due to gpu slides between origin at time step 0
65  * and origin at current time step
66  * ATTENTION: splash offset are globalSlideOffset + picongpu offsets
67  */
68  DataSpace<simDim> globalSlideOffset;
69  globalSlideOffset.y() = numSlides * localDomain.size.y();
70 
71  Dimensions domain_offset(0, 0, 0);
72  for (uint32_t d = 0; d < simDim; ++d)
73  domain_offset[d] = localDomain.offset[d] + globalSlideOffset[d];
74 
75  if (Environment<simDim>::get().GridController().getPosition().y() == 0)
76  domain_offset[1] += params->window.globalDimensions.offset.y();
77 
78  Dimensions local_domain_size;
79  for (uint32_t d = 0; d < simDim; ++d)
80  local_domain_size[d] = params->window.localDimensions.size[d];
81 
82  // avoid deadlock between not finished pmacc tasks and mpi calls in splash/HDF5
83  __getTransactionEvent().waitForFinished();
84 
85  auto destBox = field.getHostBuffer().getDataBox();
86  for (uint32_t i = 0; i < numComponents; ++i)
87  {
88  // Read the subdomain which belongs to our mpi position.
89  // The total grid size must match the grid size of the stored data.
90  log<picLog::INPUT_OUTPUT > ("Read from domain: offset=%1% size=%2%") %
91  domain_offset.toString() % local_domain_size.toString();
92  DomainCollector::DomDataClass data_class;
93  DataContainer *field_container =
94  params->dataCollector->readDomain(params->currentStep,
95  (std::string("fields/") + objectName +
96  std::string("/") + name_lookup[i]).c_str(),
97  Domain(domain_offset, local_domain_size),
98  &data_class);
99 
100  int elementCount = params->window.localDimensions.size.productOfComponents();
101 
102  for (int linearId = 0; linearId < elementCount; ++linearId)
103  {
104  /* calculate index inside the moving window domain which is located on the local grid*/
105  DataSpace<simDim> destIdx = DataSpaceOperations<simDim>::map(params->window.localDimensions.size, linearId);
106  /* jump over guard and local sliding window offset*/
107  destIdx += field_guard + params->localWindowToDomainOffset;
108 
109  destBox(destIdx)[i] = ((float_X*) (field_container->getIndex(0)->getData()))[linearId];
110  }
111 
112  delete field_container;
113  }
114 
115  field.hostToDevice();
116 
117  __getTransactionEvent().waitForFinished();
118 
119  log<picLog::INPUT_OUTPUT > ("Read from domain: offset=%1% size=%2%") %
120  domain_offset.toString() % local_domain_size.toString();
121  log<picLog::INPUT_OUTPUT > ("Finished loading field '%1%'") % objectName;
122  }

124  template<class Data>
125  static void cloneField(Data& fieldDest, Data& fieldSrc, std::string objectName)
126  {
127  log<picLog::INPUT_OUTPUT > ("Begin cloning field '%1%'") % objectName;
128  DataSpace<simDim> field_grid = fieldDest.getGridLayout().getDataSpace();
129 
130  size_t elements = field_grid.productOfComponents();
131  float3_X *ptrDest = fieldDest.getHostBuffer().getDataBox().getPointer();
132  float3_X *ptrSrc = fieldSrc.getHostBuffer().getDataBox().getPointer();
133 
134  for (size_t k = 0; k < elements; ++k)
135  {
136  ptrDest[k] = ptrSrc[k];
137  }
138 
139  fieldDest.hostToDevice();
140 
141  __getTransactionEvent().waitForFinished();
142 
143  log<picLog::INPUT_OUTPUT > ("Finished cloning field '%1%'") % objectName;
144  }
