86  template< typename T_Particle >
87  HDINLINE bool isParticleInsideRange( T_Particle const & particle, DataSpace< simDim > const & globalSuperCellOffset ) const
88  {
89  using SuperCellSize = typename T_Particle::SuperCellSize;
90 
91  int const particleCellIdx = particle[ localCellIdx_ ];
92  DataSpace< simDim > const cellInSuperCell( DataSpaceOperations< simDim >::
93  template map< SuperCellSize >( particleCellIdx ) );
94  DataSpace< simDim > const globalParticleOffset(
95  globalSuperCellOffset +
96  cellInSuperCell
97  );
98 
99  float_X const relativePosition = float_X( globalParticleOffset[ Params::dimension ] ) /
100  float_X( m_globalDomainSize[ Params::dimension ] );
101 
102  return ( Params::lowerBound <= relativePosition &&
103  relativePosition < Params::upperBound );
104  }
