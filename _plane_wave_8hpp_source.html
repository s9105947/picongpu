93  template< typename T_Acc >
94  HDINLINE
95  void operator( )(
96  T_Acc const &,
97  DataSpace< simDim > const & cellIndexInSuperCell
98  )
99  {
100  // coordinate system to global simulation as origin
101  DataSpace< simDim > const localCell(
102  cellIndexInSuperCell +
103  m_superCellToLocalOriginCellOffset
104  );
105 
106  if( Unitless::initPlaneY != 0 ) // compile time if
107  {
108  /* If the laser is not initialized in the first cell we emit a
109  * negatively and positively propagating wave. Therefore we need to multiply the
110  * amplitude with a correction factor depending of the cell size in
111  * propagation direction.
112  * The negatively propagating wave is damped by the absorber.
113  *
114  * The `correctionFactor` assume that the wave is moving in y direction.
115  */
116  auto const correctionFactor = ( SPEED_OF_LIGHT * DELTA_T ) / CELL_HEIGHT * 2._X;
117 
118  // jump over the guard of the electric field
119  m_dataBoxE( localCell + SuperCellSize::toRT() * GuardSize::toRT() ) += correctionFactor * m_elong;
120  }
121  else
122  {
123  // jump over the guard of the electric field
124  m_dataBoxE( localCell + SuperCellSize::toRT() * GuardSize::toRT() ) = m_elong;
125  }
126  }

144  HINLINE PlaneWave( uint32_t currentStep ) :
145  phase( 0.0_X )
146  {
147  // get data
148  DataConnector & dc = Environment< >::get( ).DataConnector( );
149  dataBoxE = dc.get< FieldE >(
150  FieldE::getName(),
151  true
152  )->getDeviceDataBox();
153 
154  // get meta data for offsets
155  SubGrid< simDim > const & subGrid = Environment< simDim >::get().SubGrid();
156  // const DataSpace< simDim > totalCellOffset( subGrid.getGlobalDomain().offset );
157  DataSpace< simDim > const globalCellOffset( subGrid.getLocalDomain().offset );
158  DataSpace< simDim > const halfSimSize( subGrid.getGlobalDomain().size / 2 );
159 
160  // transform coordinate system to center of global simulation as origin [cells]
161  offsetToTotalDomain = /* totalCellOffset + */ globalCellOffset - halfSimSize;
162 
163  // @todo reset origin of direction of moving window
164  // offsetToTotalDomain.y() = 0
165 
166  float_64 const runTime = DELTA_T * currentStep - Unitless::laserTimeShift;
167 
168  elong = float3_X::create( 0.0 );
169 
170  float_64 envelope = float_64( Unitless::AMPLITUDE );
171 
172  float_64 const mue = 0.5 * Unitless::RAMP_INIT * Unitless::PULSE_LENGTH;
173 
174  float_64 const w = 2.0 * PI * Unitless::f;
175  float_64 const tau = Unitless::PULSE_LENGTH * math::sqrt( 2.0 );
176 
177  float_64 const endUpramp = mue;
178  float_64 const startDownramp = mue + Unitless::LASER_NOFOCUS_CONSTANT;
179 
180  float_64 integrationCorrectionFactor = 0.0;
181 
182  if( runTime > startDownramp )
183  {
184  // downramp = end
185  float_64 const exponent = ( runTime - startDownramp ) / tau;
186  envelope *= exp( -0.5 * exponent * exponent );
187  integrationCorrectionFactor = ( runTime - startDownramp )/ ( w * tau * tau );
188  }
189  else if( runTime < endUpramp )
190  {
191  // upramp = start
192  float_64 const exponent = ( runTime - endUpramp ) / tau;
193  envelope *= exp( -0.5 * exponent * exponent );
194  integrationCorrectionFactor = ( runTime - endUpramp )/ ( w * tau * tau );
195  }
196 
197  float_64 const timeOszi = runTime - endUpramp;
198  float_64 const t_and_phase = w * timeOszi + Unitless::LASER_PHASE;
199  // to understand both components [sin(...) + t/tau^2 * cos(...)] see description above
200  if( Unitless::Polarisation == Unitless::LINEAR_X )
201  {
202  elong.x() = float_X( envelope * ( math::sin( t_and_phase )
203  + math::cos( t_and_phase ) * integrationCorrectionFactor ) );
204  }
205  else if( Unitless::Polarisation == Unitless::LINEAR_Z)
206  {
207  elong.z() = float_X( envelope * ( math::sin( t_and_phase )
208  + math::cos( t_and_phase ) * integrationCorrectionFactor ) );
209  }
210  else if( Unitless::Polarisation == Unitless::CIRCULAR )
211  {
212  elong.x() = float_X( envelope / math::sqrt(2.0) * ( math::sin( t_and_phase )
213  + math::cos( t_and_phase ) * integrationCorrectionFactor));
214  elong.z() = float_X( envelope / math::sqrt(2.0) * ( math::cos( t_and_phase )
215  - math::sin( t_and_phase ) * integrationCorrectionFactor ) );
216  }
217  }
