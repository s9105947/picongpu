44 template< typename T_Identifier>
45 struct ParticleAttributeSize
46 {
52  HINLINE void operator()(
53  ThreadParams* params,
54  const std::string speciesGroup,
55  const uint64_t elements,
56  const uint64_t globalElements,
57  const uint64_t globalOffset)
58  {
59 
60  typedef T_Identifier Identifier;
61  typedef typename pmacc::traits::Resolve<Identifier>::type::type ValueType;
62  const uint32_t components = GetNComponents<ValueType>::value;
63  typedef typename GetComponentsType<ValueType>::type ComponentType;
64 
65  params->adiosGroupSize += elements * components * sizeof(ComponentType);
66 
67  /* define adios var for particle attribute */
68  PICToAdios<ComponentType> adiosType;
69  PICToAdios<float_X> adiosFloatXType;
70  PICToAdios<float_64> adiosDoubleType;
71  PICToAdios<uint32_t> adiosUInt32Type;
72 
73  const std::string name_lookup[] = {"x", "y", "z"};
74 
75  OpenPMDName<T_Identifier> openPMDName;
76  const std::string recordPath( params->adiosBasePath +
77  std::string(ADIOS_PATH_PARTICLES) + speciesGroup + openPMDName() );
78 
79  // get the SI scaling, dimensionality and weighting of the attribute
80  OpenPMDUnit<T_Identifier> openPMDUnit;
81  std::vector<float_64> unit = openPMDUnit();
82  OpenPMDUnitDimension<T_Identifier> openPMDUnitDimension;
83  std::vector<float_64> unitDimension = openPMDUnitDimension();
84  const bool macroWeightedBool = MacroWeighted<T_Identifier>::get();
85  const uint32_t macroWeighted = (macroWeightedBool ? 1 : 0);
86  const float_64 weightingPower = WeightingPower<T_Identifier>::get();
87 
88  PMACC_ASSERT(unit.size() == components); // unitSI for each component
89  PMACC_ASSERT(unitDimension.size() == 7); // seven openPMD base units
90 
91  for (uint32_t d = 0; d < components; d++)
92  {
93  std::stringstream datasetName;
94  datasetName << recordPath;
95  if (components > 1)
96  datasetName << "/" << name_lookup[d];
97 
98  const char* path = nullptr;
99  int64_t adiosParticleAttrId = defineAdiosVar<DIM1>(
100  params->adiosGroupHandle,
101  datasetName.str().c_str(),
102  path,
103  adiosType.type,
104  pmacc::math::UInt64<DIM1>(elements),
105  pmacc::math::UInt64<DIM1>(globalElements),
106  pmacc::math::UInt64<DIM1>(globalOffset),
107  true,
108  params->adiosCompression);
109 
110  params->adiosParticleAttrVarIds.push_back(adiosParticleAttrId);
111 
112  /* already add the unitSI and further attribute so `adios_group_size`
113  * calculates the reservation for the buffer correctly */
114 
115  /* check if this attribute actually has a unit (unit.size() == 0 is no unit) */
116  if (unit.size() >= (d + 1))
117  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
118  "unitSI", datasetName.str().c_str(),
119  adiosDoubleType.type, 1, &unit.at(d) ));
120  }
121 
122  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
123  "unitDimension", recordPath.c_str(),
124  adiosDoubleType.type, 7, &(*unitDimension.begin()) ));
125 
126  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
127  "macroWeighted", recordPath.c_str(),
128  adiosUInt32Type.type, 1, (void*)&macroWeighted ));
129 
130  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
131  "weightingPower", recordPath.c_str(),
132  adiosDoubleType.type, 1, (void*)&weightingPower ));
133 
136  const float_X timeOffset = 0.0;
137  ADIOS_CMD(adios_define_attribute_byvalue(params->adiosGroupHandle,
138  "timeOffset", recordPath.c_str(),
139  adiosFloatXType.type, 7, (void*)&timeOffset ));
140 
141  }
142 
143 };
