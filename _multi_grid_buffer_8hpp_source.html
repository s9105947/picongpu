247  void init(GridLayout<DIM> gridLayout, bool sizeOnDevice)
248  {
249  DataSpace<DIM> blobOffset;
250  blobOffset[DIM - 1] = gridLayout.getDataSpace()[DIM - 1];
251 
252  DataSpace<DIM> blobSize = gridLayout.getDataSpace() + blobOffset * (BufferNames::Count - 1);
253 
254  blobDeviceBuffer = new DeviceBufferIntern<Type_, DIM > (blobSize, false);
255  blobHostBuffer = new HostBufferIntern<Type_, DIM > (blobSize);
256 
257  for (uint32_t i = 0; i < BufferNames::Count; ++i)
258  {
259  DataSpace<DIM> offset = blobOffset*i;
260  gridBuffers[i] = new GridBuffer<Type, DIM > (
261  *blobHostBuffer, offset,
262  *blobDeviceBuffer, offset,
263  gridLayout, sizeOnDevice);
264  }
