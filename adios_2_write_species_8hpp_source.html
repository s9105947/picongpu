66 template< typename T_SpeciesFilter >
67 struct WriteSpecies
68 {
69 public:
70 
71  typedef typename T_SpeciesFilter::Species ThisSpecies;
72  typedef typename ThisSpecies::FrameType FrameType;
73  typedef typename FrameType::ParticleDescription ParticleDescription;
74  typedef typename FrameType::ValueTypeSeq ParticleAttributeList;
75 
76  /* delete multiMask and localCellIdx in adios particle*/
77  typedef bmpl::vector<multiMask,localCellIdx> TypesToDelete;
78  typedef typename RemoveFromSeq<ParticleAttributeList, TypesToDelete>::type ParticleCleanedAttributeList;
79 
80  /* add totalCellIdx for adios particle*/
81  typedef typename MakeSeq<
82  ParticleCleanedAttributeList,
83  totalCellIdx
84  >::type ParticleNewAttributeList;
85 
86  typedef
87  typename ReplaceValueTypeSeq<ParticleDescription, ParticleNewAttributeList>::type
88  NewParticleDescription;
89 
90  typedef Frame<OperatorCreateVectorBox, NewParticleDescription> AdiosFrameType;
91 
92  template<typename Space>
93  HINLINE void operator()(ThreadParams* params,
94  const Space particleOffset)
95  {
96  log<picLog::INPUT_OUTPUT > ("ADIOS: (begin) write species: %1%") % T_SpeciesFilter::getName();
97  DataConnector &dc = Environment<>::get().DataConnector();
98  /* load particle without copy particle data to host */
99  auto speciesTmp = dc.get< ThisSpecies >( ThisSpecies::FrameType::getName(), true );
100 
101  /* count total number of particles on the device */
102  log<picLog::INPUT_OUTPUT > ("ADIOS: (begin) count particles: %1%") % T_SpeciesFilter::getName();
103  // enforce that the filter interface is fulfilled
104  particles::filter::IUnary< typename T_SpeciesFilter::Filter > particleFilter{ params->currentStep };
105  uint64_cu totalNumParticles = 0;
106  totalNumParticles = pmacc::CountParticles::countOnDevice < CORE + BORDER > (
107  *speciesTmp,
108  *(params->cellDescription),
109  params->localWindowToDomainOffset,
110  params->window.localDimensions.size,
111  particleFilter);
112  log<picLog::INPUT_OUTPUT > ("ADIOS: ( end ) count particles: %1% = %2%") % T_SpeciesFilter::getName() % totalNumParticles;
113 
114  AdiosFrameType hostFrame;
115 
116  /* malloc host memory */
117  log<picLog::INPUT_OUTPUT > ("ADIOS: (begin) malloc host memory: %1%") % T_SpeciesFilter::getName();
118  ForEach<typename AdiosFrameType::ValueTypeSeq, MallocHostMemory<bmpl::_1> > mallocMem;
119  mallocMem(forward(hostFrame), totalNumParticles);
120  log<picLog::INPUT_OUTPUT > ("ADIOS: ( end ) malloc host memory: %1%") % T_SpeciesFilter::getName();
121 
122  if (totalNumParticles > 0)
123  {
124  log<picLog::INPUT_OUTPUT > ("ADIOS: (begin) copy particle host (with hierarchy) to host (without hierarchy): %1%") % T_SpeciesFilter::getName();
125  typedef bmpl::vector< typename GetPositionFilter<simDim>::type > usedFilters;
126  typedef typename FilterFactory<usedFilters>::FilterType MyParticleFilter;
127  MyParticleFilter filter;
128  /* activate filter pipeline if moving window is activated */
129  filter.setStatus(MovingWindow::getInstance().isSlidingWindowActive());
130  filter.setWindowPosition(params->localWindowToDomainOffset,
131  params->window.localDimensions.size);
132 
133  DataConnector &dc = Environment<>::get().DataConnector();
134 #if( PMACC_CUDA_ENABLED == 1 )
135  auto mallocMCBuffer = dc.get< MallocMCBuffer< DeviceHeap > >( MallocMCBuffer< DeviceHeap >::getName(), true );
136 #endif
137  int globalParticleOffset = 0;
138  AreaMapping < CORE + BORDER, MappingDesc > mapper(*(params->cellDescription));
139 
140  pmacc::particles::operations::ConcatListOfFrames<simDim> concatListOfFrames(mapper.getGridDim());
141 
142 #if( PMACC_CUDA_ENABLED == 1 )
143  auto particlesBox = speciesTmp->getHostParticlesBox( mallocMCBuffer->getOffset() );
144 #else
145  /* This separate code path is only a workaround until MallocMCBuffer
146  * is alpaka compatible.
147  *
148  * @todo remove this workaround: we know that we are allowed to access the
149  * device memory directly.
150  */
151  auto particlesBox = speciesTmp->getDeviceParticlesBox( );
152  /* Notify to the event system that the particles box is used on the host.
153  *
154  * @todo remove this workaround
155  */
156  __startOperation(ITask::TASK_HOST);
157 
158 #endif
159  concatListOfFrames(
160  globalParticleOffset,
161  hostFrame,
162  particlesBox,
163  filter,
164  particleOffset, /*relative to data domain (not to physical domain)*/
165  totalCellIdx_,
166  mapper,
167  particleFilter
168  );
169 #if( PMACC_CUDA_ENABLED == 1 )
170  dc.releaseData( MallocMCBuffer< DeviceHeap >::getName() );
171 #endif
172  /* this costs a little bit of time but adios writing is slower */
173  PMACC_ASSERT((uint64_cu) globalParticleOffset == totalNumParticles);
174  }
175  /* dump to adios file */
176  ForEach<typename AdiosFrameType::ValueTypeSeq, adios::ParticleAttribute<bmpl::_1> > writeToAdios;
177  writeToAdios(params, forward(hostFrame), totalNumParticles);
178 
179  /* free host memory */
180  ForEach<typename AdiosFrameType::ValueTypeSeq, FreeHostMemory<bmpl::_1> > freeMem;
181  freeMem(forward(hostFrame));
182  log<picLog::INPUT_OUTPUT > ("ADIOS: ( end ) writing species: %1%") % T_SpeciesFilter::getName();
183 
184  /* write species counter table to adios file */
185  log<picLog::INPUT_OUTPUT > ("ADIOS: (begin) writing particle index table for %1%") % T_SpeciesFilter::getName();
186  {
187  GridController<simDim>& gc = Environment<simDim>::get().GridController();
188 
189  const size_t pos_offset = 2;
190 
191  /* particlesMetaInfo = (num particles, scalar position, particle offset x, y, z) */
192  uint64_t particlesMetaInfo[5] = {totalNumParticles, gc.getScalarPosition(), 0, 0, 0};
193  for (size_t d = 0; d < simDim; ++d)
194  particlesMetaInfo[pos_offset + d] = particleOffset[d];
195 
196  /* prevent that top (y) gpus have negative value here */
197  if (gc.getPosition().y() == 0)
198  particlesMetaInfo[pos_offset + 1] = 0;
199 
200  if (particleOffset[1] < 0) // 1 == y
201  particlesMetaInfo[pos_offset + 1] = 0;
202 
203  int64_t adiosIndexVarId = *(params->adiosSpeciesIndexVarIds.begin());
204  params->adiosSpeciesIndexVarIds.pop_front();
205  ADIOS_CMD(adios_write_byid(params->adiosFileHandle, adiosIndexVarId, particlesMetaInfo));
206  }
207  log<picLog::INPUT_OUTPUT > ("ADIOS: ( end ) writing particle index table for %1%") % T_SpeciesFilter::getName();
208  }
209 };
