57  template<typename FrameType>
58  HINLINE void operator()(
59  ThreadParams* params,
60  FrameType& frame,
61  const std::string particlePath,
62  const uint64_t particlesOffset,
63  const uint64_t elements)
64  {
65 
66  typedef T_Identifier Identifier;
67  typedef typename pmacc::traits::Resolve<Identifier>::type::type ValueType;
68  const uint32_t components = GetNComponents<ValueType>::value;
69  typedef typename GetComponentsType<ValueType>::type ComponentType;
70 
71  log<picLog::INPUT_OUTPUT > ("ADIOS: ( begin ) load species attribute: %1%") % Identifier::getName();
72 
73  const std::string name_lookup[] = {"x", "y", "z"};
74 
75  ComponentType* tmpArray = nullptr;
76  if( elements > 0 )
77  tmpArray = new ComponentType[elements];
78 
79  // dev assert!
80  if( elements > 0 )
81  PMACC_ASSERT(tmpArray);
82 
83  for (uint32_t n = 0; n < components; ++n)
84  {
85  OpenPMDName<T_Identifier> openPMDName;
86  std::stringstream datasetName;
87  datasetName << particlePath << openPMDName();
88  if (components > 1)
89  datasetName << "/" << name_lookup[n];
90 
91  ValueType* dataPtr = frame.getIdentifier(Identifier()).getPointer();
92 
93  ADIOS_VARINFO* varInfo = adios_inq_var( params->fp, datasetName.str().c_str() );
94  // it's possible to aquire the local block with that call again and
95  // the local elements to-be-read, but the block-ID must be known (MPI rank?)
96  //ADIOS_CMD(adios_inq_var_blockinfo( params->fp, varInfo ));
97 
98  ADIOS_SELECTION* sel = adios_selection_boundingbox( 1, &particlesOffset, &elements );
99 
102  if( elements > 0 )
103  {
104  // avoid deadlock between not finished pmacc tasks and mpi calls in adios
105  __getTransactionEvent().waitForFinished();
106  ADIOS_CMD(adios_schedule_read( params->fp,
107  sel,
108  datasetName.str().c_str(),
109  0, /* from_step (not used in streams) */
110  1, /* nsteps to read (must be 1 for stream) */
111  (void*)tmpArray ));
112  }
113 
116  ADIOS_CMD(adios_perform_reads( params->fp, 1 ));
117 
118  log<picLog::INPUT_OUTPUT > ("ADIOS: Did read %1% local of %2% global elements for %3%") %
119  elements % varInfo->dims[0] % datasetName.str();
120 
121  /* copy component from temporary array to array of structs */
122  #pragma omp parallel for
123  for (size_t i = 0; i < elements; ++i)
124  {
125  ComponentType& ref = ((ComponentType*) dataPtr)[i * components + n];
126  ref = tmpArray[i];
127  }
128 
129  adios_selection_delete( sel );
130  adios_free_varinfo( varInfo );
131  }
132  __deleteArray(tmpArray);
133 
134  log<picLog::INPUT_OUTPUT > ("ADIOS: ( end ) load species attribute: %1%") %
135  Identifier::getName();
136  }
137 
138 };
