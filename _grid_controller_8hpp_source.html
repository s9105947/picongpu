255  void updateDomainOffset(size_t numSlides = 1)
256  {
257  /* if we slide we must change our localDomain.offset of the simulation
258  * (only change slide direction Y)
259  */
260  int gpuOffset_y = this->getPosition().y();
261  const SubGrid<DIM>& subGrid = Environment<DIM>::get().SubGrid();
262  DataSpace<DIM> localDomainOffset(subGrid.getLocalDomain().offset);
263  DataSpace<DIM> globalDomainOffset(subGrid.getGlobalDomain().offset);
264  /* this is allowed in the case that we use sliding window
265  * because size in Y direction is the same for all gpus domains
266  */
267  localDomainOffset.y() = gpuOffset_y * subGrid.getLocalDomain().size.y();
268  globalDomainOffset.y() += numSlides * subGrid.getLocalDomain().size.y();
269 
270  Environment<DIM>::get().SubGrid().setLocalDomainOffset(localDomainOffset);
271  Environment<DIM>::get().SubGrid().setGlobalDomainOffset(globalDomainOffset);
272  }
