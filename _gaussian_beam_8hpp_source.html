78  HDINLINE float_X simpleLaguerre( const uint32_t n, const float_X x )
79  {
80  //Result for special case n == 0
81  if (n == 0) return 1.0_X;
82  uint32_t currentN = 1;
83  float_X laguerreNMinus1 = 1.0_X;
84  float_X laguerreN = 1.0_X - x;
85  float_X laguerreNPlus1( 0.0_X );
86  while (currentN < n)
87  {
88  //Core statement of the algorithm
89  laguerreNPlus1 = ( ( 2.0_X * float_X(currentN) + 1.0_X - x) * laguerreN - float_X(currentN) * laguerreNMinus1 ) / float_X(currentN + 1u);
90  //Advance by one order
91  laguerreNMinus1 = laguerreN;
92  laguerreN = laguerreNPlus1;
93  currentN++;
94  }
95  return laguerreN;
96  }

124  template< typename T_Acc >
125  HDINLINE
126  void operator( )(
127  T_Acc const &,
128  DataSpace< simDim > const & cellIndexInSuperCell
129  )
130  {
131  // coordinate system to global simulation as origin
132  DataSpace< simDim > const localCell(
133  cellIndexInSuperCell +
134  m_superCellToLocalOriginCellOffset
135  );
136 
137  // transform coordinate system to center of x-z plane of initialization
138  constexpr uint8_t planeNormalDir = 1u;
139  DataSpace< simDim > offsetToCenterOfPlane( m_offsetToTotalDomain );
140  offsetToCenterOfPlane[ planeNormalDir ] = 0; // do not shift origin of plane normal
141  floatD_X const pos = precisionCast< float_X >( localCell + offsetToCenterOfPlane ) * cellSize.shrink< simDim >();
142  // @todo add half-cells via traits::FieldPosition< Solver::NumicalCellType, FieldE >()
143 
144  // transversal position only
145  floatD_X planeNoNormal = floatD_X::create( 1.0_X );
146  planeNoNormal[ planeNormalDir ] = 0.0_X;
147  float_X const r2 = math::abs2( pos * planeNoNormal );
148 
149  // calculate focus position relative to the laser initialization plane
150  float_X const focusPos = Unitless::FOCUS_POS - pos.y();
151 
152  // rayleigh length (in y-direction)
153  float_X const y_R = float_X( PI ) * Unitless::W0 * Unitless::W0 / Unitless::WAVE_LENGTH;
154 
155  // the radius of curvature of the beam's wavefronts
156  float_X const R_y = -focusPos * ( 1.0_X + ( y_R / focusPos )*( y_R / focusPos ) );
157 
158  // initialize temporary variables
159  float_X etrans( 0.0_X );
160  float_X etrans_norm( 0.0_X );
161  PMACC_CASSERT_MSG(
162  MODENUMBER_must_be_smaller_than_number_of_entries_in_LAGUERREMODES_vector,
163  Unitless::MODENUMBER < Unitless::LAGUERREMODES_t::dim
164  );
165  for( uint32_t m = 0 ; m <= Unitless::MODENUMBER ; ++m )
166  etrans_norm += typename Unitless::LAGUERREMODES_t{}[m];
167 
168  // beam waist in the near field: w_y(y=0) == W0
169  float_X const w_y = Unitless::W0 * algorithms::math::sqrt( 1.0_X + ( focusPos / y_R )*( focusPos / y_R ) );
171  float_X const xi_y = algorithms::math::atan( -focusPos / y_R );
172 
173  if( Unitless::Polarisation == Unitless::LINEAR_X || Unitless::Polarisation == Unitless::LINEAR_Z )
174  {
175  for( uint32_t m = 0 ; m <= Unitless::MODENUMBER ; ++m )
176  {
177  etrans += typename Unitless::LAGUERREMODES_t{}[m] * simpleLaguerre( m, 2.0_X * r2 / w_y / w_y )
178  * math::exp( -r2 / w_y / w_y ) * math::cos( 2.0_X * float_X( PI ) / Unitless::WAVE_LENGTH * focusPos - 2.0_X * float_X( PI ) / Unitless::WAVE_LENGTH * r2 / 2.0_X / R_y + ( 2._X * float_X( m ) + 1._X ) * xi_y + m_phase )
179  * math::exp( -( r2 / 2.0_X / R_y - focusPos - m_phase / 2.0_X / float_X( PI ) * Unitless::WAVE_LENGTH )
180  *( r2 / 2.0_X / R_y - focusPos - m_phase / 2.0_X / float_X( PI ) * Unitless::WAVE_LENGTH )
181  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( 2.0_X * Unitless::PULSE_LENGTH ) / ( 2.0_X * Unitless::PULSE_LENGTH ) );
182  }
183  m_elong *= etrans / etrans_norm;
184  }
185  else if( Unitless::Polarisation == Unitless::CIRCULAR )
186  {
187  for( uint32_t m = 0 ; m <= Unitless::MODENUMBER ; ++m )
188  {
189  etrans += typename Unitless::LAGUERREMODES_t{}[m] * simpleLaguerre( m, 2.0_X * r2 / w_y / w_y )
190  * math::exp( -r2 / w_y / w_y ) * math::cos( 2.0_X * float_X( PI ) / Unitless::WAVE_LENGTH * focusPos - 2.0_X * float_X( PI ) / Unitless::WAVE_LENGTH * r2 / 2.0_X / R_y + ( 2._X * float_X( m ) + 1._X ) * xi_y + m_phase )
191  * math::exp( -( r2 / 2.0_X / R_y - focusPos - m_phase / 2.0_X / float_X( PI ) * Unitless::WAVE_LENGTH )
192  *( r2 / 2.0_X / R_y - focusPos - m_phase / 2.0_X / float_X( PI ) * Unitless::WAVE_LENGTH )
193  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( 2.0_X * Unitless::PULSE_LENGTH ) / ( 2.0_X * Unitless::PULSE_LENGTH ) );
194  }
195  m_elong.x() *= etrans / etrans_norm;
196  m_phase += float_X( PI / 2.0 );
197  etrans = 0.0_X;
198  for( uint32_t m = 0 ; m <= Unitless::MODENUMBER ; ++m )
199  {
200  etrans += typename Unitless::LAGUERREMODES_t{}[m] * simpleLaguerre( m, 2.0_X * r2 / w_y / w_y )
201  * math::exp( -r2 / w_y / w_y ) * math::cos( 2.0_X * float_X( PI ) / Unitless::WAVE_LENGTH * focusPos - 2.0_X * float_X( PI ) / Unitless::WAVE_LENGTH * r2 / 2.0_X / R_y + ( 2._X * float_X( m ) + 1._X ) * xi_y + m_phase )
202  * math::exp( -( r2 / 2.0_X / R_y - focusPos - m_phase / 2.0_X / float_X( PI ) * Unitless::WAVE_LENGTH )
203  *( r2 / 2.0_X / R_y - focusPos - m_phase / 2.0_X / float_X( PI ) * Unitless::WAVE_LENGTH )
204  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( 2.0_X * Unitless::PULSE_LENGTH ) / ( 2.0_X * Unitless::PULSE_LENGTH ) );
205  }
206  m_elong.z() *= etrans / etrans_norm;
207  // reminder: if you want to use phase below, substract pi/2
208  // m_phase -= float_X( PI / 2.0 );
209  }
210 
211  if( Unitless::initPlaneY != 0 ) // compile time if
212  {
213  /* If the laser is not initialized in the first cell we emit a
214  * negatively and positively propagating wave. Therefore we need to multiply the
215  * amplitude with a correction factor depending of the cell size in
216  * propagation direction.
217  * The negatively propagating wave is damped by the absorber.
218  *
219  * The `correctionFactor` assume that the wave is moving in y direction.
220  */
221  auto const correctionFactor = ( SPEED_OF_LIGHT * DELTA_T ) / CELL_HEIGHT * 2._X;
222 
223  // jump over the guard of the electric field
224  m_dataBoxE( localCell + SuperCellSize::toRT() * GuardSize::toRT() ) += correctionFactor * m_elong;
225  }
226  else
227  {
228  // jump over the guard of the electric field
229  m_dataBoxE( localCell + SuperCellSize::toRT() * GuardSize::toRT() ) = m_elong;
230  }
231  }
232  };

249  HINLINE GaussianBeam( uint32_t currentStep )
250  {
251  // get data
252  DataConnector & dc = Environment< >::get( ).DataConnector( );
253  dataBoxE = dc.get< FieldE >(
254  FieldE::getName(),
255  true
256  )->getDeviceDataBox();
257 
258  // get meta data for offsets
259  SubGrid< simDim > const & subGrid = Environment< simDim >::get().SubGrid();
260  // const DataSpace< simDim > totalCellOffset( subGrid.getGlobalDomain().offset );
261  DataSpace< simDim > const globalCellOffset( subGrid.getLocalDomain().offset );
262  DataSpace< simDim > const halfSimSize( subGrid.getGlobalDomain().size / 2 );
263 
264  // transform coordinate system to center of global simulation as origin [cells]
265  offsetToTotalDomain = /* totalCellOffset + */ globalCellOffset - halfSimSize;
266 
267  // @todo reset origin of direction of moving window
268  // offsetToTotalDomain.y() = 0
269 
270  float_64 const runTime = DELTA_T * currentStep - Unitless::laserTimeShift;
271 
272  // calculate focus position relative to the laser initialization plane
273  float_X const focusPos = Unitless::FOCUS_POS - Unitless::initPlaneY * CELL_HEIGHT;
274 
275  elong = float3_X::create( 0.0_X );
276 
277  // This check is done here on HOST, since std::numeric_limits<float_X>::epsilon() does not compile on laserTransversal(), which is on DEVICE.
278  float_X etrans_norm( 0.0_X );
279 
280  PMACC_CASSERT_MSG(
281  MODENUMBER_must_be_smaller_than_number_of_entries_in_LAGUERREMODES_vector,
282  Unitless::MODENUMBER < Unitless::LAGUERREMODES_t::dim
283  );
284  for( uint32_t m = 0 ; m <= Unitless::MODENUMBER ; ++m )
285  etrans_norm += typename Unitless::LAGUERREMODES_t{}[m];
286  PMACC_VERIFY_MSG(
287  algorithms::math::abs( etrans_norm ) > std::numeric_limits< float_X >::epsilon(),
288  "Sum of LAGUERREMODES can not be 0."
289  );
290 
291 
292  // a symmetric pulse will be initialized at position z=0 for
293  // a time of PULSE_INIT * PULSE_LENGTH = INIT_TIME.
294  // we shift the complete pulse for the half of this time to start with
295  // the front of the laser pulse.
296  constexpr float_64 mue = 0.5 * Unitless::INIT_TIME;
297 
298  // rayleigh length (in y-direction)
299  constexpr float_64 y_R = PI * Unitless::W0 * Unitless::W0 / Unitless::WAVE_LENGTH;
300  // gaussian beam waist in the nearfield: w_y(y=0) == W0
301  float_64 const w_y = Unitless::W0 * math::sqrt( 1.0 + ( focusPos / y_R )*( focusPos / y_R ) );
302 
303  float_64 envelope = float_64( Unitless::AMPLITUDE );
304  if( simDim == DIM2 )
305  envelope *= math::sqrt( float_64( Unitless::W0 ) / w_y );
306  else if( simDim == DIM3 )
307  envelope *= float_64( Unitless::W0 ) / w_y;
308  /* no 1D representation/implementation */
309 
310  if( Unitless::Polarisation == Unitless::LINEAR_X )
311  {
312  elong.x() = float_X( envelope );
313  }
314  else if( Unitless::Polarisation == Unitless::LINEAR_Z )
315  {
316  elong.z() = float_X( envelope );
317  }
318  else if( Unitless::Polarisation == Unitless::CIRCULAR )
319  {
320  elong.x() = float_X( envelope ) / math::sqrt( 2.0_X );
321  elong.z() = float_X( envelope ) / math::sqrt( 2.0_X );
322  }
323 
324  phase = 2.0_X * float_X( PI ) * float_X( Unitless::f ) * ( runTime - float_X( mue ) - focusPos / SPEED_OF_LIGHT ) + Unitless::LASER_PHASE;
325  }
