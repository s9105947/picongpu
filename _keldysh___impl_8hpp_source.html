233  template< typename T_Acc >
234  DINLINE uint32_t numNewParticles(T_Acc const & acc, FrameType& ionFrame, int localIdx)
235  {
236  /* alias for the single macro-particle */
237  auto particle = ionFrame[localIdx];
238  /* particle position, used for field-to-particle interpolation */
239  floatD_X pos = particle[position_];
240  const int particleCellIdx = particle[localCellIdx_];
241  /* multi-dim coordinate of the local cell inside the super cell */
242  DataSpace<TVec::dim> localCell(DataSpaceOperations<TVec::dim>::template map<TVec > (particleCellIdx));
243  /* interpolation of E- */
244  const picongpu::traits::FieldPosition<typename fields::Solver::NummericalCellType, FieldE> fieldPosE;
245  ValueType_E eField = Field2ParticleInterpolation()
246  (cachedE.shift(localCell).toCursor(), pos, fieldPosE());
247  /* and B-field on the particle position */
248  const picongpu::traits::FieldPosition<typename fields::Solver::NummericalCellType, FieldB> fieldPosB;
249  ValueType_B bField = Field2ParticleInterpolation()
250  (cachedB.shift(localCell).toCursor(), pos, fieldPosB());
251 
252  /* define number of bound macro electrons before ionization */
253  float_X prevBoundElectrons = particle[boundElectrons_];
254 
255  IonizationAlgorithm ionizeAlgo;
256  /* determine number of new macro electrons to be created */
257  uint32_t newMacroElectrons = ionizeAlgo(
258  bField, eField,
259  particle, this->randomGen(acc)
260  );
261 
262  return newMacroElectrons;
263 
264  }

266  /* Functor implementation
267  *
268  * Ionization model specific particle creation
269  *
270  * \tparam T_parentIon type of ion species that is being ionized
271  * \tparam T_childElectron type of electron species that is created
272  * \param parentIon ion instance that is ionized
273  * \param childElectron electron instance that is created
274  */
275  template<typename T_parentIon, typename T_childElectron, typename T_Acc>
276  DINLINE void operator()(T_Acc const & acc, T_parentIon& parentIon,T_childElectron& childElectron)
277  {
278  /* for not mixing operations::assign up with the nvidia functor assign */
279  namespace partOp = pmacc::particles::operations;
280  /* each thread sets the multiMask hard on "particle" (=1) */
281  childElectron[multiMask_] = 1u;
282  const float_X weighting = parentIon[weighting_];
283 
284  /* each thread initializes a clone of the parent ion but leaving out
285  * some attributes:
286  * - multiMask: reading from global memory takes longer than just setting it again explicitly
287  * - momentum: because the electron would get a higher energy because of the ion mass
288  * - boundElectrons: because species other than ions or atoms do not have them
289  * (gets AUTOMATICALLY deselected because electrons do not have this attribute)
290  */
291  auto targetElectronClone = partOp::deselect<bmpl::vector2<multiMask, momentum> >(childElectron);
292 
293  partOp::assign(targetElectronClone, partOp::deselect<particleId>(parentIon));
294 
295  const float_X massIon = attribute::getMass(weighting,parentIon);
296  const float_X massElectron = attribute::getMass(weighting,childElectron);
297 
298  const float3_X electronMomentum (parentIon[momentum_]*(massElectron/massIon));
299 
300  childElectron[momentum_] = electronMomentum;
301 
302  /* conservation of momentum
303  * \todo add conservation of mass */
304  parentIon[momentum_] -= electronMomentum;
305 
312  parentIon[boundElectrons_] -= float_X(1.);
313  }
