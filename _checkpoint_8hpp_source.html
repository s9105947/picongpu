99  ( std::string( "Optional backend for checkpointing [" ) + activeBackends + "] default: " + checkpointBackendName ).c_str( )
100  )(
101  "checkpoint.file",
102  po::value< std::string >( &checkpointFilename ),
103  "Optional checkpoint filename (prefix)"
104  )(
105  "checkpoint.restart.backend",
106  po::value< std::string >( &restartBackendName ),
107  ( std::string( "Optional backend for restarting [" ) + activeBackends + "] default: " + restartBackendName ).c_str( )
108  )(
109  "checkpoint.restart.file",
110  po::value< std::string >( &restartFilename ),
111  "checkpoint restart filename (prefix)"
112  )(
113  /* 1,000,000 particles are around 3900 frames at 256 particles per frame
114  * and match ~30MiB with typical picongpu particles.
115  * The only reason why we use 1M particles per chunk is that we can get a
116  * frame overflow in our memory manager if we process all particles in one kernel.
117  **/
118  "checkpoint.restart.chunkSize",
119  po::value< uint32_t >(&restartChunkSize)->default_value( 1000000u ),
120  "Number of particles processed in one kernel call during restart to prevent frame count blowup"
121  );
122 
123  for( auto & backend : ioBackendsHelp )
124  backend.second->expandHelp(
125  desc,
126  "checkpoint."
127  );

177  void pluginLoad( )
178  {
179  for( auto & backendHelp : ioBackendsHelp )
180  {
181  if( backendHelp.second->getNumPlugins() > 0u )
182  backendHelp.second->validateOptions();
183 
184  size_t const numSlaves = backendHelp.second->getNumPlugins( );
185  if( numSlaves > 1u )
186  throw std::runtime_error( pluginGetName() + ": is no a multi plugin, each option can only be selected once." );
187  }
188 
189  // create checkpoint creation backend
190  if( !ioBackendsHelp.empty( ) )
191  {
192  auto cBackendHelp = ioBackendsHelp.find( checkpointBackendName );
193  if( cBackendHelp == ioBackendsHelp.end( ) )
194  throw std::runtime_error( std::string( "IO-backend " ) +
195  checkpointBackendName +
196  " for checkpoints not found, possible backends: " +
197  activeBackends
198  );
199  else
200  ioBackends[ checkpointBackendName ] = std::static_pointer_cast< IIOBackend >(
201  cBackendHelp->second->create(
202  cBackendHelp->second,
203  0,
204  m_cellDescription
205  )
206  );
207  }
208  // create restart backend
209  if( !ioBackendsHelp.empty( ) && checkpointBackendName != restartBackendName )
210  {
211  auto rBackend = ioBackendsHelp.find( restartBackendName );
212  if( rBackend == ioBackendsHelp.end( ) )
213  throw std::runtime_error( std::string( "IO-backend " ) +
214  restartBackendName +
215  " for restarts not found, possible backends: " +
216  activeBackends
217  );
218  else
219  ioBackends[ restartBackendName ] = std::static_pointer_cast< IIOBackend >(
220  rBackend->second->create(
221  rBackend->second,
222  0,
223  m_cellDescription
224  )
225  );
226  }
227 
228  if( restartFilename.empty( ) )
229  {
230  restartFilename = checkpointFilename;
231  }
232  }
