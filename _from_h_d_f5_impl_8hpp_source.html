50  HINLINE FromHDF5Impl(uint32_t currentStep)
51  {
52  const uint32_t numSlides = MovingWindow::getInstance( ).getSlideCounter( currentStep );
53  auto window = MovingWindow::getInstance().getWindow(currentStep);
54  loadHDF5(window);
55  const SubGrid<simDim>& subGrid = Environment<simDim>::get().SubGrid();
56  DataSpace<simDim> localCells = subGrid.getLocalDomain( ).size;
57  totalGpuOffset = subGrid.getLocalDomain( ).offset;
58  totalGpuOffset.y( ) += numSlides * localCells.y( );
59  }
60 
65  HDINLINE float_X operator()(const DataSpace<simDim>& totalCellOffset)
66  {
67  const DataSpace<simDim> localCellIdx(totalCellOffset - totalGpuOffset);
68  return precisionCast<float_X>(deviceDataBox(localCellIdx + SuperCellSize::toRT() * GuardSize::toRT()).x());
69  }

73  void loadHDF5(Window &window)
74  {
75  using namespace splash;
76  DataConnector &dc = Environment<>::get().DataConnector();
77 
78  PMACC_CASSERT_MSG(
79  _please_allocate_at_least_one_FieldTmp_in_memory_param,
80  fieldTmpNumSlots > 0
81  );
82  auto fieldTmp = dc.get< FieldTmp >( FieldTmp::getUniqueId( 0 ), true );
83  auto& fieldBuffer = fieldTmp->getGridBuffer();
84 
85  deviceDataBox = fieldBuffer.getDeviceBuffer().getDataBox();
86 
87  GridController<simDim> &gc = Environment<simDim>::get().GridController();
88  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
89  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(0);
90  const uint32_t maxOpenFilesPerNode = 1;
91 
92  /* get a new ParallelDomainCollector for our MPI rank only*/
93  ParallelDomainCollector pdc(
94  MPI_COMM_SELF,
95  gc.getCommunicator().getMPIInfo(),
96  Dimensions(1, 1, 1),
97  maxOpenFilesPerNode);
98 
99  try
100  {
101  /* setup ParallelDomainCollector pdc to read the density information from hdf5 */
102  DataCollector::FileCreationAttr attr;
103  DataCollector::initFileCreationAttr(attr);
104  attr.fileAccType = DataCollector::FAT_READ;
105 
106  pdc.open(ParamClass::filename, attr);
107 
108  /* set which part of the hdf5 file our MPI rank reads */
109  DataSpace<simDim> globalSlideOffset;
110  globalSlideOffset.y() = numSlides * localDomain.size.y();
111 
112  Dimensions domainOffset(0, 0, 0);
113  for (uint32_t d = 0; d < simDim; ++d)
114  domainOffset[d] = localDomain.offset[d] + globalSlideOffset[d];
115 
116  if (gc.getPosition().y() == 0)
117  domainOffset[1] += window.globalDimensions.offset.y();
118 
119  DataSpace<simDim> localDomainSize = localDomain.size;
120  Dimensions domainSize(1, 1, 1);
121  for (uint32_t d = 0; d < simDim; ++d)
122  domainSize[d] = localDomainSize[d];
123 
124  /* clear host buffer with default value */
125  fieldBuffer.getHostBuffer().setValue(float1_X(ParamClass::defaultDensity));
126 
127  /* get dimensions and offsets (collective call) */
128  Domain fileDomain = pdc.getGlobalDomain(ParamClass::iteration, ParamClass::datasetName);
129  Dimensions fileDomainEnd = fileDomain.getOffset() + fileDomain.getSize();
130  DataSpace<simDim> accessSpace;
131  DataSpace<simDim> accessOffset;
132 
133  Dimensions fileAccessSpace(1, 1, 1);
134  Dimensions fileAccessOffset(0, 0, 0);
135 
136  /* For each dimension, compute how file domain and local simulation domain overlap
137  * and which sizes and offsets are required for loading data from the file.
138  **/
139  for (uint32_t d = 0; d < simDim; ++d)
140  {
141  /* file domain in/in-after sim domain */
142  if (fileDomain.getOffset()[d] >= domainOffset[d] &&
143  fileDomain.getOffset()[d] <= domainOffset[d] + domainSize[d])
144  {
145  accessSpace[d] = std::min(domainOffset[d] + domainSize[d] - fileDomain.getOffset()[d],
146  fileDomain.getSize()[d]);
147  fileAccessSpace[d] = accessSpace[d];
148 
149  accessOffset[d] = fileDomain.getOffset()[d] - domainOffset[d];
150  fileAccessOffset[d] = 0;
151  continue;
152  }
153 
154  /* file domain before-in sim domain */
155  if (fileDomainEnd[d] >= domainOffset[d] &&
156  fileDomainEnd[d] <= domainOffset[d] + domainSize[d])
157  {
158  accessSpace[d] = fileDomainEnd[d] - domainOffset[d];
159  fileAccessSpace[d] = accessSpace[d];
160 
161  accessOffset[d] = 0;
162  fileAccessOffset[d] = domainOffset[d] - fileDomain.getOffset()[d];
163  continue;
164  }
165 
166  /* sim domain in file domain */
167  if (domainOffset[d] >= fileDomain.getOffset()[d] &&
168  domainOffset[d] + domainSize[d] <= fileDomainEnd[d])
169  {
170  accessSpace[d] = domainSize[d];
171  fileAccessSpace[d] = accessSpace[d];
172 
173  accessOffset[d] = 0;
174  fileAccessOffset[d] = domainOffset[d] - fileDomain.getOffset()[d];
175  continue;
176  }
177 
178  /* file domain and sim domain do not intersect, do not load anything */
179  accessSpace[d] = 0;
180  break;
181  }
182 
183  /* allocate temporary buffer for hdf5 data */
184  using ValueType = typename FieldTmp::ValueType::type;
185  ValueType *tmpBfr = nullptr;
186 
187  size_t accessSize = accessSpace.productOfComponents();
188  if (accessSize > 0)
189  {
190  tmpBfr = new ValueType[accessSize];
191 
192  Dimensions sizeRead(0, 0, 0);
193  pdc.read(
194  ParamClass::iteration,
195  fileAccessSpace,
196  fileAccessOffset,
197  ParamClass::datasetName,
198  sizeRead,
199  tmpBfr);
200 
201  if (sizeRead.getScalarSize() != accessSize)
202  {
203  __delete(tmpBfr);
204  return;
205  }
206 
207  /* get the databox of the host buffer */
208  auto dataBox = fieldBuffer.getHostBuffer().getDataBox();
209  /* get a 1D access object to the databox */
210  using D1Box = DataBoxDim1Access< typename FieldTmp::DataBoxType >;
211  DataSpace<simDim> guards = fieldBuffer.getGridLayout().getGuard();
212  D1Box d1RAccess(dataBox.shift(guards + accessOffset), accessSpace);
213 
214  /* copy from temporary buffer to fieldTmp host buffer */
215  for (int i = 0; i < accessSpace.productOfComponents(); ++i)
216  {
217  d1RAccess[i].x() = tmpBfr[i];
218  }
219 
220  __delete(tmpBfr);
221  }
222 
223  pdc.close();
224 
225  /* copy host data to the device */
226  fieldBuffer.hostToDevice();
227  __getTransactionEvent().waitForFinished();
228 
229  }
230  catch (const DCException& e)
231  {
232  std::cerr << e.what() << std::endl;
233  return;
234  }
235 
236  return;
237  }
238 
239  PMACC_ALIGN(deviceDataBox,FieldTmp::DataBoxType);
240  PMACC_ALIGN(totalGpuOffset,DataSpace<simDim>);
241 };
242 }
