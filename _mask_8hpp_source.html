140  HDINLINE bool containsExchangeType(uint32_t ex) const
141  {
142  for (uint32_t i = 1; i < 27; i++) //first bit in mask is 1u<<RIGHT
143  {
144  if (isSet(i))
145  {
146  uint32_t tmp = i;
147  uint32_t tmp_ex = ex;
148  while (tmp_ex >= 3)
149  {

151  tmp /= 3;
152  }
153  if (tmp % 3 == tmp_ex)
154  return true;
155  }
156  }

204  static ExchangeType getMirroredExchangeType(uint32_t ex)
205  {
206  if (ex >= traits::NumberOfExchanges<DIM3>::value)
207  throw std::runtime_error("parameter exceeds allowed maximum");
208 
209  Mask mask(ex);
210  uint32_t tmp = 0;
211  if (mask.containsExchangeType(RIGHT))
212  tmp += LEFT;
213  if (mask.containsExchangeType(LEFT))
214  tmp += RIGHT;
215  if (mask.containsExchangeType(BOTTOM))
216  tmp += TOP;
217  if (mask.containsExchangeType(TOP))
218  tmp += BOTTOM;
219  if (mask.containsExchangeType(FRONT))
220  tmp += BACK;
221  if (mask.containsExchangeType(BACK))
222  tmp += FRONT;
223 
224  return (ExchangeType) tmp;
