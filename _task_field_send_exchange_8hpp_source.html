57  bool executeIntern()
58  {
59  switch (m_state)
60  {
61  case Init:
62  break;
63  case WaitForBash:
64 
65  if (nullptr == Environment<>::get().Manager().getITaskIfNotFinished(m_initDependency.getTaskId()) )
66  {
67  m_state = InitSend;
68  m_sendEvent = m_buffer.getGridBuffer().asyncSend(EventTask(), m_exchange);
69  m_initDependency = m_sendEvent;
70  m_state = WaitForSendEnd;
71  }
72 
73  break;
74  case InitSend:
75  break;
76  case WaitForSendEnd:
77  if (nullptr == Environment<>::get().Manager().getITaskIfNotFinished(m_sendEvent.getTaskId()))
78  {
79  m_state = Finished;
80  return true;
81  }
82  break;
83  case Finished:
84  return true;
85  default:
86  return false;
87  }
88 
89  return false;
90  }
