66  {
67  /* The compiler is allowed to evaluate an expression that does not depend on a template parameter
68  * even if the class is never instantiated. In that case static assert is always
69  * evaluated (e.g. with clang), this results in an error if the condition is false.
70  * http://www.boost.org/doc/libs/1_60_0/doc/html/boost_staticassert.html
71  *
72  * A workaround is to add a template dependency to the expression.
73  * `sizeof(ANY_TYPE) != 0` is always true and defers the evaluation.
74  */
75  PMACC_CASSERT_MSG(This_species_has_only_one_charge_state,1==2 && (sizeof(T_Particle) != 0));
76  }
