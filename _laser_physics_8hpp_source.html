50  struct KernelLaser
51  {
52  template<
53  typename T_Acc,
54  typename T_LaserFunctor
55  >
56  DINLINE void operator()(
57  T_Acc const & acc,
58  T_LaserFunctor laserFunctor
59  ) const
60  {
61  using LaserPlaneSizeInSuperCell = T_LaserPlaneSizeInSuperCell;
62  using LaserFunctor = T_LaserFunctor;
63 
64  PMACC_CASSERT_MSG(
65  __LaserPlaneSizeInSuperCell_y_must_be_less_or_equal_than_SuperCellSize_y,
66  LaserPlaneSizeInSuperCell::y::value <= SuperCellSize::y::value
67  );
68 
69  constexpr uint32_t planeSize = pmacc::math::CT::volume< LaserPlaneSizeInSuperCell >::type::value;
70  PMACC_CONSTEXPR_CAPTURE uint32_t numWorkers = T_numWorkers;
71 
72  const uint32_t workerIdx = threadIdx.x;
73 
74  // offset of the superCell (in cells, without any guards) to the origin of the local domain
75 
76  DataSpace< simDim > localSuperCellOffset = DataSpace< simDim >( blockIdx );
77 
78  // add not handled supercells from LaserFunctor::Unitless::initPlaneY
79  localSuperCellOffset.y() += LaserFunctor::Unitless::initPlaneY / SuperCellSize::y::value;
80 
81  uint32_t cellOffsetInSuperCellFromInitPlaneY = LaserFunctor::Unitless::initPlaneY % SuperCellSize::y::value;
82 
83  mappings::threads::ForEachIdx<
84  mappings::threads::IdxConfig<
85  planeSize,
86  numWorkers
87  >
88  > { workerIdx }(
89  [&](
90  uint32_t const linearIdx,
91  uint32_t const
92  )
93  {
94  auto accLaserFunctor = laserFunctor(
95  acc,
96  localSuperCellOffset,
97  mappings::threads::WorkerCfg< numWorkers >{ workerIdx }
98  );
99 
100  /* cell index within the superCell */
101  DataSpace< simDim > cellIdxInSuperCell = DataSpaceOperations< simDim >::template map< LaserPlaneSizeInSuperCell >( linearIdx );
102  cellIdxInSuperCell.y() += cellOffsetInSuperCellFromInitPlaneY;
103 
104  accLaserFunctor( acc, cellIdxInSuperCell );
105  }
106  );
107  }

111  struct LaserPhysics
112  {
113  void operator()(uint32_t currentStep) const
114  {
115  /* The laser can be initialized in the plane of the first cell or
116  * any later x-z plane inside the simulation. Initializing the
117  * laser in planes inside the simulation corresponds to an
118  * evaluation of the field at negatively shifted time.
119  */
120  constexpr float_X laserTimeShift = laserProfiles::Selected::Unitless::initPlaneY * CELL_HEIGHT / SPEED_OF_LIGHT;
121 
122  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(currentStep);
123 
124  /* Disable laser if
125  * - init time of laser is over or
126  * - we have periodic boundaries in Y direction or
127  * - we already performed a slide
128  */
129  bool const laserNone = ( laserProfiles::Selected::Unitless::INIT_TIME == float_X(0.0) );
130  bool const laserInitTimeOver =
131  ( ( currentStep * DELTA_T - laserTimeShift ) >= laserProfiles::Selected::Unitless::INIT_TIME );
132  bool const topBoundariesArePeriodic =
133  ( Environment<simDim>::get().GridController().getCommunicationMask( ).isSet( TOP ) );
134  bool const boxHasSlided = ( numSlides != 0 );
135 
136  bool const disableLaser =
137  laserNone ||
138  laserInitTimeOver ||
139  topBoundariesArePeriodic ||
140  boxHasSlided;
141  if( !disableLaser )
142  {
143  PMACC_VERIFY_MSG(
144  laserProfiles::Selected::Unitless::initPlaneY < static_cast<uint32_t>( Environment<simDim>::get().SubGrid().getLocalDomain().size.y() ),
145  "initPlaneY must be located in the top GPU"
146  );
147 
148  // laser is disabled e.g. laserNone
149  constexpr bool isLaserDisabled = laserProfiles::Selected::Unitless::INIT_TIME == 0.0_X;
150  constexpr bool isLaserInitInFirstCell = laserProfiles::Selected::Unitless::initPlaneY == 0;
151  // X + 1 is a workaround to avoid warning: pointless comparison of unsigned integer with zero
152  constexpr bool isInitPlaneYOutsideOfAbsorber =
153  laserProfiles::Selected::Unitless::initPlaneY + 1 > absorber::numCells[1][0] + 1;
154  PMACC_CASSERT_MSG(
155  __initPlaneY_needs_to_be_greater_than_the_top_absorber_cells_or_zero,
156  isLaserDisabled || isLaserInitInFirstCell || isInitPlaneYOutsideOfAbsorber
157  );
158 
159  /* Calculate how many neighbors to the left we have
160  * to initialize the laser in the E-Field
161  *
162  * Example: Yee needs one neighbor to perform dB = curlE
163  * -> initialize in y=0 plane
164  * A second order solver could need 2 neighbors left:
165  * -> initialize in y=0 and y=1 plane
166  *
167  * Question: Why do other codes initialize the B-Field instead?
168  * Answer: Because our fields are defined on the lower cell side
169  * (C-Style ftw). Therefore, our curls (for example Yee)
170  * are shifted nabla+ <-> nabla- compared to Fortran codes
171  * (in other words: curlLeft <-> curlRight)
172  * for E and B.
173  * For this reason, we have to initialize E instead of B.
174  *
175  * Problem: that's still not our case. For example our Yee does a
176  * dE = curlLeft(B) - therefor, we should init B, too.
177  *
178  *
179  * @todo: might also lack temporal offset since our formulas are E(x,z,t) instead of E(x,y,z,t)
180  * `const int max_y_neighbors = Get<fields::Solver::OffsetOrigin_E, 1 >::value;`
181  *
182  * @todo Right now, the phase could be wrong ( == is cloned)
183  * @see LaserPhysics.hpp
184  *
185  * @todo What about the B-Field in the second plane?
186  *
187  */
188  constexpr int laserInitCellsInY = 1;
189 
190  using LaserPlaneSizeInSuperCells = typename pmacc::math::CT::AssignIfInRange<
191  typename SuperCellSize::vector_type,
192  bmpl::integral_c< uint32_t, 1 >, /* y direction */
193  bmpl::integral_c< int, laserInitCellsInY >
194  >::type;
195 
196  DataSpace< simDim > gridBlocks = Environment< simDim >::get().SubGrid().getLocalDomain().size / SuperCellSize::toRT();
197  // use the one supercell in y to initialize the laser plane
198  gridBlocks.y() = 1;
199 
200  constexpr uint32_t numWorkers = pmacc::traits::GetNumWorkers<
201  pmacc::math::CT::volume< LaserPlaneSizeInSuperCells >::type::value
202  >::value;
203 
204  PMACC_KERNEL(
205  KernelLaser<
206  numWorkers,
207  LaserPlaneSizeInSuperCells
208  >{}
209  )(
210  gridBlocks,
211  numWorkers
212  )(
213  laserProfiles::Selected( currentStep )
214  );
215  }
216  }
217  };
