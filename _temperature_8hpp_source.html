88  // => p = sqrt(m)
89  //
90  // Note on macro particle energies, with weighting w:
91  // p_1 = m_1 * v
92  // v = v_1 = v_w
93  // p_w = p_1 * w
94  // E_w = E_1 * w
95  // Since masses, energies and momenta add up linear, we can
96  // just take w times the p_1. Take care, E means E_1 !
97  // This goes to:
98  // p_w = w * p_1 = w * m_1 * sqrt( E / m_1 )
99  // = sqrt( E * w^2 * m_1 )
100  // = sqrt( E * w * m_w )
101  // Which makes sense, since it means that we use a macroMass
102  // and a macroEnergy now.
103  float3_X const mom = tmpRand * ( float_X )math::sqrt(
104  precisionCast< sqrt_X >(
105  macroEnergy *

107  )
108  );
