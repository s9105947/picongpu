51  T_ParticleCreator particleCreator,
52  T_CellDescription* cellDesc)
53 {
54  typedef typename MappingDesc::SuperCellSize SuperCellSize;
55  const pmacc::math::Int<simDim> coreBorderGuardSuperCells = cellDesc->getGridSuperCells();
56  const uint32_t guardSuperCells = cellDesc->getGuardingSuperCells();
57  const pmacc::math::Int<simDim> coreBorderSuperCells = coreBorderGuardSuperCells - 2*guardSuperCells;
58 
59  constexpr uint32_t numWorkers = pmacc::traits::GetNumWorkers<
60  pmacc::math::CT::volume< SuperCellSize >::type::value
61  >::value;
62 
63  /* Functor holding the actual generic particle creation kernel */
64  auto createParticlesKernel = make_CreateParticlesKernel< numWorkers >(
65  sourceSpecies.getDeviceParticlesBox(),
66  targetSpecies.getDeviceParticlesBox(),
67  particleCreator,
68  guardSuperCells);
69 
70  /* This zone represents the core+border area with guard offset in unit of cells */
71  const zone::SphericZone<simDim> zone(
72  static_cast<pmacc::math::Size_t<simDim> >(coreBorderSuperCells * SuperCellSize::toRT()),
73  guardSuperCells * SuperCellSize::toRT());
74 
75  algorithm::kernel::ForeachLockstep<numWorkers, SuperCellSize> foreach;
76  foreach(zone, createParticlesKernel, cursor::make_MultiIndexCursor<simDim>());
