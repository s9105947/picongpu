74  /* iterate over all grid points */

80  /* At the moment Esirkepov only supports Yee cells where W is defined at origin (0,0,0)
81  *
82  * \todo: please fix me that we can use CenteredCell
83  */
84  template<
85  typename DataBoxJ,
86  typename PosType,
87  typename VelType,
88  typename ChargeType,
89  typename T_Acc
90  >
91  DINLINE void operator()(
92  T_Acc const & acc,
93  DataBoxJ dataBoxJ,
94  const PosType pos,
95  const VelType velocity,
96  const ChargeType charge, const float_X deltaTime
97  )
98  {
99  this->charge = charge;
100  const float3_X deltaPos = float3_X(velocity.x() * deltaTime / cellSize.x(),
101  velocity.y() * deltaTime / cellSize.y(),
102  velocity.z() * deltaTime / cellSize.z());
103  const PosType oldPos = pos - deltaPos;
104  Line<float3_X> line(oldPos, pos);
105  auto cursorJ = dataBoxJ.toCursor();
106 
114  using namespace cursor::tools;
115  cptCurrent1D(acc, twistVectorFieldAxes<pmacc::math::CT::Int < 1, 2, 0 > >(cursorJ), rotateOrigin < 1, 2, 0 > (line), cellSize.x());
116  cptCurrent1D(acc, twistVectorFieldAxes<pmacc::math::CT::Int < 2, 0, 1 > >(cursorJ), rotateOrigin < 2, 0, 1 > (line), cellSize.y());
117  cptCurrent1D(acc, cursorJ, line, cellSize.z());
118  }

126  template<
127  typename CursorJ,
128  typename T_Acc
129  >
130  DINLINE void cptCurrent1D(
131  T_Acc const & acc,
132  CursorJ cursorJ,
133  const Line<float3_X>& line,
134  const float_X cellEdgeLength
135  )
136  {
137  /* pick every cell in the xy-plane that is overlapped by particle's
138  * form factor and deposit the current for the cells above and beneath
139  * that cell and for the cell itself.
140  */
141  for (int i = begin; i < end; ++i)
142  {
143  for (int j = begin; j < end; ++j)
144  {
145  float_X tmp =
146  S0(line, i, 1) * S0(line, j, 2) +
147  float_X(0.5) * DS(line, i, 1) * S0(line, j, 2) +
148  float_X(0.5) * S0(line, i, 1) * DS(line, j, 2) +
149  (float_X(1.0) / float_X(3.0)) * DS(line, i, 1) * DS(line, j, 2);
150 
151  float_X accumulated_J = float_X(0.0);
152  for (int k = begin; k < end; ++k)
153  {
154  float_X W = DS(line, k, 3) * tmp;
155  /* We multiply with `cellEdgeLength` due to the fact that the attribute for the
156  * in-cell particle `position` (and it's change in DELTA_T) is normalize to [0,1) */
157  accumulated_J += -this->charge * (float_X(1.0) / float_X(CELL_VOLUME * DELTA_T)) * W * cellEdgeLength;
158  atomicAdd(&((*cursorJ(i, j, k)).z()), accumulated_J, ::alpaka::hierarchy::Threads{});
159  }
160  }
161  }
