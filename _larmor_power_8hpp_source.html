44  DINLINE float_X
45  LarmorPower::operator()( T_Particle& particle ) const
46  {
47 
48  constexpr bool hasMomentumPrev1 = pmacc::traits::HasIdentifier<
49  typename T_Particle::FrameType,
50  momentumPrev1
51  >::type::value;
52  PMACC_CASSERT_MSG_TYPE( species_must_have_the_attribute_momentumPrev1, T_Particle, hasMomentumPrev1 );
53 
54  /* read existing attributes */
55  const float3_X mom = particle[momentum_];
56  const float3_X mom_mt1 = particle[momentumPrev1_];
57  const float_X weighting = particle[weighting_];
58  const float_X charge = attribute::getCharge( weighting, particle );
59  const float_X mass = attribute::getMass( weighting, particle );
60 
61  /* calculate new attribute */
62  Gamma<float_X> calcGamma;
63  const typename Gamma<float_X>::valueType gamma = calcGamma( mom, mass );
64  const float_X gamma2 = gamma * gamma;
65  const float_X c2 = SPEED_OF_LIGHT * SPEED_OF_LIGHT;
66 
67  const float3_X mom_dt = (mom - mom_mt1) / float_X(DELTA_T);
68  const float_X el_factor = charge * charge
69  / (float_X(6.0) * PI * EPS0 *
70  c2 * SPEED_OF_LIGHT * mass * mass) * gamma2 * gamma2;
71  const float_X momentumToBetaConvert = float_X(1.0)/ (mass * SPEED_OF_LIGHT * gamma);
72  const float_X larmorPower = el_factor
73  * ( math::abs2(mom_dt)
74  - momentumToBetaConvert * momentumToBetaConvert
75  * math::abs2(math::cross(mom, mom_dt))
76  );
77 
78  /* return attribute */
79  return larmorPower;
80  }
