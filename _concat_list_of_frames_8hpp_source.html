104 
105  const int particlesPerFrame = pmacc::math::CT::volume<SuperCellSize>::type::value;
106  int localIdxs[particlesPerFrame];
107 
108  const DataSpace<Mapping::Dim> superCellIdx = mapper.getSuperCellIndex(blockIndex);
109  const DataSpace<Mapping::Dim> superCellPosition((superCellIdx - mapper.getGuardingSuperCells()) * mapper.getSuperCellSize());
110  filter.setSuperCellPosition(superCellPosition);
111  auto accParFilter = parFilter(
112  1, /* @todo this is a hack, please add a alpaka accelerator here*/
113  superCellIdx - mapper.getGuardingSuperCells( ),
114  WorkerCfg< 1 >{ 0 } /* @todo this is a workaround because we use no alpaka*/
115  );
116 
117  SrcFramePtr srcFramePtr = srcBox.getFirstFrame(superCellIdx);
118 
119  /* Loop over all frames in current super cell */
120  while (srcFramePtr.isValid())
121  {
122  /* Count number of particles in current frame and init its indices */
123  int curNumParticles = 0;
124  for (int particleIdx = 0; particleIdx < particlesPerFrame; ++particleIdx)
125  {
126  localIdxs[particleIdx] = -1;
127  auto parSrc = (srcFramePtr[particleIdx]);
128  /* Check if particle exists and is not filtered */
129  if (parSrc[multiMask_] == 1 && filter(*srcFramePtr, particleIdx))
130  if(
131  accParFilter(
132  1, /* @todo this is a hack, please add a alpaka accelerator here*/
133  parSrc
134  )
135  )
136  localIdxs[particleIdx] = curNumParticles++;
137  }
138 
139  int globalOffset;
140  /* atomic update with openmp*/
141  #pragma omp critical
142  {
143  globalOffset = counter;
144  counter += curNumParticles;
145  }
146 
147  for (int particleIdx = 0; particleIdx < particlesPerFrame; ++particleIdx)
148  {
149  if (localIdxs[particleIdx] != -1)
150  {
151  auto parSrc = (srcFramePtr[particleIdx]);
152  auto parDest = destFrame[globalOffset + localIdxs[particleIdx]];
153  auto parDestNoDomainIdx = deselect<T_Identifier>(parDest);
154  assign(parDestNoDomainIdx, parSrc);
155  /* calculate cell index for user-defined domain */
156  DataSpace<Mapping::Dim> localCellIdx(DataSpaceOperations<Mapping::Dim>::template map<SuperCellSize>(parSrc[localCellIdx_]));
157  parDest[domainCellIdxIdentifier] = domainOffset + superCellPosition + localCellIdx;
158  }
159  }
160  /*get next frame in supercell*/
161  srcFramePtr = srcBox.getNextFrame(srcFramePtr);
162 
163  }
164  }
