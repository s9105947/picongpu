67  template<uint32_t r_dir, uint32_t num_pbins, typename SuperCellSize>
68  struct FunctorParticle
69  {
70  typedef void result_type;
71 
80  template<typename FramePtr, typename float_PS, typename Pitch, typename T_Acc >
81  DINLINE void
82  operator()( const T_Acc & acc,
83  FramePtr frame,
84  uint16_t particleID,
85  cursor::CT::BufferCursor<float_PS, Pitch> curDBufferOriginInBlock,
86  const uint32_t el_p,
87  const std::pair<float_X, float_X>& axis_p_range )
88  {
89  auto particle = frame[particleID];
91  const float_X mom_i = particle[momentum_][el_p];
92 
93  /* cell id in this block */
94  const int linearCellIdx = particle[localCellIdx_];
95  const pmacc::math::UInt32<simDim> cellIdx(
96  pmacc::math::MapToPos<simDim>()( SuperCellSize(), linearCellIdx ) );
97 
98  const uint32_t r_bin = cellIdx[r_dir];
99  const float_X weighting = particle[weighting_];
100  const float_X charge = attribute::getCharge( weighting,particle );
101  const float_PS particleChargeDensity =
102  precisionCast<float_PS>( charge / CELL_VOLUME );
103 
104  const float_X rel_bin = (mom_i / weighting - axis_p_range.first)
105  / (axis_p_range.second - axis_p_range.first);
106  int p_bin = int( rel_bin * float_X(num_pbins) );
107 
108  /* out-of-range bins back to min/max */
109  if( p_bin < 0 )
110  p_bin = 0;
111  if( p_bin >= num_pbins )
112  p_bin = num_pbins - 1;
113 
115  atomicAdd(
116  &(*curDBufferOriginInBlock( p_bin, r_bin )),
117  particleChargeDensity,
118  ::alpaka::hierarchy::Threads{}
119  );
120  }
121  };
