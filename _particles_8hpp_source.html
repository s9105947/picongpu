188  static pmacc::traits::StringProperty getStringProperties()
189  {
190  pmacc::traits::StringProperty propList;
191  const DataSpace<DIM3> periodic =
192  Environment<simDim>::get().EnvironmentController().getCommunicator().getPeriodic();
193 
194  for( uint32_t i = 1; i < NumberOfExchanges<simDim>::value; ++i )
195  {
196  // for each planar direction: left right top bottom back front
197  if( FRONT % i == 0 )
198  {
199  const std::string directionName = ExchangeTypeNames()[i];
200  const DataSpace<DIM3> relDir = Mask::getRelativeDirections<DIM3>(i);
201 
202  const bool isPeriodic =
203  (relDir * periodic) != DataSpace<DIM3>::create(0);
204 
205  std::string boundaryName = "absorbing";
206  if( isPeriodic )
207  boundaryName = "periodic";
208 
209  if( boundaryName == "absorbing" )
210  {
211  propList[directionName]["param"] = std::string("without field correction");
212  }
213  else
214  {
215  propList[directionName]["param"] = std::string("none");
216  }
217 
218  propList[directionName]["name"] = boundaryName;
219  }
220  }
221  return propList;
222  }
