48 struct LoadBoundElectrons
49 {
56  template<typename T_Particle>
57  HDINLINE float_X operator()(const float_X weighting, const T_Particle& particle)
58  {
59  using HasAtomicNumbers = typename pmacc::traits::HasFlag<
60  T_Particle,
61  atomicNumbers<>
62  >::type;
63  PMACC_CASSERT_MSG_TYPE(
64  Having_boundElectrons_particle_attribute_requires_atomicNumbers_flag,
65  T_Particle,
66  HasAtomicNumbers::value
67  );
68  const float_X protonNumber = GetAtomicNumbers<T_Particle>::type::numberOfProtons;
69 
70  /* note: ELECTRON_CHARGE is negative and the second term is also negative
71  */
72  return
73  ELECTRON_CHARGE *
74  ( particle[boundElectrons_] - protonNumber ) *
75  weighting;
76  }
77 };
