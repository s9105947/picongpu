105  /* Shift all particle in a AREA
106  * @tparam AREA area which is used (CORE,BORDER,GUARD or a combination)
107  */
108  template<uint32_t AREA>
109  void shiftParticles()
110  {
111  StrideMapping<AREA, 3, MappingDesc> mapper(this->cellDescription);
112  ParticlesBoxType pBox = particlesBuffer->getDeviceParticleBox();
113 
114  constexpr uint32_t numWorkers = traits::GetNumWorkers<
115  math::CT::volume<typename FrameType::SuperCellSize>::type::value
116  >::value;
117  __startTransaction(__getTransactionEvent());
118  do
119  {
120  PMACC_KERNEL(KernelShiftParticles< numWorkers >{})
121  (mapper.getGridDim(), numWorkers)
122  (pBox, mapper);
123  }
124  while (mapper.next());
125 
126  __setTransactionEvent(__endTransaction());
127 
128  }
