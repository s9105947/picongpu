130  inline void combine(ITask* s1, ITask* s2)
131  {
132  s1->addObserver(this);
133  s2->addObserver(this);
134  if(s1->getTaskType() == ITask::TASK_CUDA && s2->getTaskType() == ITask::TASK_CUDA)
135  {
136  this->setTaskType(ITask::TASK_CUDA);
137  this->setEventStream(static_cast<StreamTask*> (s2)->getEventStream());
138  if(static_cast<StreamTask*> (s1)->getEventStream() != static_cast<StreamTask*> (s2)->getEventStream())
139  this->getEventStream()->waitOn(static_cast<StreamTask*> (s1)->getCudaEventHandle());
140  this->activate();
141  }
142  else if(s1->getTaskType() == ITask::TASK_MPI && s2->getTaskType() == ITask::TASK_CUDA)
143  {
144  this->setTaskType(ITask::TASK_MPI);
145  }
146  else if(s2->getTaskType() == ITask::TASK_MPI && s1->getTaskType() == ITask::TASK_CUDA)
147  {
148  this->setTaskType(ITask::TASK_MPI);

151  {
