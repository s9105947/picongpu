289 struct PushAllSpecies
290 {
297  HINLINE void operator()(
298  const uint32_t currentStep,
299  const EventTask& eventInt,
300  EventTask& pushEvent,
301  EventTask& commEvent
302  ) const
303  {
304  using EventList = std::list<EventTask>;
305  EventList updateEventList;
306  EventList commEventList;
307 
308  /* push all species */
309  using VectorSpeciesWithPusher = typename pmacc::particles::traits::FilterByFlag
310  <
311  VectorAllSpecies,
312  particlePusher<>
313  >::type;
314  meta::ForEach< VectorSpeciesWithPusher, particles::PushSpecies< bmpl::_1 > > pushSpecies;
315  pushSpecies( currentStep, eventInt, updateEventList );
316 
317  /* join all push events */
318  for (typename EventList::iterator iter = updateEventList.begin();
319  iter != updateEventList.end();
320  ++iter)
321  {
322  pushEvent += *iter;
323  }
324 
325  /* call communication for all species */
326  meta::ForEach< VectorSpeciesWithPusher, particles::CommunicateSpecies< bmpl::_1> > communicateSpecies;
327  communicateSpecies( updateEventList, commEventList );
328 
329  /* join all communication events */
330  for (typename EventList::iterator iter = commEventList.begin();
331  iter != commEventList.end();
332  ++iter)
333  {
334  commEvent += *iter;
335  }
336  }

344 template< typename T_SpeciesType, typename T_SelectIonizer >
345 struct CallIonizationScheme
346 {
347  using SpeciesType = pmacc::particles::meta::FindByNameOrType_t<
348  VectorAllSpecies,
349  T_SpeciesType
350  >;
351  using SelectIonizer = T_SelectIonizer;
352  using FrameType = typename SpeciesType::FrameType;
353 
354  /* define the type of the species to be created
355  * from inside the ionization model specialization
356  */
357  using DestSpecies = typename SelectIonizer::DestSpecies;
358  using DestFrameType = typename DestSpecies::FrameType;
359 
367  template<typename T_CellDescription>
368  HINLINE void operator()(
369  T_CellDescription cellDesc,
370  const uint32_t currentStep
371  ) const
372  {
373  DataConnector &dc = Environment<>::get().DataConnector();
374 
375  // alias for pointer on source species
376  auto srcSpeciesPtr = dc.get< SpeciesType >( FrameType::getName(), true );
377  // alias for pointer on destination species
378  auto electronsPtr = dc.get< DestSpecies >( DestFrameType::getName(), true );
379 
380  SelectIonizer selectIonizer(currentStep);
381 
382  creation::createParticlesFromSpecies(*srcSpeciesPtr, *electronsPtr, selectIonizer, cellDesc);
383 
384  /* fill the gaps in the created species' particle frames to ensure that only
385  * the last frame is not completely filled but every other before is full
386  */
387  electronsPtr->fillAllGaps();
388 
389  dc.releaseData( FrameType::getName() );
390  dc.releaseData( DestFrameType::getName() );
391 
392  }
