40 struct Field
41 {
42 
43  /* \param inCellPosition std::vector<std::vector<float_X> > with the outer
44  * vector for each component and the inner vector for
45  * the simDim position offset within the cell [0.0; 1.0)
46  */
47  template<typename T_ValueType, typename T_DataBoxType>
48  static void writeField(ThreadParams *params,
49  const std::string name,
50  std::vector<float_64> unit,
51  std::vector<float_64> unitDimension,
52  std::vector<std::vector<float_X> > inCellPosition,
53  float_X timeOffset,
54  T_DataBoxType dataBox,
55  const T_ValueType&
56  )
57  {
58  typedef T_DataBoxType NativeDataBoxType;
59  typedef T_ValueType ValueType;
60  typedef typename GetComponentsType<ValueType>::type ComponentType;
61  typedef typename PICToSplash<ComponentType>::type SplashType;
62  typedef typename PICToSplash<float_X>::type SplashFloatXType;
63 
64  const uint32_t nComponents = GetNComponents<ValueType>::value;
65 
66  SplashType splashType;
67  ColTypeDouble ctDouble;
68  SplashFloatXType splashFloatXType;
69 
70  log<picLog::INPUT_OUTPUT > ("HDF5 write field: %1% %2%") %
71  name % nComponents;
72 
73  /* parameter checking */
74  PMACC_ASSERT( unit.size() == nComponents );
75  PMACC_ASSERT( inCellPosition.size() == nComponents );
76  for( uint32_t n = 0; n < nComponents; ++n )
77  PMACC_ASSERT( inCellPosition.at(n).size() == simDim );
78  PMACC_ASSERT(unitDimension.size() == 7); // seven openPMD base units
79 
80  /* component names */
81  const std::string recordName = std::string("fields/") + name;
82 
83  std::vector<std::string> name_lookup;
84  {
85  const std::string name_lookup_tpl[] = {"x", "y", "z", "w"};
86  for (uint32_t n = 0; n < nComponents; n++)
87  name_lookup.push_back(name_lookup_tpl[n]);
88  }
89 
90  /*data to describe source buffer*/
91  GridLayout<simDim> field_layout = params->gridLayout;
92  DataSpace<simDim> field_no_guard = params->window.localDimensions.size;
93  DataSpace<simDim> field_guard = field_layout.getGuard() + params->localWindowToDomainOffset;
94  /* globalSlideOffset due to gpu slides between origin at time step 0
95  * and origin at current time step
96  * ATTENTION: splash offset are globalSlideOffset + picongpu offsets
97  */
98  DataSpace<simDim> globalSlideOffset;
99  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
100  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(params->currentStep);
101  globalSlideOffset.y() += numSlides * localDomain.size.y();
102 
103  Dimensions splashGlobalDomainOffset(0, 0, 0);
104  Dimensions splashGlobalOffsetFile(0, 0, 0);
105  Dimensions splashGlobalDomainSize(1, 1, 1);
106 
107  for (uint32_t d = 0; d < simDim; ++d)
108  {
109  splashGlobalOffsetFile[d] = localDomain.offset[d];
110  splashGlobalDomainOffset[d] = params->window.globalDimensions.offset[d] + globalSlideOffset[d];
111  splashGlobalDomainSize[d] = params->window.globalDimensions.size[d];
112  }
113 
114  splashGlobalOffsetFile[1] = std::max(0, localDomain.offset[1] -
115  params->window.globalDimensions.offset[1]);
116 
117  size_t tmpArraySize = field_no_guard.productOfComponents();
118  ComponentType* tmpArray = new ComponentType[tmpArraySize];
119 
120  typedef DataBoxDim1Access<NativeDataBoxType > D1Box;
121  D1Box d1Access(dataBox.shift(field_guard), field_no_guard);
122 
123  for (uint32_t n = 0; n < nComponents; n++)
124  {
125  /* copy data to temp array
126  * tmpArray has the size of the data without any offsets
127  */
128  for (size_t i = 0; i < tmpArraySize; ++i)
129  {
130  tmpArray[i] = d1Access[i][n];
131  }
132 
133  std::stringstream datasetName;
134  datasetName << recordName;
135  if (nComponents > 1)
136  datasetName << "/" << name_lookup.at(n);
137 
138  Dimensions sizeSrcData(1, 1, 1);
139 
140  for (uint32_t d = 0; d < simDim; ++d)
141  {
142  sizeSrcData[d] = field_no_guard[d];
143  }
144 
145  // avoid deadlock between not finished pmacc tasks and mpi calls in splash/HDF5
146  __getTransactionEvent().waitForFinished();
147  params->dataCollector->writeDomain(params->currentStep, /* id == time step */
148  splashGlobalDomainSize, /* total size of dataset over all processes */
149  splashGlobalOffsetFile, /* write offset for this process */
150  splashType, /* data type */
151  simDim, /* NDims spatial dimensionality of the field */
152  splash::Selection(sizeSrcData), /* data size of this process */
153  datasetName.str().c_str(), /* data set name */
154  splash::Domain(
155  splashGlobalDomainOffset, /* offset of the global domain */
156  splashGlobalDomainSize /* size of the global domain */
157  ),
158  DomainCollector::GridType,
159  tmpArray);
160 
161  /* attributes */
162  params->dataCollector->writeAttribute(params->currentStep,
163  splashFloatXType, datasetName.str().c_str(),
164  "position",
165  1u, Dimensions(simDim,0,0),
166  &(*inCellPosition.at(n).begin()));
167 
168  params->dataCollector->writeAttribute(params->currentStep,
169  ctDouble, datasetName.str().c_str(),
170  "unitSI", &(unit.at(n)));
171  }
172  __deleteArray(tmpArray);
173 
174 
175  params->dataCollector->writeAttribute(params->currentStep,
176  ctDouble, recordName.c_str(),
177  "unitDimension",
178  1u, Dimensions(7,0,0),
179  &(*unitDimension.begin()));
180 
181  params->dataCollector->writeAttribute(params->currentStep,
182  splashFloatXType, recordName.c_str(),
183  "timeOffset", &timeOffset);
184 
185  const std::string geometry("cartesian");
186  ColTypeString ctGeometry(geometry.length());
187  params->dataCollector->writeAttribute(params->currentStep,
188  ctGeometry, recordName.c_str(),
189  "geometry", geometry.c_str());
190 
191  const std::string dataOrder("C");
192  ColTypeString ctDataOrder(dataOrder.length());
193  params->dataCollector->writeAttribute(params->currentStep,
194  ctDataOrder, recordName.c_str(),
195  "dataOrder", dataOrder.c_str());
196 
197  char axisLabels[simDim][2];
198  ColTypeString ctAxisLabels(1);
199  for( uint32_t d = 0; d < simDim; ++d )
200  {
201  axisLabels[simDim-1-d][0] = char('x' + d); // 3D: F[z][y][x], 2D: F[y][x]
202  axisLabels[simDim-1-d][1] = '\0'; // terminator is important!
203  }
204  params->dataCollector->writeAttribute(params->currentStep,
205  ctAxisLabels, recordName.c_str(),
206  "axisLabels",
207  1u, Dimensions(simDim,0,0),
208  axisLabels);
209 
210  // cellSize is {x, y, z} but fields are F[z][y][x]
211  std::vector<float_X> gridSpacing(simDim, 0.0);
212  for( uint32_t d = 0; d < simDim; ++d )
213  gridSpacing.at(simDim-1-d) = cellSize[d];
214  params->dataCollector->writeAttribute(params->currentStep,
215  splashFloatXType, recordName.c_str(),
216  "gridSpacing",
217  1u, Dimensions(simDim,0,0),
218  &(*gridSpacing.begin()));
219 
220  // splashGlobalDomainOffset is {x, y, z} but fields are F[z][y][x]
221  std::vector<float_64> gridGlobalOffset(simDim, 0.0);
222  for( uint32_t d = 0; d < simDim; ++d )
223  gridGlobalOffset.at(simDim-1-d) =
224  float_64(cellSize[d]) *
225  float_64(splashGlobalDomainOffset[d]);
226  params->dataCollector->writeAttribute(params->currentStep,
227  ctDouble, recordName.c_str(),
228  "gridGlobalOffset",
229  1u, Dimensions(simDim,0,0),
230  &(*gridGlobalOffset.begin()));
231 
232  params->dataCollector->writeAttribute(params->currentStep,
233  ctDouble, recordName.c_str(),
234  "gridUnitSI", &UNIT_LENGTH);
235 
236  const std::string fieldSmoothing("none");
237  ColTypeString ctFieldSmoothing(fieldSmoothing.length());
238  params->dataCollector->writeAttribute(params->currentStep,
239  ctFieldSmoothing, recordName.c_str(),
240  "fieldSmoothing", fieldSmoothing.c_str());
241  }
242 
243 };
