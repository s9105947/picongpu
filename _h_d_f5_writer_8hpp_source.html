402  void doRestart(
403  const uint32_t restartStep,
404  const std::string& restartDirectory,
405  const std::string& constRestartFilename,
406  const uint32_t restartChunkSize
407  )
408  {
409  // restart is only allowed if the plugin is controlled by the class Checkpoint
410  assert(!m_help->selfRegister);
411 
412  // allow to modify the restart file name
413  std::string restartFilename{ constRestartFilename };
414 
415  const uint32_t maxOpenFilesPerNode = 4;
416  GridController<simDim> &gc = Environment<simDim>::get().GridController();
417  mThreadParams.dataCollector = new ParallelDomainCollector(
418  gc.getCommunicator().getMPIComm(),
419  gc.getCommunicator().getMPIInfo(),
420  splashMpiSize,
421  maxOpenFilesPerNode);
422 
423  mThreadParams.currentStep = restartStep;
424 
425  /* set attributes for datacollector files */
426  DataCollector::FileCreationAttr attr;
427  attr.fileAccType = DataCollector::FAT_READ;
428  attr.mpiPosition.set(splashMpiPos);
429  attr.mpiSize.set(splashMpiSize);
430 
431  /* if restartFilename is relative, prepend with restartDirectory */
432  if (!boost::filesystem::path(restartFilename).has_root_path())
433  {
434  restartFilename = restartDirectory + std::string("/") + restartFilename;
435  }
436 
437  /* open datacollector */
438  try
439  {
440  log<picLog::INPUT_OUTPUT > ("HDF5 open DataCollector with file: %1%") % restartFilename;
441  mThreadParams.dataCollector->open(restartFilename.c_str(), attr);
442  }
443  catch (const DCException& e)
444  {
445  std::cerr << e.what() << std::endl;
446  throw std::runtime_error("HDF5 failed to open DataCollector");
447  }
448 
449  /* load number of slides to initialize MovingWindow */
450  uint32_t slides = 0;
451  mThreadParams.dataCollector->readAttributeInfo(restartStep, nullptr, "sim_slides").read(&slides, sizeof(slides));
452 
453  /* apply slides to set gpus to last/written configuration */
454  log<picLog::INPUT_OUTPUT > ("HDF5 setting slide count for moving window to %1%") % slides;
455  MovingWindow::getInstance().setSlideCounter(slides, restartStep);
456 
457  /* re-distribute the local offsets in y-direction
458  * this will work for restarts with moving window still enabled
459  * and restarts that disable the moving window
460  * \warning enabling the moving window from a checkpoint that
461  * had no moving window will not work
462  */
463  gc.setStateAfterSlides(slides);
464 
465  /* set window for restart, complete global domain */
466  mThreadParams.window = MovingWindow::getInstance().getDomainAsWindow(restartStep);
467  for (uint32_t i = 0; i < simDim; ++i)
468  {
469  mThreadParams.localWindowToDomainOffset[i] = 0;
470  }
471 
472  ThreadParams *params = &mThreadParams;
473 
474  /* load all fields */
475  ForEach<FileCheckpointFields, LoadFields<bmpl::_1> > forEachLoadFields;
476  forEachLoadFields(params);
477 
478  /* load all particles */
479  ForEach<FileCheckpointParticles, LoadSpecies<bmpl::_1> > forEachLoadSpecies;
480  forEachLoadSpecies(params, restartChunkSize);
481 
482  IdProvider<simDim>::State idProvState;
483  ReadNDScalars<uint64_t, uint64_t>()(mThreadParams,
484  "picongpu/idProvider/startId", &idProvState.startId,
485  "maxNumProc", &idProvState.maxNumProc);
486  ReadNDScalars<uint64_t>()(mThreadParams,
487  "picongpu/idProvider/nextId", &idProvState.nextId);
488  log<picLog::INPUT_OUTPUT > ("Setting next free id on current rank: %1%") % idProvState.nextId;
489  IdProvider<simDim>::setState(idProvState);
490 
491  /* close datacollector */
492  log<picLog::INPUT_OUTPUT > ("HDF5 close DataCollector with file: %1%") % restartFilename;
493  mThreadParams.dataCollector->close();
494 
495  if (mThreadParams.dataCollector)
496  mThreadParams.dataCollector->finalize();
497 
498  __delete(mThreadParams.dataCollector);
499  }

534  void openH5File(const std::string h5Filename)
535  {
536  const uint32_t maxOpenFilesPerNode = 4;
537  if (mThreadParams.dataCollector == nullptr)
538  {
539  GridController<simDim> &gc = Environment<simDim>::get().GridController();
540  mThreadParams.dataCollector = new ParallelDomainCollector(
541  gc.getCommunicator().getMPIComm(),
542  gc.getCommunicator().getMPIInfo(),
543  splashMpiSize,
544  maxOpenFilesPerNode);
545  }
546  // set attributes for datacollector files
547  DataCollector::FileCreationAttr attr;
548  attr.enableCompression = false;
549  attr.fileAccType = DataCollector::FAT_CREATE;
550  attr.mpiPosition.set(splashMpiPos);
551  attr.mpiSize.set(splashMpiSize);
552 
553  // open datacollector

556  log<picLog::INPUT_OUTPUT > ("HDF5 open DataCollector with file: %1%") % h5Filename;
557  mThreadParams.dataCollector->open(h5Filename.c_str(), attr);

571  void dumpData(uint32_t currentStep)
572  {
573  const pmacc::Selection<simDim>& localDomain = Environment<simDim>::get().SubGrid().getLocalDomain();
574  mThreadParams.cellDescription = m_cellDescription;
575  mThreadParams.currentStep = currentStep;
576 
577  for (uint32_t i = 0; i < simDim; ++i)
578  {
579  mThreadParams.localWindowToDomainOffset[i] = 0;
580  if (mThreadParams.window.globalDimensions.offset[i] > localDomain.offset[i])
581  {
582  mThreadParams.localWindowToDomainOffset[i] =
583  mThreadParams.window.globalDimensions.offset[i] -
584  localDomain.offset[i];
585  }
586  }
587 
588  openH5File(mThreadParams.h5Filename);
589 
590  writeHDF5((void*) &mThreadParams);
591 
592  closeH5File();
593  }

647  void writeHDF5(void *p_args)
648  {
649  ThreadParams *threadParams = (ThreadParams*) (p_args);
650 
651  const SubGrid<simDim>& subGrid = Environment<simDim>::get().SubGrid();
652  DataSpace<simDim> domainOffset(
653  subGrid.getGlobalDomain().offset +
654  subGrid.getLocalDomain().offset
655  );
656 
657  std::vector< std::string > vectorOfDataSourceNames;
658  if( m_help->selfRegister )
659  {
660  std::string dateSourceNames = m_help->source.get( m_id );
661 
662  vectorOfDataSourceNames = plugins::misc::splitString(
663  plugins::misc::removeSpaces( dateSourceNames )
664  );
665  }
666 
667  /* write all fields */
668  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) writing fields.");
669  if (threadParams->isCheckpoint)
670  {
671  ForEach<FileCheckpointFields, WriteFields<bmpl::_1> > forEachWriteFields;
672  forEachWriteFields(threadParams);
673  }
674  else
675  {
676  bool dumpFields = plugins::misc::containsObject(
677  vectorOfDataSourceNames,
678  "fields_all"
679  );
680  if( dumpFields )
681  {
682  ForEach<
683  FileOutputFields,
684  WriteFields< bmpl::_1 >
685  > forEachWriteFields;
686  forEachWriteFields(threadParams);
687  }
688 
689  ForEach<
690  typename Help::AllFieldSources,
691  CallWriteFields<
692  bmpl::_1
693  >
694  >{}(
695  vectorOfDataSourceNames,
696  threadParams
697  );
698  }
699  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) writing fields.");
700 
701  /* write all particle species */
702  log<picLog::INPUT_OUTPUT > ("HDF5: (begin) writing particle species.");
703  if (threadParams->isCheckpoint)
704  {
705  ForEach<
706  FileCheckpointParticles,
707  WriteSpecies<
708  plugins::misc::UnfilteredSpecies< bmpl::_1 >
709  >
710  > writeSpecies;
711  writeSpecies(threadParams, domainOffset);
712  }
713  else
714  {
715  bool dumpAllParticles = plugins::misc::containsObject(
716  vectorOfDataSourceNames,
717  "species_all"
718  );
719 
720  if( dumpAllParticles )
721  {
722  ForEach<
723  FileOutputParticles,
724  WriteSpecies<
725  plugins::misc::UnfilteredSpecies< bmpl::_1 >
726  >
727  > writeSpecies;
728  writeSpecies(threadParams, domainOffset);
729  }
730 
731  ForEach<
732  typename Help::AllEligibleSpeciesSources,
733  CallWriteSpecies<
734  bmpl::_1
735  >
736  >{}(
737  vectorOfDataSourceNames,
738  threadParams,
739  domainOffset
740  );
741 
742  }
743  log<picLog::INPUT_OUTPUT > ("HDF5: ( end ) writing particle species.");
744 
745  auto idProviderState = IdProvider<simDim>::getState();
746  log<picLog::INPUT_OUTPUT>("HDF5: Writing IdProvider state (StartId: %1%, NextId: %2%, maxNumProc: %3%)")
747  % idProviderState.startId % idProviderState.nextId % idProviderState.maxNumProc;
748  WriteNDScalars<uint64_t, uint64_t>()(*threadParams,
749  "picongpu/idProvider/startId", idProviderState.startId,
750  "maxNumProc", idProviderState.maxNumProc);
751  WriteNDScalars<uint64_t>()(*threadParams,
752  "picongpu/idProvider/nextId", idProviderState.nextId);
753 
754  // write global meta attributes
755  WriteMeta writeMetaAttributes;
756  writeMetaAttributes(threadParams);
757  }
