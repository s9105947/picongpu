41 HINLINE float3_X laserLongitudinal(uint32_t currentStep, float_X& phase)
42 {
43  /* initialize the laser not in the first cell is equal to a negative shift
44  * in time
45  */
46  constexpr float_X laserTimeShift = laser::initPlaneY * CELL_HEIGHT / SPEED_OF_LIGHT;
47  const float_X runTime = DELTA_T * currentStep - laserTimeShift;
48  const float_X f = SPEED_OF_LIGHT / WAVE_LENGTH;
49 
50  float3_X elong(float3_X::create(0.0));
51 
52  // a symmetric pulse will be initialized at position z=0
53  // the laser amplitude rises for T_rise
54  // and falls for T_rise
55  // making the laser pulse 2*T_rise long
56 
57  const float_X T_rise = 0.5 * PULSE_LENGTH;
58  const float_X tau = runTime / T_rise;
59 
60  const float_X omegaLaser = 2.0 * PI * f;
61 
62  elong.x() = AMPLITUDE * Tpolynomial(tau)
63  * math::sin(omegaLaser * (runTime - T_rise) + LASER_PHASE);
64 
65  phase = 0.0f;
66 
67  return elong;
68 }

86 HDINLINE float_X Tpolynomial(const float_X tau)
87 {
88  if (tau >= 0.0 && tau <= 1.0)
89  return tau * tau * tau * (10.0 - 15.0 * tau + 6.0 * tau * tau);
90  else if (tau > 1.0 && tau <= 2.0)
91  return (2.0 - tau) * (2.0 - tau) * (2.0 - tau) * (4.0 - 9.0 * tau + 6.0 * tau * tau);
92  else
93  return 0.0;
94 }
