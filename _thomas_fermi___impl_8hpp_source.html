313  template< typename T_Acc >
314  DINLINE uint32_t numNewParticles(T_Acc const & acc, FrameType& ionFrame, int localIdx)
315  {
316  /* alias for the single macro-particle */
317  auto particle = ionFrame[localIdx];
318  /* particle position, used for field-to-particle interpolation */
319  floatD_X const pos = particle[position_];
320  int const particleCellIdx = particle[localCellIdx_];
321  /* multi-dim coordinate of the local cell inside the super cell */
322  DataSpace<SuperCellSize::dim> localCell(DataSpaceOperations<SuperCellSize::dim>::template map<SuperCellSize > (particleCellIdx));
323  /* interpolation of density */
324  const picongpu::traits::FieldPosition<typename fields::Solver::NummericalCellType, FieldTmp> fieldPosRho;
325  ValueType_Rho densityV = Field2ParticleInterpolation()
326  (cachedRho.shift(localCell).toCursor(), pos, fieldPosRho());
327  /* and energy density field on the particle position */
328  const picongpu::traits::FieldPosition<typename fields::Solver::NummericalCellType, FieldTmp> fieldPosEne;
329  ValueType_Ene kinEnergyV = Field2ParticleInterpolation()
330  (cachedEne.shift(localCell).toCursor(), pos, fieldPosEne());
331 
332  /* density in sim units */
333  float_X const density = densityV[0];
334  /* energy density in sim units */
335  float_X const kinEnergyDensity = kinEnergyV[0];
336 
337  /* Returns the new number of bound electrons for an integer number of macro electrons */
338  IonizationAlgorithm ionizeAlgo;
339  uint32_t newMacroElectrons = ionizeAlgo(
340  kinEnergyDensity,
341  density,
342  particle,
343  this->randomGen(acc)
344  );
345 
346 
347  return newMacroElectrons;

351  /* Functor implementation
352  *
353  * Ionization model specific particle creation
354  *
355  * \tparam T_parentIon type of ion species that is being ionized
356  * \tparam T_childElectron type of electron species that is created
357  * \param parentIon ion instance that is ionized
358  * \param childElectron electron instance that is created
359  */
360  template<typename T_parentIon, typename T_childElectron, typename T_Acc>
361  DINLINE void operator()(T_Acc const & acc, T_parentIon& parentIon,T_childElectron& childElectron)
362  {
363  /* for not mixing operations::assign up with the nvidia functor assign */
364  namespace partOp = pmacc::particles::operations;
365  /* each thread sets the multiMask hard on "particle" (=1) */
366  childElectron[multiMask_] = 1u;
367  const float_X weighting = parentIon[weighting_];
368 
369  /* each thread initializes a clone of the parent ion but leaving out
370  * some attributes:
371  * - multiMask: reading from global memory takes longer than just setting it again explicitly
372  * - momentum: because the electron would get a higher energy because of the ion mass
373  * - boundElectrons: because species other than ions or atoms do not have them
374  * (gets AUTOMATICALLY deselected because electrons do not have this attribute)
375  */
376  auto targetElectronClone = partOp::deselect<bmpl::vector2<multiMask, momentum> >(childElectron);
377 
378  partOp::assign(targetElectronClone, partOp::deselect<particleId>(parentIon));
379 
380  const float_X massIon = attribute::getMass(weighting,parentIon);
381  const float_X massElectron = attribute::getMass(weighting,childElectron);
382 
383  const float3_X electronMomentum (parentIon[momentum_]*(massElectron/massIon));
384 
385  childElectron[momentum_] = electronMomentum;
386 
387  /* conservation of momentum
388  * \todo add conservation of mass */
389  parentIon[momentum_] -= electronMomentum;
390 
397  parentIon[boundElectrons_] -= float_X(1.);
398  }
