template<class T_RNGMethod>
class pmacc::random::distributions::detail::Uniform< uniform::ExcludeOne< double >::Reduced, T_RNGMethod, void >

Returns a random double value uniformly distributed in [0,1)

Number of unique random numbers is reduced to 2^53. Uses a uniform distance of 2^-53 (epsilon/2) between each possible random number.
