152  virtual void pluginLoad()
153  {
154  //fill periodic with 0
155  while (periodic.size() < 3)
156  periodic.push_back(0);
157 
158  // check on correct number of devices. fill with default value 1 for missing dimensions
159  if (devices.size() > 3)
160  {
161  std::cerr << "Invalid number of devices.\nuse [-d dx=1 dy=1 dz=1]" << std::endl;
162  }
163  else
164  while (devices.size() < 3)
165  devices.push_back(1);
166 
167  // check on correct grid size. fill with default grid size value 1 for missing 3. dimension
168  if (gridSize.size() < 2 || gridSize.size() > 3)
169  {
170  std::cerr << "Invalid or missing grid size.\nuse -g width height [depth=1]" << std::endl;
171  }
172  else
173  if (gridSize.size() == 2)
174  gridSize.push_back(1);
175 
176  if (slidingWindow && devices[1] == 1)
177  {
178  std::cerr << "Invalid configuration. Can't use moving window with one device in Y direction" << std::endl;
179  }
180 
181  DataSpace<simDim> global_grid_size;
182  DataSpace<simDim> gpus;
183  DataSpace<simDim> isPeriodic;
184 
185  for (uint32_t i = 0; i < simDim; ++i)
186  {
187  global_grid_size[i] = gridSize[i];
188  gpus[i] = devices[i];
189  isPeriodic[i] = periodic[i];
190  }
191 
192  Environment<simDim>::get().initDevices(gpus, isPeriodic);
193  pmacc::GridController< simDim > & gc = pmacc::Environment<simDim>::get().GridController();
194 
195  DataSpace<simDim> myGPUpos(gc.getPosition());
196 
197  if( gc.getGlobalRank() == 0 )
198  {
199  if( showVersionOnce )
200  {
201  void( getSoftwareVersions( std::cout ) );
202  }
203  }
204 
205  // calculate the number of local grid cells and
206  // the local cell offset to the global box
207  for (uint32_t dim = 0; dim < gridDistribution.size() && dim < simDim; ++dim)
208  {
209  // parse string
210  ParserGridDistribution parserGD(gridDistribution.at(dim));
211 
212  // calculate local grid points & offset
213  gridSizeLocal[dim] = parserGD.getLocalSize(myGPUpos[dim]);
214  gridOffset[dim] = parserGD.getOffset(myGPUpos[dim], global_grid_size[dim]);
215  }
216  // by default: use an equal distributed box for all omitted params
217  for (uint32_t dim = gridDistribution.size(); dim < simDim; ++dim)
218  {
219  gridSizeLocal[dim] = global_grid_size[dim] / gpus[dim];
220  gridOffset[dim] = gridSizeLocal[dim] * myGPUpos[dim];
221  }
222 
223  Environment<simDim>::get().initGrids(global_grid_size, gridSizeLocal, gridOffset);
224 
225  MovingWindow::getInstance().setSlidingWindow(slidingWindow);
226 
227  log<picLog::DOMAINS > ("rank %1%; localsize %2%; localoffset %3%;") %
228  myGPUpos.toString() % gridSizeLocal.toString() % gridOffset.toString();
229 
230  SimulationHelper<simDim>::pluginLoad();
231 
232  GridLayout<SIMDIM> layout(gridSizeLocal, MappingDesc::SuperCellSize::toRT());
233  cellDescription = new MappingDesc(layout.getDataSpace(), GUARD_SIZE, GUARD_SIZE);
234 
235  checkGridConfiguration(global_grid_size, cellDescription->getGridLayout());
236 
237  if (gc.getGlobalRank() == 0)
238  {
239  if (slidingWindow)
240  log<picLog::PHYSICS > ("Sliding Window is ON");
241  else
242  log<picLog::PHYSICS > ("Sliding Window is OFF");
243  }
244 
245  for (uint32_t i = 0; i < simDim; ++i)
246  {
247 
248  /*
249  * absorber must be smaller than local gridsize if direction is periodic.
250  * absorber can't go over more than one device.
251  */
252  if (isPeriodic[i] == 0)
253  {
254  /*negativ direction*/
255  PMACC_VERIFY((int) ABSORBER_CELLS[i][0] <= (int) cellDescription->getGridLayout().getDataSpaceWithoutGuarding()[i]);
256  /*positiv direction*/
257  PMACC_VERIFY((int) ABSORBER_CELLS[i][1] <= (int) cellDescription->getGridLayout().getDataSpaceWithoutGuarding()[i]);
258  }
259  }
260  }

289  virtual void init()
290  {
291  namespace nvmem = pmacc::nvidia::memory;
292 
293  DataConnector &dc = Environment<>::get().DataConnector();
294 
295  // create simulation data such as fields and particles
296  auto fieldB = new FieldB( *cellDescription );
297  dc.share( std::shared_ptr< ISimulationData >( fieldB ) );
298  auto fieldE = new FieldE( *cellDescription );
299  dc.share( std::shared_ptr< ISimulationData >( fieldE ) );
300  auto fieldJ = new FieldJ( *cellDescription );
301  dc.share( std::shared_ptr< ISimulationData >( fieldJ ) );
302 
303  std::vector< FieldTmp * > fieldTmp;
304  for( uint32_t slot = 0; slot < fieldTmpNumSlots; ++slot)
305  {
306  auto newFld = new FieldTmp( *cellDescription, slot );
307  fieldTmp.push_back( newFld );
308  dc.share( std::shared_ptr< ISimulationData >( newFld ) );
309  }
310  pushBGField = new cellwiseOperation::CellwiseOperation < CORE + BORDER + GUARD > (*cellDescription);
311  currentBGField = new cellwiseOperation::CellwiseOperation < CORE + BORDER > (*cellDescription);
312 
313  // Initialize random number generator and synchrotron functions, if there are synchrotron or bremsstrahlung Photons
314  typedef typename pmacc::particles::traits::FilterByFlag<VectorAllSpecies,
315  synchrotronPhotons<> >::type AllSynchrotronPhotonsSpecies;
316  typedef typename pmacc::particles::traits::FilterByFlag<VectorAllSpecies,
317  bremsstrahlungPhotons<> >::type AllBremsstrahlungPhotonsSpecies;
318 
319  // create factory for the random number generator
320  using RNGFactory = pmacc::random::RNGProvider< simDim, pmacc::random::methods::XorMin< cupla::Acc> >;
321  auto rngFactory = new RNGFactory( Environment<simDim>::get().SubGrid().getLocalDomain().size );
322 
323  // init and share random number generator
324  pmacc::GridController<simDim>& gridCon = pmacc::Environment<simDim>::get().GridController();
325  rngFactory->init( gridCon.getScalarPosition() );
326  dc.share( std::shared_ptr< ISimulationData >( rngFactory ) );
327 
328  // Initialize synchrotron functions, if there are synchrotron photon species
329  if(!bmpl::empty<AllSynchrotronPhotonsSpecies>::value)
330  {
331  this->synchrotronFunctions.init();
332  }
333 #if( PMACC_CUDA_ENABLED == 1 )
334  // Initialize bremsstrahlung lookup tables, if there are species containing bremsstrahlung photons
335  if(!bmpl::empty<AllBremsstrahlungPhotonsSpecies>::value)
336  {
337  ForEach<
338  AllBremsstrahlungPhotonsSpecies,
339  particles::bremsstrahlung::FillScaledSpectrumMap< bmpl::_1 >
340  > fillScaledSpectrumMap;
341  fillScaledSpectrumMap(forward(this->scaledBremsstrahlungSpectrumMap));
342 
343  this->bremsstrahlungPhotonAngle.init();
344  }
345 
346  /* Create an empty allocator. This one is resized after all exchanges
347  * for particles are created */
348  deviceHeap.reset(new DeviceHeap(0));
349 #endif
350 
351  /* Allocate helper fields for FLYlite population kinetics for atomic physics
352  * (histograms, rate matrix, etc.)
353  */
354  using AllFlyLiteIons = typename pmacc::particles::traits::FilterByFlag<
355  VectorAllSpecies,
356  populationKinetics<>
357  >::type;
358 
359  ForEach<
360  AllFlyLiteIons,
361  particles::CallPopulationKineticsInit< bmpl::_1 >,
362  bmpl::_1
363  > initPopulationKinetics;
364  initPopulationKinetics(
365  gridSizeLocal
366  );
367 
368  // Allocate and initialize particle species with all left-over memory below
369  ForEach< VectorAllSpecies, particles::CreateSpecies<bmpl::_1> > createSpeciesMemory;
370  createSpeciesMemory( deviceHeap, cellDescription );
371 
372  size_t freeGpuMem(0);
373  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
374  if(freeGpuMem < reservedGpuMemorySize)
375  {
376  pmacc::log< picLog::MEMORY > ("%1% MiB free memory < %2% MiB required reserved memory")
377  % (freeGpuMem / 1024 / 1024) % (reservedGpuMemorySize / 1024 / 1024) ;
378  std::stringstream msg;
379  msg << "Cannot reserve "
380  << (reservedGpuMemorySize / 1024 / 1024) << " MiB as there is only "
381  << (freeGpuMem / 1024 / 1024) << " MiB free GPU memory left";
382  throw std::runtime_error(msg.str());
383  }
384 
385 #if( PMACC_CUDA_ENABLED == 1 )
386  size_t heapSize = freeGpuMem - reservedGpuMemorySize;
387 
388  if( Environment<>::get().MemoryInfo().isSharedMemoryPool() )
389  {
390  heapSize /= 2;
391  log<picLog::MEMORY > ("Shared RAM between GPU and host detected - using only half of the 'device' memory.");
392  }
393  else
394  log<picLog::MEMORY > ("RAM is NOT shared between GPU and host.");
395 
396  // initializing the heap for particles
397  deviceHeap->destructiveResize(heapSize);
398  MallocMCBuffer<DeviceHeap>* mallocMCBuffer = new MallocMCBuffer<DeviceHeap>(deviceHeap);
399  dc.share( std::shared_ptr< ISimulationData >( mallocMCBuffer ) );
400 #endif
401  ForEach< VectorAllSpecies, particles::LogMemoryStatisticsForSpecies<bmpl::_1> > logMemoryStatisticsForSpecies;
402  logMemoryStatisticsForSpecies( deviceHeap );
403 
404  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
405  log<picLog::MEMORY > ("free mem after all mem is allocated %1% MiB") % (freeGpuMem / 1024 / 1024);
406 
407  IdProvider<simDim>::init();
408 
409  // create field solver
410  this->myFieldSolver = new fields::Solver(*cellDescription);
411 
412  // create current interpolation
413  this->myCurrentInterpolation = new typename fields::Solver::CurrentInterpolation;
414 #if( PMACC_CUDA_ENABLED == 1 )
415  /* add CUDA streams to the StreamController for concurrent execution */
416  Environment<>::get().StreamController().addStreams(6);
417 #endif
418  }

420  virtual uint32_t fillSimulation()
421  {
422  /* assume start (restart in initialiserController might change that) */
423  uint32_t step = 0;
424 
425  /* set slideCounter properties for PIConGPU MovingWindow: assume start
426  * (restart in initialiserController might change this again)
427  */
428  MovingWindow::getInstance().setSlideCounter(0, 0);
429  /* Update MPI domain decomposition: will also update SubGrid domain
430  * information such as local offsets in y-direction
431  */
432  GridController<simDim> &gc = Environment<simDim>::get().GridController();
433  gc.setStateAfterSlides(0);
434 
435  DataConnector &dc = Environment<>::get().DataConnector();
436  auto fieldE = dc.get< FieldE >( FieldE::getName(), true );
437  auto fieldB = dc.get< FieldB >( FieldB::getName(), true );
438 
439  /* fill all objects registed in DataConnector */
440  if (initialiserController)
441  {
442  initialiserController->printInformation();
443  if (this->restartRequested)
444  {
445  /* we do not require --restart.step if a master checkpoint file is found */
446  if (this->restartStep < 0)
447  {
448  std::vector<uint32_t> checkpoints = readCheckpointMasterFile();
449 
450  if (checkpoints.empty())
451  {
452  throw std::runtime_error(
453  "Restart failed. You must provide the '--restart.step' argument. See picongpu --help."
454  );
455  } else
456  this->restartStep = checkpoints.back();
457  }
458 
459  initialiserController->restart((uint32_t)this->restartStep, this->restartDirectory);
460  step = this->restartStep;
461 
469  cellwiseOperation::CellwiseOperation< GUARD > guardBGField( *cellDescription );
470  namespace nvfct = pmacc::nvidia::functors;
471  guardBGField( fieldE, nvfct::Add(), FieldBackgroundE( fieldE->getUnit() ),
472  step, FieldBackgroundE::InfluenceParticlePusher );
473  guardBGField( fieldB, nvfct::Add(), FieldBackgroundB( fieldB->getUnit() ),
474  step, FieldBackgroundB::InfluenceParticlePusher );
475  }
476  else
477  {
478  initialiserController->init();
479  ForEach< particles::InitPipeline, particles::CallFunctor<bmpl::_1> > initSpecies;
480  initSpecies( step );
481  }
482  }
483 
484  size_t freeGpuMem(0u);
485  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
486  log<picLog::MEMORY > ("free mem after all particles are initialized %1% MiB") % (freeGpuMem / 1024 / 1024);
487 
488  // generate valid GUARDS (overwrite)
489  EventTask eRfieldE = fieldE->asyncCommunication(__getTransactionEvent());
490  __setTransactionEvent(eRfieldE);
491  EventTask eRfieldB = fieldB->asyncCommunication(__getTransactionEvent());
492  __setTransactionEvent(eRfieldB);
493 
494  dc.releaseData( FieldE::getName() );
495  dc.releaseData( FieldB::getName() );
496 
497  return step;
498  }

505  virtual void runOneStep(uint32_t currentStep)
506  {
507  namespace nvfct = pmacc::nvidia::functors;
508 
509  typedef typename pmacc::particles::traits::FilterByIdentifier
510  <
511  VectorAllSpecies,
512  momentumPrev1
513  >::type VectorSpeciesWithMementumPrev1;
514 
515  /* copy attribute momentum to momentumPrev1 */
516  ForEach<
517  VectorSpeciesWithMementumPrev1,
518  particles::Manipulate<
519  particles::manipulators::unary::CopyAttribute<
520  momentumPrev1,
521  momentum
522  >,
523  bmpl::_1
524  >
525  > copyMomentumPrev1;
526  copyMomentumPrev1( currentStep );
527 
528  DataConnector &dc = Environment<>::get().DataConnector();
529 
530  /* Initialize ionization routine for each species with the flag `ionizers<>` */
531  using VectorSpeciesWithIonizers = typename pmacc::particles::traits::FilterByFlag<
532  VectorAllSpecies,
533  ionizers<>
534  >::type;
535  ForEach< VectorSpeciesWithIonizers, particles::CallIonization< bmpl::_1 > > particleIonization;
536  particleIonization( cellDescription, currentStep );
537 
538  /* FLYlite population kinetics for atomic physics */
539  using AllFlyLiteIons = typename pmacc::particles::traits::FilterByFlag<
540  VectorAllSpecies,
541  populationKinetics<>
542  >::type;
543 
544  ForEach<
545  AllFlyLiteIons,
546  particles::CallPopulationKinetics< bmpl::_1 >,
547  bmpl::_1
548  > populationKinetics;
549  populationKinetics( currentStep );
550 
551  /* call the synchrotron radiation module for each radiating species (normally electrons) */
552  typedef typename pmacc::particles::traits::FilterByFlag<VectorAllSpecies,
553  synchrotronPhotons<> >::type AllSynchrotronPhotonsSpecies;
554 
555  ForEach<
556  AllSynchrotronPhotonsSpecies,
557  particles::CallSynchrotronPhotons< bmpl::_1 >
558  > synchrotronRadiation;
559  synchrotronRadiation( cellDescription, currentStep, this->synchrotronFunctions );
560 
561 #if( PMACC_CUDA_ENABLED == 1 )
562  /* Bremsstrahlung */
563  typedef typename pmacc::particles::traits::FilterByFlag
564  <
565  VectorAllSpecies,
566  bremsstrahlungIons<>
567  >::type VectorSpeciesWithBremsstrahlung;
568  ForEach<
569  VectorSpeciesWithBremsstrahlung,
570  particles::CallBremsstrahlung< bmpl::_1 >
571  > particleBremsstrahlung;
572  particleBremsstrahlung(
573  cellDescription,
574  currentStep,
575  this->scaledBremsstrahlungSpectrumMap,
576  this->bremsstrahlungPhotonAngle);
577 #endif
578  EventTask initEvent = __getTransactionEvent();
579  EventTask updateEvent;
580  EventTask commEvent;
581 
582  /* push all species */
583  particles::PushAllSpecies pushAllSpecies;
584  pushAllSpecies( currentStep, initEvent, updateEvent, commEvent );
585 
586  __setTransactionEvent(updateEvent);
588  auto fieldE = dc.get< FieldE >( FieldE::getName(), true );
589  auto fieldB = dc.get< FieldB >( FieldB::getName(), true );
590  (*pushBGField)(fieldE, nvfct::Sub(), FieldBackgroundE(fieldE->getUnit()),
591  currentStep, FieldBackgroundE::InfluenceParticlePusher);
592  (*pushBGField)(fieldB, nvfct::Sub(), FieldBackgroundB(fieldB->getUnit()),
593  currentStep, FieldBackgroundB::InfluenceParticlePusher);
594  dc.releaseData( FieldE::getName() );
595  dc.releaseData( FieldB::getName() );
596 
597  this->myFieldSolver->update_beforeCurrent(currentStep);
598 
599  auto fieldJ = dc.get< FieldJ >( FieldJ::getName(), true );
600  FieldJ::ValueType zeroJ( FieldJ::ValueType::create(0.) );
601  fieldJ->assign( zeroJ );
602 
603  __setTransactionEvent(commEvent);
604  (*currentBGField)(fieldJ, nvfct::Add(), FieldBackgroundJ(fieldJ->getUnit()),
605  currentStep, FieldBackgroundJ::activated);
606 #if (ENABLE_CURRENT == 1)
607  typedef typename pmacc::particles::traits::FilterByFlag
608  <
609  VectorAllSpecies,
610  current<>
611  >::type VectorSpeciesWithCurrentSolver;
612  ForEach<
613  VectorSpeciesWithCurrentSolver,
614  ComputeCurrent<
615  bmpl::_1,
616  bmpl::int_<CORE + BORDER>
617  >
618  > computeCurrent;
619  computeCurrent( currentStep );
620 #endif
621 
622 #if (ENABLE_CURRENT == 1)
623  if(bmpl::size<VectorSpeciesWithCurrentSolver>::type::value > 0)
624  {
625  EventTask eRecvCurrent = fieldJ->asyncCommunication(__getTransactionEvent());
626 
627  const DataSpace<simDim> currentRecvLower( GetMargin<typename fields::Solver::CurrentInterpolation>::LowerMargin( ).toRT( ) );
628  const DataSpace<simDim> currentRecvUpper( GetMargin<typename fields::Solver::CurrentInterpolation>::UpperMargin( ).toRT( ) );
629 
630  /* without interpolation, we do not need to access the FieldJ GUARD
631  * and can therefor overlap communication of GUARD->(ADD)BORDER & computation of CORE */
632  if( currentRecvLower == DataSpace<simDim>::create(0) &&
633  currentRecvUpper == DataSpace<simDim>::create(0) )
634  {
635  fieldJ->addCurrentToEMF<CORE >(*myCurrentInterpolation);
636  __setTransactionEvent(eRecvCurrent);
637  fieldJ->addCurrentToEMF<BORDER >(*myCurrentInterpolation);
638  } else
639  {
640  /* in case we perform a current interpolation/filter, we need
641  * to access the BORDER area from the CORE (and the GUARD area
642  * from the BORDER)
643  * `fieldJ->asyncCommunication` first adds the neighbors' values
