259  * bmpl::contains has lower complexity than traits::HasIdentifier
260  * and was used for this reason

262  typedef typename bmpl::copy_if<
263  DestTypeSeq,

268  /* create sequences with disjunct attributes from `DestTypeSeq` */

270  DestTypeSeq,

272  bmpl::back_inserter< bmpl::vector0<> >
273  >::type UniqueInDestTypeSeq;
