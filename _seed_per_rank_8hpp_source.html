40  template <unsigned T_DIM>
41  struct SeedPerRank
42  {
55  uint32_t
56  operator()( uint32_t localSeed )
57  {
58  auto& gc = pmacc::Environment<T_DIM>::get().GridController();
59 
60  uint32_t rank = gc.getGlobalRank( );
61  /* We put the rank into the upper bits to allow values which start
62  * from zero (e.g. cellIdxs, time steps) to be used as additional seed contributors
63  * Those would then write to the lower bits leaving the upper bits alone
64  * which still results in globally unique seeds
65  */
66  uint32_t globalUniqueSeed = reverseBits(rank);
67  /* localSeed often contains a counted number, so we rotate it by some bits to not "destroy"
68  * the counted rank that is already there. Also it is not reversed to get a different pattern
69  */
70  localSeed = (localSeed << 16) | (localSeed >> (sizeof(uint32_t) * CHAR_BIT - 16));
71  globalUniqueSeed ^= localSeed;
72  /* For any globally constant localSeed globalUniqueSeed is now guaranteed
73  * to be globally unique
74  */
75  return globalUniqueSeed;
76  }
77  };
