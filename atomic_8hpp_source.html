84  template<typename T_Type>
85  struct AtomicAllIncKepler<T_Type, true>
86  {
87  template< typename T_Acc, typename T_Hierarchy >
88  HDINLINE T_Type
89  operator()(const T_Acc& acc,T_Type* ptr, const T_Hierarchy& hierarchy)
90  {
91  /* Get a bitmask with 1 for each thread in the warp, that executes this */
92 #if(__CUDACC_VER_MAJOR__ >= 9)
93  const int mask = __activemask();
94 #else
95  const int mask = __ballot(1);
96 #endif
97  /* select the leader */
98  const int leader = __ffs(mask) - 1;
99  T_Type result;
100  const int laneId = getLaneId();
101  /* Get the start value for this warp */
102  if (laneId == leader)
103  result = ::alpaka::atomic::atomicOp<::alpaka::atomic::op::Add>(acc,ptr, static_cast<T_Type>(__popc(mask)), hierarchy);
104  result = warpBroadcast(result, leader);
105  /* Add offset per thread */
106  return result + static_cast<T_Type>(__popc(mask & ((1 << laneId) - 1)));
107  }
108  };
