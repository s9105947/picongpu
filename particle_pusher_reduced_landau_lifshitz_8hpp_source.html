34 /* This pusher uses the Lorentz force and a reduced
35  * Landau Lifshitz term to push particles based on the
36  * Runge Kutta solver 4th order. It takes into account
37  * the energy loss due to radiation.
38  *
39  * More details on this approach can be found in
40  * Marija Vranic's paper: Classical Radiation Reaction
41  * in Particle-In-Cell Simulations
42  * http://arxiv.org/abs/1502.02432
43  */
44 template<class Velocity, class Gamma>
45 struct Push
46 {
47  /* this is an optional extention for sub-sampling pushes that enables grid to particle interpolation
48  * for particle positions outside the super cell in one push
49  */
50  using LowerMargin = typename pmacc::math::CT::make_Int<simDim,1>::type;
51  using UpperMargin = typename pmacc::math::CT::make_Int<simDim,1>::type;
52 
53  template<
54  typename T_FunctorFieldE,
55  typename T_FunctorFieldB,
56  typename T_Particle,
57  typename T_Pos
58  >
59  HDINLINE void operator()(
60  const T_FunctorFieldB functorBField, /* at t=0 */
61  const T_FunctorFieldE functorEField, /* at t=0 */
62  T_Particle & particle,
63  T_Pos & pos, /* at t=0 */
64  const uint32_t
65  )
66  {
67  float_X const weighting = particle[ weighting_ ];
68  float_X const mass = attribute::getMass( weighting, particle );
69  float_X const charge = attribute::getCharge( weighting, particle );
70 
71  using TypeBFieldFunctor = T_FunctorFieldB;
72  using TypeEFieldFunctor = T_FunctorFieldE;
73  using TypePosition = position_pic::type;
74  using TypeMomentum = momentum::type;
75  using TypeMass = float_X;
76  using TypeCharge = float_X;
77  using TypeWeighting = weighting::type;
78 
79  TypeMomentum mom = particle[ momentum_ ];
80 
81  const float_X deltaT = DELTA_T;
82  const uint32_t dimMomentum = GetNComponents<TypeMomentum>::value;
83  // the transver data type adjust to 3D3V, 2D3V, 2D2V, ...
84  using VariableType = pmacc::math::Vector< picongpu::float_X, simDim + dimMomentum >;
85  VariableType var;
86 
87  // transfer position
88  for(uint32_t i=0; i<picongpu::simDim; ++i)
89  var[i] = pos[i];
90 
91  // transfer momentum
92  for(uint32_t i=0; i<dimMomentum; ++i)
93  var[simDim + i] = mom[i];
94 
95  using DiffEqType = DiffEquation<VariableType, float_X, TypeEFieldFunctor, TypeBFieldFunctor, TypePosition, TypeMomentum, TypeMass, TypeCharge, TypeWeighting, Velocity, Gamma>;
96  DiffEqType diffEq(functorEField, functorBField, mass, charge, weighting);
97 
98  VariableType varNew = pmacc::math::RungeKutta4()(diffEq, var, float_X(0.0), deltaT);
99 
100  // transfer position
101  for(uint32_t i=0; i<picongpu::simDim; ++i)
102  pos[i] = varNew[i];
103 
104  // transfer momentum
105  for(uint32_t i=0; i<dimMomentum; ++i)
106  mom[i] = varNew[simDim+i];
107 
108  particle[ momentum_ ] = mom;
109  }

111  template<typename T_Var, typename T_Time,
112  typename T_FieldEFunc, typename T_FieldBFunc,
113  typename T_Pos, typename T_Mom,
114  typename T_Mass, typename T_Charge, typename T_Weighting,
115  typename T_Velocity, typename T_Gamma>
116  struct DiffEquation
117  {
118 
119  // alias for types to follow coding guide line
120  using VariableType = T_Var;
121  using TimeType = T_Time;
122  using EFieldFuncType = T_FieldEFunc;
123  using BFieldFuncType = T_FieldBFunc;
124  using PositionType = T_Pos;
125  using MomentumType = T_Mom;
126  using MassType = T_Mass;
127  using ChargeType = T_Charge;
128  using WeightingType = T_Weighting;
129  using VelocityType = T_Velocity;
130  using GammaType = T_Gamma;
131 
132 
133  HDINLINE DiffEquation(EFieldFuncType funcE, BFieldFuncType funcB, MassType m, ChargeType q, WeightingType w)
134  : fieldEFunc(funcE), fieldBFunc(funcB), mass(m), charge(q), weighting(w)
135  { }
136 
137  HDINLINE VariableType operator()(TimeType time, VariableType var) const
138  {
139  PositionType pos;
140  PositionType posInterpolation;
141  MomentumType mom;
142  // transfer position
143  for(uint32_t i=0; i<picongpu::simDim; ++i)
144  {
145  posInterpolation[i] = var[i];
146  pos[i] = var[i] * cellSize[i];
147  }
148 
149  auto fieldE = fieldEFunc( posInterpolation,
150  picongpu::particles::interpolationMemoryPolicy::ShiftToValidRange() );
151  auto fieldB = fieldBFunc( posInterpolation,
152  picongpu::particles::interpolationMemoryPolicy::ShiftToValidRange() );
153 
154  // transfer momentum
155  const uint32_t dimMomentum = GetNComponents<MomentumType>::value;
156  for(uint32_t i=0; i<dimMomentum; ++i)
157  mom[i] = var[simDim+i];
158 
159  VelocityType velocityCalc;
160  GammaType gammaCalc;
161  const float_X c = SPEED_OF_LIGHT;
162  const float3_X velocity = velocityCalc(mom, mass);
163  const float_X gamma = gammaCalc(mom, mass);
164  const float_X conversionMomentum2Beta = 1.0 / (gamma * mass * c);
165 
166  const float_X c2 = c*c;
167  const float_X charge2 = charge*charge;
168  const float3_X beta = velocity / c;
169 
170  const float_X prefactorRR = 2./3. * charge2 * charge2 / (4.*PI*EPS0 * mass*mass * c2*c2);
171  const float3_X lorentz = fieldE + conversionMomentum2Beta * c * math::cross(mom, fieldB);
172  const float_X fieldETimesBeta = math::dot(fieldE, mom) * conversionMomentum2Beta;
173  const float3_X radReactionVec = c * (math::cross(fieldE, fieldB) +
174  c * conversionMomentum2Beta * math::cross(fieldB, math::cross(fieldB, mom)))
175  + conversionMomentum2Beta * fieldE * math::dot(mom, fieldE)
176  - gamma * gamma * conversionMomentum2Beta * (mom * (math::dot(lorentz, lorentz) - fieldETimesBeta*fieldETimesBeta));
177 
178  const float3_X diffMom = charge * lorentz + (prefactorRR / weighting) * radReactionVec;
179  const float3_X diffPos = velocity;
180 
181  VariableType returnVar;
182  for(uint32_t i=0; i<picongpu::simDim; ++i)
183  returnVar[i] = diffPos[i] / cellSize[i];
184 
185  for(uint32_t i=0; i<dimMomentum; ++i)
186  returnVar[simDim+i] = diffMom[i];
187 
188  return returnVar;
189  }
190 
191 
192  private:
193  EFieldFuncType fieldEFunc; /* functor E field interpolation */
194  BFieldFuncType fieldBFunc; /* functor B field interpolation */
195  MassType mass; /* mass of the macro particle */
196  ChargeType charge; /* charge of the macro particle */
197  WeightingType weighting; /* weighting of the macro particle */
198  };
